020/**
021 * This is the common super-class for different variants of the Lucas-Kanade
022 * matcher.
023 *  @author Wilhelm Burger
024 *  @version 2019/01/03
025 */
026public abstract class LucasKanadeMatcher {
027
028        /**
029         * Default parameters for the containing class and its sub-classes; 
030         * a (usually modified) instance of this class is passed to the constructor 
031         * of a non-abstract sub-class.
032         */
033        public static class Parameters {
034                /** Convergence limit */
035                public double tolerance = 0.00001;
036                /** Maximum number of iterations */
037                public int maxIterations = 100;
038                /** Set true to output debug information */
039                public boolean debug = false;
040                /** Set true to display the steepest-descent images */
041                public boolean showSteepestDescentImages = false;
042                /** Set true to display the Hessian matrices */
043                public boolean showHessians = false;
044        }
045        
046        final FloatProcessor I;                 // search image
047        final FloatProcessor R;                 // reference image
048        final Parameters params;                // parameter object
049        
050        final int wR, hR;                               // width/height of R
051        final double xc, yc;                    // center (origin) of R
052        
053        int iteration = -1;
054        
055        /**
056         * Creates a new Lucas-Kanade-type matcher.
057         * @param I the search image (of type {@link FloatProcessor}).
058         * @param R the reference image (of type {@link FloatProcessor})
059         * @param params a parameter object of type  {@link LucasKanadeMatcher.Parameters}.
060         */
061        protected LucasKanadeMatcher(FloatProcessor I, FloatProcessor R, Parameters params) {
062                this.I = I;     // search image
063                this.R = R;     // reference image
064                this.params = params;
065                wR = R.getWidth();
066                hR = R.getHeight();
067                xc = 0.5 * (wR - 1);
068                yc = 0.5 * (hR - 1);
069        }
070        
071        /**
072         * Calculates the transformation that maps the reference image {@code R} (centered
073         * around the origin) to the given quad Q.
074         * @param Q an arbitrary quad (should be inside the search image I);
075         * @return the transformation from {@code R}'s bounding rectangle to {@code Q}.
076         */
077        public ProjectiveMapping2D getReferenceMappingTo(Point[] Q) {
078                Point[] Rpts = getReferencePoints();
079                return ProjectiveMapping2D.fromPoints(Rpts, Q);
080        }
081        
082        /**
083         * @return the corner points of the bounding rectangle of R, centered at the origin.
084         */
085        public Point[] getReferencePoints() {
086                double xmin = -xc;
087                double xmax = -xc + wR - 1;
088                double ymin = -yc;
089                double ymax = -yc + hR - 1;
090                Point[] pts = new Point[4];
091                pts[0] = Point.create(xmin, ymin);
092                pts[1] = Point.create(xmax, ymin);
093                pts[2] = Point.create(xmax, ymax);
094                pts[3] = Point.create(xmin, ymax);
095                return pts;
096        }
097        
098        
099        /**
100         * Performs the full optimization on the given image pair (I, R).
101         * @param Tinit the transformation from the reference image R to
102         * the initial search patch, assuming that R is centered at the coordinate
103         * origin!
104         * @return the transformation to the best-matching patch in the search image I
105         * (again assuming that R is centered at the coordinate origin) or null if
106         * no match was found.
107         */
108        public ProjectiveMapping2D getMatch(ProjectiveMapping2D Tinit) {
109//              initializeMatch(Tinit);                 // to be implemented by sub-classes
110                // ProjectiveMappingP Tp = Tinit.duplicate();
111                ProjectiveMapping2D Tp = Tinit;
112                do {
113                        Tp = iterateOnce(Tp);           // to be implemented by sub-classes
114                } while (Tp != null && !hasConverged() && getIteration() < params.maxIterations);
115                return Tp;
116        }
117        
118        
119        /**
120         * Performs a single matching iteration on the given image pair (I, R).
121         * 
122         * @param Tp the warp transformation from the reference image R to
123         * the initial search patch, assuming that R is centered at the coordinate
124         * origin! 
125         * @return a new warp transformation (again assuming that R is centered at 
126         * the coordinate origin) or null if the iteration was unsuccessful.
127         */
128        public abstract ProjectiveMapping2D iterateOnce(ProjectiveMapping2D Tp);
129        
130        
131        /**
132         * Checks if the matcher has converged.
133         * @return true if minimization criteria have been reached.
134         */
135        public abstract boolean hasConverged();
136        
137        /**
138         * Measures the RMS intensity difference between the reference image R and the 
139         * patch in the search image I defined by the current warp Tp.
140         * @return the RMS error under the current warp
141         */
142        public abstract double getRmsError();           
143        
144        public int getIteration() {
145                return iteration;
146        }
147        
148        // ------------------------------------------------------------------------------------
149        
150        protected FloatProcessor gradientX(FloatProcessor fp) {
151                // Sobel-kernel for x-derivatives:
152            final float[] Hx = Matrix.multiply(1f/8, new float[] {
153                                -1, 0, 1,
154                            -2, 0, 2,
155                            -1, 0, 1
156                            });
157            FloatProcessor fpX = (FloatProcessor) fp.duplicate();
158            fpX.convolve(Hx, 3, 3);
159            return fpX;
160        }
161        
162        protected FloatProcessor gradientY(FloatProcessor fp) {
163                // Sobel-kernel for y-derivatives:
164                final float[] Hy = Matrix.multiply(1f/8, new float[] {
165                                                -1, -2, -1,
166                                                 0,  0,  0,
167                                                 1,  2,  1
168                                                 });
169            FloatProcessor fpY = (FloatProcessor) fp.duplicate();
170            fpY.convolve(Hy, 3, 3);
171            return fpY;
172        }

200        protected void showSteepestDescentImages(double[][][] S) {      // S[u][v][n]
201                String titlePrefix = "sd";
202                int w = S.length;
203                int h = S[0].length;
204                int n = S[0][0].length;
205                for (int i = 0; i < n; i++) {
206                        FloatProcessor sdip = new FloatProcessor(w, h);
207                        for (int u = 0; u < w; u++) {
208                                for (int v = 0; v < h; v++) {
209                                        sdip.setf(u, v, (float) S[u][v][i]);
210                                }
211                        }
212                        (new ImagePlus(titlePrefix + i, sdip)).show();
213                }
214        }

218        public double[] getParameters(ProjectiveMapping2D map) {
219                double[][] A = map.getTransformationMatrix();
220                return new double[] {
221                                A[0][0]-1, A[0][1], A[1][0], A[1][1]-1, A[2][0], A[2][1], A[0][2], A[1][2]};
222                //return new double[] { a00 - 1, a01, a10, a11 - 1, a20, a21, a02, a12 };
223        }
224        
225        public ProjectiveMapping2D toProjectiveMap(double[] param) {
226                if (param.length < 8) {
227                        throw new IllegalArgumentException("Affine mapping requires 8 parameters");
228                }
229                return new ProjectiveMapping2D(
230                                param[0] + 1,   param[1],        param[6],
231                                param[2],       param[3] + 1,    param[7],
232                                param[4],       param[5]             );
233        }

237        public double[] getParameters(AffineMapping2D map) {
238                double[][] A = map.getTransformationMatrix();
239                return new double[] { 
240                                A[0][0] - 1, A[0][1], A[1][0], A[1][1]-1, A[0][2], A[1][2]};
241                //return new double[] { a00 - 1, a01, a10, a11 - 1, a02, a12 };
242        }
243        
244        public AffineMapping2D toAffineMap(double[] param) {
245                if (param.length < 6) {
246                        throw new IllegalArgumentException("Affine mapping requires 6 parameters");
247                }
248                return new AffineMapping2D(
249                                param[0] + 1, param[1],     param[4],
250                                param[2],     param[3] + 1, param[5]);
251        }
252        
253        // --------------------
254        
255        public double[] getParameters(Translation2D map) {
256                double[][] A = map.getTransformationMatrix();
257//              double[] p = new double[] {a02, a12};
258                return new double[] {A[0][2], A[1][2]};
259        }
260        
261        public Translation2D toTranslation(double[] p) {
262                if (p.length < 2) {
263                        throw new IllegalArgumentException("Translation requires 2 parameters");
264                }
265                return new Translation2D(p[0], p[1]);
266        }
