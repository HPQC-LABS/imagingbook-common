005/**
006 * Direct (slow) implementation of the 1-dimensional DFT
007 * using tabulated sine and cosine values.
008 * @author WB
009 */
010public abstract class Dft1dDirect {
011        
012        final int M;                    // size (length) of the signal or spectrum
013        final ScalingMode sm;   
014        final double[] cosTable;
015        final double[] sinTable;
016        
017        private Dft1dDirect(int M, ScalingMode sm) {
018                this.M = M;
019                this.sm = sm;
020                this.cosTable = makeCosTable();
021                this.sinTable = makeSinTable();
022        }
023        
024        private double[] makeCosTable() {
025                double[] cosTable = new double[M];
026                double theta = 2 * Math.PI / M;
027                for (int i = 0; i < M; i++) {
028                        cosTable[i] = Math.cos(theta * i);
029                }
030                return cosTable;
031        }
032
033        private double[] makeSinTable() {
034                double[] sinTable = new double[M];
035                double theta = 2 * Math.PI / M;
036                for (int i = 0; i < M; i++) {
037                        sinTable[i] = Math.sin(theta * i);
038                }
039                return sinTable;
040        }
041        
042        // ----------------------------------------------------------------------
043        
044        /**
045         * One-dimensional DFT implementation using single precision (float) data. 
046         */
047        public static class Float extends Dft1dDirect implements Dft1d.Float {
048                
049                private final float[] outRe;
050                private final float[] outIm;
051                
052                public Float(int M, ScalingMode sm) {
053                        super(M, sm);
054                        this.outRe = new float[M];
055                        this.outIm = new float[M];
056                }
057                
058                public Float(int M) {
059                        this(M, ScalingMode.DEFAULT);
060                }
061                
062                @Override
063                public void forward(float[] gRe, float[] gIm) {
064                        checkSize(gRe, gIm, M);
065                        transform(gRe, gIm, true);
066                }
067                
068                @Override
069                public void inverse(float[] GRe, float[] GIm) {
070                        checkSize(GRe, GIm, M);
071                        transform(GRe, GIm, false);
072                }
073                
074                public void transform(float[] inRe, float[] inIm, boolean forward) {
075                        checkSize(inRe, inIm, M);
076                        final double scale = sm.getScale(M, forward);
077                        for (int u = 0; u < M; u++) {
078                                double sumRe = 0;
079                                double sumIm = 0;
080                                for (int m = 0; m < M; m++) {
081                                        final double re = inRe[m];
082                                        final double im = (inIm == null) ? 0 : inIm[m];
083                                        final int k = (u * m) % M;
084                                        final double cosPhi = cosTable[k];
085                                        final double sinPhi = (forward) ? -sinTable[k] : sinTable[k];
086                                        // complex multiplication: (gRe + i gIm) * (cosPhi + i sinPhi)
087                                        sumRe += re * cosPhi - im * sinPhi;
088                                        sumIm += re * sinPhi + im * cosPhi;
089                                }
090                                outRe[u] = (float) (scale * sumRe);     
091                                outIm[u] = (float) (scale * sumIm);
092                        }
093                        System.arraycopy(outRe, 0, inRe, 0, M);
094                        System.arraycopy(outIm, 0, inIm, 0, M);
095                }
096        }
