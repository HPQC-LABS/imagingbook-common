008/**
009 * FFT (fast) implementation of the DFT.
010 */
011public abstract class Dft1dFast {
012        
013        final int M;                    // size (length) of the signal or spectrum
014        final ScalingMode sm;   
015        
016        private Dft1dFast(int M, ScalingMode sm) {
017                this.M = M;
018                this.sm = sm;
019        }
020        
021        // ----------------------------------------------------------------------
022        
023        /**
024         * One-dimensional FFT implementation using single precision (float) data. 
025         */
026        public static class Float extends Dft1dFast implements Dft1d.Float {
027        
028                private final float[] A;                // temporary array for FFT composed of re/im values
029                private final FloatFFT_1D fft;
030                
031                public Float(int M, ScalingMode sm) {
032                        super(M, sm);
033                        this.A = new float[2 * M];
034                        this.fft = new FloatFFT_1D(M);
035                }
036                
037                public Float(int M) {
038                        this(M, ScalingMode.DEFAULT);
039                }
040                
041                @Override
042                public void forward(float[] gRe, float[] gIm) {
043                        checkSize(gRe, gIm, M);
044//                      float scale = (float) sm.getScale(M, true);
045//                      setupA(gRe, gIm, A);    
046//                      fft.complexForward(A);
047//                      decompA(A, gRe, gIm, scale);
048                        transform(gRe, gIm, true);
049                }
050                
051                @Override
052                public void inverse(float[] GRe, float[] GIm) {
053                        checkSize(GRe, GIm, M);
054//                      float scale = (float) sm.getScale(M, false);
055//                      setupA(GRe, GIm, A);    
056//                      fft.complexInverse(A, false);
057//                      decompA(A, GRe, GIm, scale);
058                        transform(GRe, GIm, false);
059                }
060                
061                public void transform(float[] inRe, float[] inIm, boolean forward) {
062                        checkSize(inRe, inIm, M);
063                        final float scale = (float) sm.getScale(M, forward);
064                        setupA(inRe, inIm, A);  
065                        if (forward)
066                                fft.complexForward(A);
067                        else
068                                fft.complexInverse(A, false);
069                        decompA(A, inRe, inIm, scale);
070                }
071                
072                // (re, im) -> A
073                private void setupA(float[] re, float[] im, float[] A) {
074                        for (int i = 0; i < M; i++) {
075                                A[2*i] = re[i];
076                                A[2*i + 1] = im[i];
077                        }
078                }
079                
080                // A -> (re, im)
081                private void decompA(float[] A, float[] re, float[] im, float scale) {
082                        for (int i = 0; i < M; i++) {
083                                re[i] = A[2*i] * scale;
084                                im[i] = A[2*i + 1] * scale;
085                        }
086                }
087        }
088

162         * This DFT uses the same definition as Mathematica. Example:
163         * Fourier[{1, 2, 3, 4, 5, 6, 7, 8}, FourierParameters -> {0, -1}]:
164                {12.7279 + 0. i, 
165                -1.41421 + 3.41421 i, 
166                -1.41421 + 1.41421 i, 
167                -1.41421 + 0.585786 i, 
168                -1.41421 + 0. i, 
169                -1.41421 - 0.585786 i, 
170                -1.41421 - 1.41421 i, 
171                -1.41421 - 3.41421 i}
172         */
