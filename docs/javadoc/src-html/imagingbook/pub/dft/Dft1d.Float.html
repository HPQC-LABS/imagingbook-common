001package imagingbook.pub.dft;
002
003/**
004 * Interface specifying all one-dimensional DFT/FFT implementations.
005 */
006public interface Dft1d {
007        
008        public interface Float extends Dft1d {
009                /**
010                 * Performs an "in-place" 1D DFT forward transformation on the supplied data.
011                 * The input signal is replaced by the associated DFT spectrum.
012                 * @param gRe real part of the signal (modified)
013                 * @param gIm imaginary part of the signal (modified)
014                 */
015                void forward(float[] gRe, float[] gIm);
016                
017                /**
018                 * Performs an "in-place" 1D DFT inverse transformation on the supplied spectrum.
019                 * The input spectrum is replaced by the associated signal.
020                 * @param GRe real part of the spectrum (modified)
021                 * @param GIm imaginary part of the spectrum (modified)
022                 */
023                void inverse(float[] GRe, float[] GIm);
024                
025                /**
026                 * Transforms the given 1D arrays 'in-place'. Separate arrays of identical size
027                 * must be supplied for the real and imaginary parts of the signal (forward) 
028                 * or spectrum (inverse), neither of which may be {@code null}.
029                 * 
030                 * @param inRe real part of the input signal or spectrum (modified)
031                 * @param inIm imaginary part of the input signal or spectrum (modified)
032                 * @param forward forward transformation if {@code true}, inverse transformation if {@code false}
033                 */
034                void transform(float[] inRe, float[] inIm, boolean forward);
035                
036                default void checkSize(float[] re, float[] im, int n) {
037                        if (re.length != n || im.length != n)
038                                throw new IllegalArgumentException("arrays for real/imagingary parts must be of size " + n);
039                }
040                
041        }
042        
043        public interface Double extends Dft1d {
044                
045                /**
046                 * Performs an "in-place" 1D DFT forward transformation on the supplied data.
047                 * The input signal is replaced by the associated DFT spectrum.
048                 * @param gRe real part of the signal (modified)
049                 * @param gIm imaginary part of the signal (modified)
050                 */
051                void forward(double[] gRe, double[] gIm);
052                
053                /**
054                 * Performs an "in-place" 1D DFT inverse transformation on the supplied spectrum.
055                 * The input spectrum is replaced by the associated signal.
056                 * @param GRe real part of the spectrum (modified)
057                 * @param GIm imaginary part of the spectrum (modified)
058                 */
059                void inverse(double[] GRe, double[] GIm);
060                
061                /**
062                 * Transforms the given 1D arrays 'in-place'. Separate arrays of identical size
063                 * must be supplied for the real and imaginary parts of the signal (forward) 
064                 * or spectrum (inverse), neither of which may be {@code null}.
065                 * 
066                 * @param inRe real part of the input signal or spectrum (modified)
067                 * @param inIm imaginary part of the input signal or spectrum (modified)
068                 * @param forward forward transformation if {@code true}, inverse transformation if {@code false}
069                 */
070                void transform(double[] inRe, double[] inIm, boolean forward);
071                
072                default void checkSize(double[] re, double[] im, int n) {
073                        if (re.length != n || im.length != n)
074                                throw new IllegalArgumentException("arrays for real/imagingary parts must be of size " + n);
075                }
076        }
077        
078}
