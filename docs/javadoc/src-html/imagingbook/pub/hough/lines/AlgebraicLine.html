009package imagingbook.pub.hough.lines;

011import java.awt.geom.Point2D;
012import java.util.Locale;
013
014import ij.process.ImageProcessor;
015
016/**
017 * This class represents an algebraic line of the form a * x + b * y + c = 0.
018 * All instances are normalized such that ||(a,b)|| = 1.
019 * Instances of this class are immutable.
020 * @author WB
021 *
022 */
023public class AlgebraicLine {
024        
025        protected final double a;
026        protected final double b;
027        protected final double c;
028        
029        // static factory methods ----------------------------------------
030        
031        public static AlgebraicLine create(double a, double b, double c) {
032                double norm = Math.sqrt(a * a + b * b);
033                return new AlgebraicLine(a / norm, b / norm , c / norm);
034        }
035        
036        public static AlgebraicLine create(Point2D p1, Point2D p2) {
037                double a = p1.getY() - p2.getY();
038                double b = p2.getX() - p1.getX();
039                double c = -a * p1.getX() - b * p1.getY();
040                return create(a, b, c);
041        }
042        
043        // constructors --------------------------------------------------
044        
045        protected AlgebraicLine(double a, double b, double c) {
046                this.a = a;
047                this.b = b;
048                this.c = c;
049        }
050        
051        // getter/setter methods ------------------------------------------
052        
053        public final double getA() {
054                return a;
055        }
056
057        public final double getB() {
058                return b;
059        }
060
061        public final double getC() {
062                return c;
063        }
064
065        // other methods ------------------------------------------
066        
067        /**
068         * Returns the perpendicular distance between this line and the point (x, y).
069         * The result may be positive or negative, depending on which side of the line
070         * (x, y) is located. It is assumed that the line is normalized
071         * 
072         * @param x x-coordinate of point position.
073         * @param y y-coordinate of point position.
074         * @return The perpendicular distance between this line and the point (x, y).
075         */
076        public double getDistance(double x, double y) {
077                return (a * x + b * y + c) / Math.sqrt(a * a + b * b);
078        }
079        
080        
081        /**
082         * Returns the perpendicular distance between this line and the point p. The
083         * result may be positive or negative, depending on which side of the line p is
084         * located.
085         * 
086         * @param p point position.
087         * @return The perpendicular distance between this line and the point p.
088         */
089        public double getDistance(Point2D p) {
090                return getDistance(p.getX(), p.getY());
091        }
092        
093        @Override
094        public String toString() {
095                return String.format(Locale.US, "%s <a = %.3f, b = %.3f, c = %.3f>",
096                                this.getClass().getSimpleName(), a, b, c);
097        }
098        
099        /**
100         * This is a brute-force drawing method which simply marks all image pixels that
101         * are sufficiently close to the HoughLine hl. The drawing color for ip must be
102         * previously set.
103         * 
104         * @param ip        The ImageProcessor to draw to.
105         * @param thickness The thickness of the lines to be drawn.
106         */
107        public void draw(ImageProcessor ip, double thickness) {
108                final int w = ip.getWidth();
109                final int h = ip.getHeight();
110                final double dmax = 0.5 * thickness;
111                for (int u = 0; u < w; u++) {
112                        for (int v = 0; v < h; v++) {
113                                // get the distance between (u,v) and the line hl:
114                                double d = Math.abs(this.getDistance(u, v));
115                                if (d <= dmax) {
116                                        ip.drawPixel(u, v);
117                                }
118                        }
119                }
120
121        }
