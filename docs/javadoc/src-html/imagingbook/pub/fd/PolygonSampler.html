010package imagingbook.pub.fd;
011
012import imagingbook.pub.geometry.basic.Point;
013
014/**
015 * @version 2020/04/01
016 * @author WB
017 *
018 */
019public class PolygonSampler {
020        
021        public PolygonSampler() {
022        }
023        
024        /**
025         * Samples the closed polygon path specified by V at M
026         * equi-distant positions. 
027         * @param V the vertices of the (closed) polygon.
028         * @param M the number of sample points.
029         * @return the sample points as an array of Point objects.
030         */
031        public Point[] samplePolygonUniformly(Point[] V, int M) {
032                int N = V.length;
033                double Delta = pathLength(V) / M;       // constant segment length in Q
034                // distribute N points along polygon path P
035                Point[] S = new Point[M];
036//              S[0] = (Point) V[0].clone();    // q_0 = p_0 (duplicate p_0)
037                S[0] = Point.create(V[0]);              // q_0 = p_0 (duplicate p_0)
038                int i = 0;                      // lower index of segment (i,i+1) in P
039                int j = 1;                      // index of next point to be added to Q
040                double alpha = 0;       // lower boundary of current path segment in P
041                double beta = Delta;    // path position of next point to be added to Q
042                // for all M segments in P do:
043                while (i < N && j < M) {
044                        Point vA = V[i];
045                        Point vB = V[(i + 1) % N];
046                        double delta = vA.distance(vB);
047                        // handle segment (i,i+1) with path boundaries (a,a+d), knowing a < b
048                        while (beta <= alpha + delta && j < M) {
049                                // a < b <= a+d
050                                S[j] = interpolate(vA, vB, (beta - alpha) / delta);
051                                j = j + 1;
052                                beta = beta + Delta;
053                        }
054                        alpha = alpha + delta;
055                        i = i + 1;
056                }       
057                return S;
058        }
059        
060        /**
061         * This is for testing: allows to choose an arbitrary start point by
062         * setting 'startFrac' in [0,1].
063         * @param V the vertices of the (closed) polygon.
064         * @param M the number of sample points.
065         * @param startFrac the position of the first sample as a fraction of the 
066         * polggon's circumference in [0,1].
067         * @return the sample points as an array of Point objects.
068         */
069        public Point[] samplePolygonUniformly(Point[] V, int M, double startFrac) {
070                int startPos = (int) Math.round(V.length * startFrac) % V.length;
071                return samplePolygonUniformly(shiftLeft(V, startPos), M);
072        }
073        
074        private Point[] shiftLeft(Point[] V, int startPos) {
075                int polyLen = V.length;
076                Point[] P2 = new Point[polyLen]; 
077                for (int i = 0; i < P2.length; i++) {
078//                      P2[i] = (Point) V[(i + startPos) % polyLen].clone();
079                        P2[i] = Point.create(V[(i + startPos) % polyLen]);
080                }
081                return P2;
082        }
083        
084        
085        protected double pathLength(Point[] V) {
086                double L = 0;
087                final int N = V.length;
088                for (int i = 0; i < N; i++) {
089                        Point p0 = V[i];
090                        Point p1 = V[(i + 1) % N];
091                        L = L + p0.distance(p1);
092                }
093                return L;
094        }
095        
096        protected Point interpolate(Point p0, Point p1, double alpha) {
097                // alpha is in [0,1]
098                double x = p0.getX() + alpha * (p1.getX() - p0.getX());
099                double y = p0.getY() + alpha * (p1.getY() - p0.getY());
100                return Point.create(x, y);
101        }
102        
103}
