010package imagingbook.pub.dct;
011
012
013import org.jtransforms.dct.DoubleDCT_1D;
014import org.jtransforms.dct.FloatDCT_1D;
015
016import imagingbook.lib.math.Matrix;
017
018/**
019 * Fast implementation of the DCT, based on the JTransforms package
020 * by Piotr Wendykier (see <a href="https://github.com/wendykierp/JTransforms">
021 * https://github.com/wendykierp/JTransforms</a>).
022 */
023public abstract class Dct1dFast {
024
025
026        final int M;                            // size of the input vector
027        final double s;                         // common scale factor
028
029        private Dct1dFast(int M) {
030                this.M = M;
031                this.s = Math.sqrt(2.0 / M); 
032        }
033        
034        // ------------------------------------------------------------------------------
035        
036        public static class Float extends Dct1dFast implements Dct1d.Float {
037                
038                private final FloatDCT_1D fct;
039                
040                public Float(int M) {
041                        super(M);
042                        this.fct = new FloatDCT_1D(M);
043                }
044
045                @Override
046                public void forward(float[] g) {
047                        checkLength(g, M);
048                        fct.forward(g, true);
049                }
050
051                @Override
052                public void inverse(float[] G) {
053                        checkLength(G, M);
054                        fct.inverse(G, true);
055                } 
056                
057        }
058
059        // ------------------------------------------------------------------------------
060        
061        public static class Double extends Dct1dFast implements Dct1d.Double {
062                
063                private final DoubleDCT_1D fct;
064                
065                public Double(int M) {
066                        super(M);
067                        this.fct = new DoubleDCT_1D(M);
068                }
069
070                @Override
071                public void forward(double[] g) {
072                        checkLength(g, M);
073                        fct.forward(g, true);
074                }
075
076                @Override
077                public void inverse(double[] G) {
078                        checkLength(G, M);
079                        fct.inverse(G, true);
080                }
081        }
082
083        // -----------------------------------------------------------------------------
084
085        // test example
086        public static void main(String[] args) {
087                {
088                        System.out.println("FLOAT (fast DCT) test:");
089                        float[] data = {1,2,3,4,5,3,0};
090                        System.out.println("Original data:      " + Matrix.toString(data));
091        
092                        Dct1d.Float dct = new Dct1dFast.Float(data.length);
093                        dct.forward(data);
094                        System.out.println("DCT spectrum:       " + Matrix.toString(data));
095        
096                        dct.inverse(data);
097                        System.out.println("Reconstructed data: " + Matrix.toString(data));
098                        System.out.println();
099                }
100                {
101                        System.out.println("DOUBLE (fast DCT) test:");
102                        double[] data = {1,2,3,4,5,3,0};
103                        System.out.println("Original data:      " + Matrix.toString(data));
104        
105                        Dct1d.Double dct = new Dct1dFast.Double(data.length);
106                        dct.forward(data);
107                        System.out.println("DCT spectrum:       " + Matrix.toString(data));
108        
109                        dct.inverse(data);
110                        System.out.println("Reconstructed data: " + Matrix.toString(data));
111                        System.out.println();
112                }
113        }
114
115        //      Original data:      {1.000, 2.000, 3.000, 4.000, 5.000, 3.000, 0.000}
116        //      DCT of data:        {6.803, -0.361, -3.728, 1.692, -0.888, -0.083, 0.167}
117        //      Reconstructed data: {1.000, 2.000, 3.000, 4.000, 5.000, 3.000, -0.000}
118
119
120}
