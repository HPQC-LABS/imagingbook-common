010package imagingbook.pub.dct;
011
012/**
013 * This class provides the functionality for calculating the DCT in 2D.
014 * @author W. Burger
015 * @version 2019-12-26
016 */
017public abstract class Dct2d {
018        
019        boolean useFastMode = true;
020        
021        public void useFastMode(boolean yesOrNo) {
022                this.useFastMode = yesOrNo;
023        }
024
025        private Dct2d() {
026        }
027        
028        // ---------------------------------------------------------------------------------------
029
030        public static class Float extends Dct2d {
031                
032                public Float() {
033                }
034
035                /**
036                 * Performs an "in-place" 2D DCT forward transformation on the supplied data.
037                 * The input signal is replaced by the associated DCT spectrum.
038                 * @param g the signal to be transformed (modified)
039                 */
040                public void forward(float[][] g) {
041                        transform(g, true);
042                }
043
044                /**
045                 * Performs an "in-place" 2D DCT inverse transformation on the supplied spectrum.
046                 * The input spectrum is replaced by the associated signal.
047                 * @param G the spectrum to be transformed (modified)
048                 */
049                public void inverse(float[][] G) {
050                        transform(G, false);
051                }
052
053                // in-place 2D DCT
054                private void transform(final float[][] data, boolean forward) {
055                        final int width = data.length;
056                        final int height = data[0].length;
057
058                        // do the rows:
059                        float[] row = new float[width];
060                        Dct1d.Float dct1R = 
061                                        useFastMode ? new Dct1dFast.Float(width) : new Dct1dDirect.Float(width);
062                        for (int v = 0; v < height; v++) {
063                                //IJ.showProgress(v, height);
064                                extractRow(data, v, row);
065                                if (forward)
066                                        dct1R.forward(row);
067                                else
068                                        dct1R.inverse(row);
069                                insertRow(data, v, row);
070                        }
071
072                        // do the columns:
073                        float[] col = new float[height];
074                        Dct1d.Float dct1C = 
075                                        useFastMode ? new Dct1dFast.Float(height) : new Dct1dDirect.Float(height);
076                        for (int u = 0; u < width; u++) {
077                                //IJ.showProgress(u, width);
078                                extractCol(data, u, col);
079                                if (forward)
080                                        dct1C.forward(col);
081                                else
082                                        dct1C.inverse(col);
083                                insertCol(data, u, col);
084                        }
085                }
086
087                private void extractRow(float[][] data, int v, float[] row) {
088                        final int width = data.length;
089                        for (int u = 0; u < width; u++) {
090                                row[u] = data[u][v];
091                        }
092                }
093
094                private void insertRow(float[][] data, int v, float[] row) {
095                        final int width = data.length;
096                        for (int u = 0; u < width; u++) {
097                                data[u][v] = (float) row[u];
098                        }
099                }
100
101                private void extractCol(float[][] data, int u, float[] column) {
102                        final int height = data[0].length;
103                        for (int v = 0; v < height; v++) {
104                                column[v] = data[u][v];
105                        }
106                }
107
108                private void insertCol(float[][] data, int u, float[] column) {
109                        final int height = data[0].length;
110                        for (int v = 0; v < height; v++) {
111                                data[u][v] = (float) column[v];
112                        }
113                }
114        }
115
116}
