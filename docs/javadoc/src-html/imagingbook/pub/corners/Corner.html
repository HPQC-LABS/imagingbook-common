010package imagingbook.pub.corners;
011
012import imagingbook.pub.geometry.basic.Point;
013
014/**
015 * This class represents a 2D corner.
016 * A corner is essentially a {@link Point} plus a scalar quantity
017 * {@link #q} for the corner strength.
018 * 
019 * @version 2020-01-09
020 */
021public class Corner implements Point, Comparable<Corner> {
022        protected final float x, y, q;
023
024        public double getX() {
025                return x;
026        }
027
028        public double getY() {
029                return y;
030        }
031
032        public double getQ() {
033                return q;
034        }
035        
036        public Corner (float x, float y, float q) {
037                this.x = x;
038                this.y = y;
039                this.q = q;
040        }
041        
042        double dist2 (Corner c2){
043                //returns the squared distance between this corner and corner c2
044                float dx = this.x - c2.x;
045                float dy = this.y - c2.y;
046                return (dx * dx) + (dy * dy);   
047        }
048
049        // used for sorting corners by corner strength q
050        @Override
051        public int compareTo (Corner c2) {
052                return Double.compare(c2.q, this.q);
053        }
054}
