010package imagingbook.pub.corners;
011
012import static imagingbook.lib.math.Arithmetic.sqr;
013
014import java.util.ArrayList;
015import java.util.Collections;
016import java.util.List;
017
018import ij.process.FloatProcessor;
019import ij.process.ImageProcessor;
020import imagingbook.lib.filters.GaussianFilter;
021import imagingbook.lib.image.Filter;
022import imagingbook.lib.image.ImageMath;
023
024/**
025 * Abstract super class for all corner detectors based on local 
026 * structure tensor.
027 * 
028 * @author W. Burger
029 * @version 2020/02/25
030 */
031public abstract class AbstractGradientCornerDetector implements PointFeatureDetector {
032        
033        /** For testing/example calculations only! */
034        public static boolean RETAIN_TEMPORARY_DATA = false;
035        
036        protected static class Parameters {
037                /** Sigma of Gaussian filter used for smoothing gradient maps */
038                public double sigma = 1.275;
039                /** Corner response threshold */
040                public double tH = 20000;
041                /** Width of border region ignored in corner search */
042                public int border = 20;
043                /** Set true to perform final corner cleanup */
044                public boolean doCleanUp = true;
045                /** Min. distance between final corners */
046                public double dmin = 10;
047        }
048        
049        //filter kernels (one-dim. part of separable 2D filters)
050        private final static float[] hp = {2f/9, 5f/9, 2f/9};   // pre-smoothing filter kernel
051        private final static float[] hd = {0.5f, 0, -0.5f};             // first derivative kernel
052//      private final static float[] hb = {1f/64, 6f/64, 15f/64, 20f/64, 15f/64, 6f/64, 1f/64}; // original gradient blur filter kernel
053                
054        protected final int M, N;
055        protected final Parameters params;
056        protected final FloatProcessor Q;
057        
058        // for testing only - REMOVE!
059        public FloatProcessor Ixx = null;
060        public FloatProcessor Iyy = null;
061        public FloatProcessor Ixy = null;
062        
063        // ---------------------------------------------------------------------------
064        
065        protected AbstractGradientCornerDetector(ImageProcessor ip, Parameters params) {
066                this.M = ip.getWidth();
067                this.N = ip.getHeight();
068                this.params = params;
069                init();
070                Q = makeCornerScores(ip);
071        }
072        
073        /**
074         * To perform any necessary initializations in subclasses.
075         * Called by the constructor.
076         */
077        protected abstract void init(); 
078        
079        /**
080         * Calculates the corner response score for a single image position (u,v)
081         * from the elements A, B, C of the local structure matrix.
082         * To be implemented by concrete sub-classes.
083         * @param A = Ixx(u,v)
084         * @param B = Iyy(u,v)
085         * @param C = Iyy(u,v)
086         * @return the corner score
087         */
088        public abstract float computeScore(float A, float B, float C);
089        
090        private FloatProcessor makeCornerScores(ImageProcessor I) {
091                FloatProcessor Ix = I.convertToFloatProcessor(); 
092                FloatProcessor Iy = I.convertToFloatProcessor();
093                
094                // nothing but a Sobel gradient estimate:
095                Filter.convolveY(Ix, hp);                               // pre-filter Ix vertically
096                Filter.convolveX(Iy, hp);                               // pre-filter Iy horizontally
097                Filter.convolveX(Ix, hd);                               // get horizontal derivative 
098                Filter.convolveY(Iy, hd);                               // get vertical derivative
099                
100                // gradient products:
101                FloatProcessor Ixx = ImageMath.sqr(Ix);                         // Ixx = Ix^2
102                FloatProcessor Iyy = ImageMath.sqr(Iy);                         // Iyy = Iy^2
103                FloatProcessor Ixy = ImageMath.mult(Ix, Iy);            // Ixy = Ix * Iy
104                
105                // blur the gradient components with a small Gaussian:
106                float[] gaussKernel = GaussianFilter.makeGaussKernel1D(params.sigma);
107                Filter.convolveXY(Ixx, gaussKernel);
108                Filter.convolveXY(Iyy, gaussKernel);
109                Filter.convolveXY(Ixy, gaussKernel);
110                
111                FloatProcessor Q = new FloatProcessor(M, N);
112                
113                final float[] A = (float[]) Ixx.getPixels();
114                final float[] B = (float[]) Iyy.getPixels();
115                final float[] C = (float[]) Ixy.getPixels();
116                final float[] q = (float[]) Q.getPixels();
117                
118                for (int i = 0; i < M * N; i++) {
119                        q[i] = computeScore(A[i], B[i], C[i]);
120                }
121
122                // for demo examples only - REMOVE!
123                if (RETAIN_TEMPORARY_DATA) {
124                        this.Ixx = Ixx;
125                        this.Iyy = Iyy;
126                        this.Ixy = Ixy;
127                }
128                
129                return Q;
130        }
131        
132        /**
133         * Returns the corner score function as a {@link FloatProcessor} object.
134         * @return the score function
135         */
136        public FloatProcessor getQ() {
137                return Q;
138        }
139        
140        @Override
141        public List<Corner> getPoints() {       
142                List<Corner> corners = collectCorners();
143                if (params.doCleanUp) {
144                        corners = cleanupCorners(corners);
145                }
146                return corners;
147        }
148        
149        public List<Corner> getCorners() {
150                return getPoints();
151        }
152        
153        // uses 8-neighborhood
154        private boolean isLocalMax (FloatProcessor Q, int u, int v) {
155                if (u <= 0 || u >= M - 1 || v <= 0 || v >= N - 1) {
156                        return false;
157                } 
158                else {
159                        final float[] q = (float[]) Q.getPixels();
160                        final int i0 = (v - 1) * M + u;
161                        final int i1 = v * M + u;
162                        final int i2 = (v + 1) * M + u;
163                        final float q0 = q[i1];
164                        return  // check 8 neighbors of q0
165                                q0 >= q[i0 - 1] && q0 >= q[i0] && q0 >= q[i0 + 1] &&
166                                q0 >= q[i1 - 1] &&                q0 >= q[i1 + 1] && 
167                                q0 >= q[i2 - 1] && q0 >= q[i2] && q0 >= q[i2 + 1] ;
168                }
169        }
170        
171        private List<Corner> cleanupCorners(List<Corner> C) {
172                final double dmin2 = sqr(params.dmin);
173                // sort corners by descending q-value:
174                Collections.sort(C);
175                // we use an array of corners for efficiency reasons:
176                Corner[] Ca = C.toArray(new Corner[C.size()]);
177                List<Corner> Cclean = new ArrayList<Corner>(C.size());
178                for (int i = 0; i < Ca.length; i++) {
179                        Corner c0 = Ca[i];              // get next strongest corner
180                        if (c0 != null) {
181                                Cclean.add(c0);
182                                // delete all remaining corners cj too close to c0:
183                                for (int j = i + 1; j < Ca.length; j++) {
184                                        Corner cj = Ca[j];
185                                        if (cj != null && c0.distance2(cj) < dmin2)
186                                                Ca[j] = null;   //delete corner cj from C
187                                }
188                        }
189                }
190                return Cclean;
191        }
192        
193        private List<Corner> collectCorners() {
194                final float tH = (float) params.tH;
195                final int border = params.border;
196                List<Corner> C = new ArrayList<Corner>();
197                for (int v = border; v < N - border; v++) {
198                        for (int u = border; u < M - border; u++) {
199                                float q = Q.getf(u, v);
200                                if (q > tH && isLocalMax(Q, u, v)) {    // 
201                                        C.add(new Corner(u, v, q));
202                                }
203                        }
204                }
205                return C;
206        }
