016/**
017 * This class represents an algebraic line of the form a * x + b * y + c = 0.
018 * All instances are normalized such that ||(a,b)|| = 1.
019 * Instances of this class are immutable.
020 * @author WB
021 *
022 */
023public class AlgebraicLine {
024        
025        protected final double a, b, c;
026
027        // static factory methods ----------------------------------------
028        
029        public static AlgebraicLine create(double a, double b, double c) {
030                double norm = Math.sqrt(a * a + b * b);
031                return new AlgebraicLine(a / norm, b / norm , c / norm);
032        }
033        
034        public static AlgebraicLine create(Point p1, Point p2) {
035                double a = p1.getY() - p2.getY();
036                double b = p2.getX() - p1.getX();
037                double c = -a * p1.getX() - b * p1.getY();
038                return create(a, b, c);
039        }
040        
041        // constructors --------------------------------------------------
042        
043        protected AlgebraicLine(double a, double b, double c) {
044                this.a = a;
045                this.b = b;
046                this.c = c;
047        }
048        
049        // getter/setter methods ------------------------------------------
050        
051        public final double getA() {
052                return a;
053        }
054
055        public final double getB() {
056                return b;
057        }
058
059        public final double getC() {
060                return c;
061        }
062
063        // other methods ------------------------------------------
064        
065        /**
066         * Returns the perpendicular distance between this line and the point (x, y).
067         * The result may be positive or negative, depending on which side of the line
068         * (x, y) is located. It is assumed that the line is normalized, i.e.,
069         * sqrt(a * a + b * b) == 1.
070         * 
071         * @param x x-coordinate of point position.
072         * @param y y-coordinate of point position.
073         * @return The perpendicular distance between this line and the point (x, y).
074         */
075        public double getDistance(double x, double y) {
076                return (a * x + b * y + c); // / Math.sqrt(a * a + b * b);
077        }
078        
079        
080        /**
081         * Returns the perpendicular distance between this line and the point p. The
082         * result may be positive or negative, depending on which side of the line p is
083         * located.
084         * 
085         * @param p point position.
086         * @return The perpendicular distance between this line and the point p.
087         */
088        public double getDistance(Point p) {
089                return getDistance(p.getX(), p.getY());
090        }
091        
092        @Override
093        public String toString() {
094                return String.format(Locale.US, "%s <a = %.3f, b = %.3f, c = %.3f>",
095                                this.getClass().getSimpleName(), a, b, c);
096        }
097        
098        /**
099         * This is a brute-force drawing method which simply marks all image pixels that
100         * are sufficiently close to the HoughLine hl. The drawing color for ip must be
101         * previously set.
102         * 
103         * @param ip        the {@link ImageProcessor} instance to draw to.
104         * @param thickness the thickness of the lines to be drawn.
105         */
106        public void draw(ImageProcessor ip, double thickness) {
107                final int w = ip.getWidth();
108                final int h = ip.getHeight();
109                final double dmax = 0.5 * thickness;
110                for (int u = 0; u < w; u++) {
111                        for (int v = 0; v < h; v++) {
112                                // get the distance between (u,v) and the line hl:
113                                double d = Math.abs(this.getDistance(u, v));
114                                if (d <= dmax) {
115                                        ip.drawPixel(u, v);
116                                }
117                        }
118                }
119        }
