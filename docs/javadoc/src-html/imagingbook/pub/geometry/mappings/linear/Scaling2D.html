010package imagingbook.pub.geometry.mappings.linear;
011
012/**
013 * This class represents a 2D scaling transformation (as a special case of 
014 * affine transformation).
015 */
016public class Scaling2D extends AffineMapping2D {
017        
018        public Scaling2D(Scaling2D sc) {
019                super(sc);
020        }
021
022        /**
023         * Constructor.
024         * Creates a mapping that scales along the x- and y-axis
025         * by the associated factors.
026         * 
027         * @param sx the scale factor in x-direction
028         * @param sy the scale factor in y-direction
029         */
030        public Scaling2D(double sx, double sy) {
031                super(
032                        sx, 0,  0,
033                        0,  sy, 0);
034        }
035        
036        /**
037         * Constructor. Creates a uniform scaling in x and y.
038         * @param s the common scale factor
039         */
040        public Scaling2D(double s) {
041                this(s, s);
042        }
043        
044        /**
045         * {@inheritDoc}
046         * @return a new scaling transformation
047         */
048        @Override
049        public Scaling2D duplicate() {
050                return new Scaling2D(this);
051        }
052}
