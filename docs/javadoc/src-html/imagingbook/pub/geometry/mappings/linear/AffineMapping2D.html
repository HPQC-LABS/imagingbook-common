017/**
018 * This class represents an affine transformation in 2D, which can be defined 
019 * by three pairs of corresponding points.
020 * It can be assumed that every instance of this class is indeed an affine mapping.

024        /**
025         * Creates an affine 2D mapping from two sequences of corresponding points.
026         * If 3 point pairs are specified, the mapping is exact, otherwise a
027         * minimum least-squares fit is calculated.
028         * @param P the source points
029         * @param Q the target points
030         * @return a new affine mapping for the two point sets
031         */
032        public static AffineMapping2D fromPoints(Point[] P, Point[] Q) {
033                AffineFit2D fit = new AffineFit2D(P, Q);
034                return new AffineMapping2D(fit.getTransformationMatrix().getData());
035        }

080        
081        /**
082         * Creates the identity mapping.
083         */
084        public AffineMapping2D() {
085                super();
086        }
087        
088        /**
089         * Creates a linear mapping from a transformation matrix A,
090         * which must be at least of size 2 x 3.
091         * If A is larger than 2 x 3, the remaining elements are ignored.
092         * @param A a 2 x 3(or larger) matrix
093         */
094        public AffineMapping2D(double[][] A) {
095                super(A[0][0], A[0][1], A[0][2], A[1][0], A[1][1], A[1][2], 0, 0);
096        }
097
098        /**
099         * Creates an affine mapping from the specified matrix elements.
100         * @param a00 matrix element A_00
101         * @param a01 matrix element A_01
102         * @param a02 matrix element A_02
103         * @param a10 matrix element A_10
104         * @param a11 matrix element A_11
105         * @param a12 matrix element A_12
106         */
107        public AffineMapping2D (
108                        double a00, double a01, double a02, 
109                        double a10, double a11, double a12) {
110                super(a00, a01, a02, a10, a11, a12, 0, 0);
111        }
112
113        /**
114         * Creates a new affine mapping from an existing affine mapping.
115         * @param m an affine mapping
116         */
117        public AffineMapping2D(AffineMapping2D m) {
118                this(m.a00, m.a01, m.a02, m.a10, m.a11, m.a11);
119        }
120        
121        // ----------------------------------------------------------
122        
123        /** Deviation limit for checking 0 and 1 values. */
124        protected static double AffineTolerance = 1e-15;
125        
126        /**
127         * Checks if the given linear mapping could be affine, i.e. if the
128         * bottom row of its transformation matrix is (0, 0, 1). 
129         * Note that this is a necessary but not sufficient requirement.
130         * The threshold {@link AffineTolerance} is used in this check.
131         * @param map a linear mapping
132         * @return true if the mapping could be affine
133         */
134        public static boolean isAffine(LinearMapping2D map) {
135                if (Math.abs(map.a20) > AffineTolerance) return false;
136                if (Math.abs(map.a21) > AffineTolerance) return false;
137                if (Math.abs(map.a22 - 1.0) > AffineTolerance) return false;
138                return true;
139        }
140
141        /**
142         * Concatenates this mapping A with another affine mapping B and returns
143         * a new mapping C, such that C(x) = B(A(x)).
144         * @param B the second mapping
145         * @return the concatenated affine mapping
146         */
147        public AffineMapping2D concat(AffineMapping2D B) {      // use some super method instead?
148                double[][] C = Matrix.multiply(B.getTransformationMatrix(), this.getTransformationMatrix());
149                return new AffineMapping2D(C[0][0], C[0][1], C[0][2], C[1][0], C[1][1], C[1][2]);
150        }
151
152        /**
153         * {@inheritDoc}
154         * Note that inverting an affine transformation always yields
155         * another affine transformation.
156         */
157        public AffineMapping2D getInverse() {
158                double det = a00 * a11 - a01 * a10;
159                double b00 = a11 / det;
160                double b01 = -a01 / det;
161                double b02 = (a01 * a12 - a02 * a11) / det;
162                double b10 = -a10 / det;
163                double b11 = a00 / det;
164                double b12 = (a02 * a10 - a00 * a12) / det;
165                return new AffineMapping2D(b00, b01, b02, b10, b11, b12);
166        }
167
168        /**
169         * {@inheritDoc}
170         * @return a new affine mapping
171         */
172        public AffineMapping2D duplicate() {
173                return new AffineMapping2D(this);
174        }
175
176        @Override
177        public double[][] getJacobian(Point xy) {
178                final double x = xy.getX();
179                final double y = xy.getY();
180                return new double[][]
181                        {{x, y, 0, 0, 1, 0},
182                         {0, 0, x, y, 0, 1}};
183        }

200                System.out.println("ai = \n" + Matrix.toString(ai));

202                LinearMapping2D Ai = new LinearMapping2D(ai);

204                
205                double[][] I = Matrix.multiply(a, ai);
206                System.out.println("\ntest: should be the  identity matrix: = \n" + Matrix.toString(I));
207        }
