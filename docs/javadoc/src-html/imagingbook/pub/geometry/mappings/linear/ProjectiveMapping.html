024/**
025 * This class represents a projective transformation in 2D (also known
026 * as a "homography"). It can be defined by four pairs of corresponding
027 * points.
028 */
029public class ProjectiveMapping extends LinearMapping implements WarpParameters {

031        /**
032         * Creates the most specific linear mapping from two sequences of corresponding
033         * 2D points.
034         * 
035         * @param P first point sequence
036         * @param Q second point sequence
037         * @return a linear mapping derived from point correspondences
038         */
039        public static ProjectiveMapping makeMapping(Point2D[] P, Point2D[] Q) { // TODO: Check for a better solution!
040                int minLen = Math.min(P.length, Q.length);
041                if (minLen < 1) {
042                        throw new IllegalArgumentException("cannot create a mapping from zero points");
043                }
044                else if (minLen <= 2) {                         // TODO: special case for minLen == 2, rigid transformation??
045                        return null; //new Translation(P, Q);
046                }
047                else if (minLen <= 3) {
048                        return AffineMapping.fromPoints(P, Q);
049                }
050                else {
051                        return ProjectiveMapping.fromQuadToQuad(P, Q);
052                }
053        }

114        /**
115         * Creates the projective mapping from the unit square S to
116         * the arbitrary quadrilateral P, specified by four points.
117         * 
118         * @param p0 point 0
119         * @param p1 point 1
120         * @param p2 point 2
121         * @param p3 point 3
122         * @return a new projective mapping
123         */
124        public static ProjectiveMapping fromUnitSquareToQuad(Point2D p0, Point2D p1, Point2D p2, Point2D p3) {
125                double x0 = p0.getX(), x1 = p1.getX(), x2 = p2.getX(), x3 = p3.getX();
126                double y0 = p0.getY(), y1 = p1.getY(), y2 = p2.getY(), y3 = p3.getY();
127                double S = (x1 - x2) * (y3 - y2) - (x3 - x2) * (y1 - y2);
128                // TODO: check S for zero value and throw exception
129                double a20 = ((x0 - x1 + x2 - x3) * (y3 - y2) - (y0 - y1 + y2 - y3) * (x3 - x2)) / S;
130                double a21 = ((y0 - y1 + y2 - y3) * (x1 - x2) - (x0 - x1 + x2 - x3) * (y1 - y2)) / S;
131                double a00 = x1 - x0 + a20 * x1;
132                double a01 = x3 - x0 + a21 * x3;
133                double a02 = x0;
134                double a10 = y1 - y0 + a20 * y1;
135                double a11 = y3 - y0 + a21 * y3;
136                double a12 = y0;
137                return new ProjectiveMapping(a00, a01, a02, a10, a11, a12, a20, a21);
138        }

166         * @param p2 point 2 of source quad P

170         * @param q2 point 2 of target quad Q
171         * @param q3 point 3 of target quad Q
172         * @return a new projective mapping
173         */
174        public static ProjectiveMapping fromQuadToQuad(
175                        Point2D p0, Point2D p1, Point2D p2, Point2D p3, 
176                        Point2D q0, Point2D q1, Point2D q2, Point2D q3) {
177                ProjectiveMapping T1 = ProjectiveMapping.fromUnitSquareToQuad(p0, p1, p2, p3);
178                ProjectiveMapping T2 = ProjectiveMapping.fromUnitSquareToQuad(q0, q1, q2, q3);
179                ProjectiveMapping T1i = T1.getInverse();
180                return T1i.concat(T2);          
181        }

240        /**
241         * Maps between more than 4 point pairs, finds a least-squares solution
242         * for the homography parameters.
243         * NOTE: this is UNFINISHED code!
244         * @param P sequence of points (source)
245         * @param Q sequence of points (target)
246         * @return a new projective mapping
247         */
248        public static ProjectiveMapping fromPointsToPoints(Point2D[] P, Point2D[] Q) {
249                final int n = P.length;
250                double[] ba = new double[2 * n];
251                double[][] Ma = new double[2 * n][];
252                for (int i = 0; i < n; i++) {
253                        double x = P[i].getX();
254                        double y = P[i].getY();
255                        double u = Q[i].getX();
256                        double v = Q[i].getY();
257                        ba[2 * i + 0] = u;
258                        ba[2 * i + 1] = v;
259                        Ma[2 * i + 0] = new double[] { x, y, 1, 0, 0, 0, -u * x, -u * y };
260                        Ma[2 * i + 1] = new double[] { 0, 0, 0, x, y, 1, -v * x, -v * y };
261                }
262                
263                RealMatrix M = MatrixUtils.createRealMatrix(Ma);
264                RealVector b = MatrixUtils.createRealVector(ba);
265                DecompositionSolver solver = new SingularValueDecomposition(M).getSolver();
266                RealVector h = solver.solve(b);
267                double a00 = h.getEntry(0);
268                double a01 = h.getEntry(1);
269                double a02 = h.getEntry(2);
270                double a10 = h.getEntry(3);
271                double a11 = h.getEntry(4);
272                double a12 = h.getEntry(5);
273                double a20 = h.getEntry(6);
274                double a21 = h.getEntry(7);
275
276                return new ProjectiveMapping(a00, a01, a02, a10, a11, a12, a20, a21);
277        }

282         * Concatenates this mapping A with another linear mapping B and returns
283         * a new mapping C, such that C(x) = B(A(x)).
284         * @param B the second mapping
285         * @return the concatenated mapping
286         */
287        public ProjectiveMapping concat(ProjectiveMapping B) {
288                LinearMapping A = this;
289                LinearMapping C = A.concat(B);
290                return new ProjectiveMapping(C);

315        /**
316         * {@inheritDoc}
317         * Note that the inverse A' of a projective transformation matrix A is again a linear transformation
318         * but its a'2' element is generally not 1. Scaling A' to A'' = A' / a22' yields a projective transformation
319         * that reverses A. While A * A' = I, the result of A * A'' is a scaled identity matrix.
320         */
321        public ProjectiveMapping getInverse() { // TODO: use super-class method, implement with Apache Commons Math?
322                double det = a00*a11 + a01*a12*a20 + a02*a10*a21 - 
323                                     a00*a12*a21 - a01*a10 - a02*a11*a20;
324                double b22 = (a00*a11 - a01*a10) / det;
325                // normalizing matrix elements by b22 (also see method normalize()):
326                double b00 = (a11 - a12*a21) / det / b22;
327                double b01 = (a02*a21 - a01) / det / b22; 
328                double b02 = (a01*a12 - a02*a11) / det / b22; 
329                double b10 = (a12*a20 - a10) / det / b22;
330                double b11 = (a00 - a02*a20) / det / b22; 
331                double b12 = (a02*a10 - a00*a12) / det / b22;
332                double b20 = (a10*a21 - a11*a20) / det / b22; 
333                double b21 = (a01*a20 - a00*a21) / det / b22;
334                
335                return new ProjectiveMapping(b00, b01, b02, b10, b11, b12, b20, b21);
336        }

380        public double[][] getWarpJacobian(double[] xy) {
381                // see Baker 2003 "20 Years" Part 1, Eq. 99 (p. 46)
382                final double x = xy[0];
383                final double y = xy[1];
384                double a = a00 * x + a01 * y + a02;     // = alpha
385                double b = a10 * x + a11 * y + a12;     // = beta
386                double c = a20 * x + a21 * y + 1;       // = gamma
387                double cc = c * c;
388                // TODO: check c for zero-value and throw exception, make more efficient
389                return new double[][]
390                        {{x/c, y/c, 0,   0,   -(x*a)/cc, -(y*a)/cc, 1/c, 0  },
391                         {0,   0,   x/c, y/c, -(x*b)/cc, -(y*b)/cc, 0,   1/c}};
392        }
