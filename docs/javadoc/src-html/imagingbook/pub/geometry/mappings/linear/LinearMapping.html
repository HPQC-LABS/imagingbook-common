010package imagingbook.pub.geometry.mappings.linear;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.lib.settings.PrintPrecision;
014import imagingbook.pub.geometry.mappings.Mapping;
015
016/**
017 * This class represents an arbitrary linear transformation in 2D.
018 * Instances of this class and any subclass are immutable.
019 */
020public class LinearMapping extends Mapping {
021        
022        protected final double 
023                a00, a01, a02,
024                a10, a11, a12,
025                a20, a21, a22;
026
027        /**
028         * Creates a new identity mapping.
029         */
030        public LinearMapping() {
031                a00 = 1; a01 = 0; a02 = 0;
032                a10 = 0; a11 = 1; a12 = 0;
033                a20 = 0; a21 = 0; a22 = 1;
034        }
035        
036        /**
037         * Creates a linear mapping from a transformation matrix A.
038         * @param A a 3 x 3 matrix
039         */
040        public LinearMapping(double[][] A) {
041                a00 = A[0][0]; a01 = A[0][1]; a02 = A[0][2];
042                a10 = A[1][0]; a11 = A[1][1]; a12 = A[1][2];
043                a20 = A[2][0]; a21 = A[2][1]; a22 = A[2][2];
044        }
045
046        /**
047         * Creates an arbitrary linear mapping from the specified matrix elements.
048         * @param a00 matrix element A_00
049         * @param a01 matrix element A_01
050         * @param a02 matrix element A_02
051         * @param a10 matrix element A_10
052         * @param a11 matrix element A_11
053         * @param a12 matrix element A_12
054         * @param a20 matrix element A_20
055         * @param a21 matrix element A_21
056         * @param a22 matrix element A_22
057         */
058        public LinearMapping (
059                        double a00, double a01, double a02, 
060                        double a10, double a11, double a12,
061                        double a20, double a21, double a22) {
062                this.a00 = a00;  this.a01 = a01;  this.a02 = a02;
063                this.a10 = a10;  this.a11 = a11;  this.a12 = a12;
064                this.a20 = a20;  this.a21 = a21;  this.a22 = a22;
065        }
066        
067        /**
068         * Creates a new linear mapping from an existing linear mapping.
069         * @param lmap a given linear mapping
070         */
071        public LinearMapping (LinearMapping lmap) {
072                this(lmap.getTransformationMatrix());
073        }
074        
075        // ----------------------------------------------------------
076        
077        /**
078         * Scales the transformation matrix such that a22 becomes 1.
079         * Any linear mapping can be normalized and thereby be converted to
080         * a projective mapping (see {@link ProjectiveMapping}.
081         * @return the normalized linear mapping (i.e., a projective mapping)
082         */
083        public ProjectiveMapping normalize() {
084                // TODO: check a22 for zero value and throw exception, move to ProjectiveMapping, make constructor?
085                double b00 = a00/a22;   double b01 = a01/a22;   double b02 = a02/a22;
086                double b10 = a10/a22;   double b11 = a11/a22;   double b12 = a12/a22;
087                double b20 = a20/a22;   double b21 = a21/a22;
088                return new ProjectiveMapping(b00, b01, b02, b10, b11, b12, b20, b21);
089        }
090        
091        // ----------------------------------------------------------
092        
093        @Override
094        public double[] applyTo (double x, double y) {
095                double h =  (a20 * x + a21 * y + a22);
096                double x1 = (a00 * x + a01 * y + a02) / h;
097                double y1 = (a10 * x + a11 * y + a12) / h;
098                return new double[] {x1, y1};
099        }
100        
101        /**
102         * Calculates and returns the inverse mapping.
103         */
104        public LinearMapping getInverse() {
105                // System.out.println("LinearMapping getInverse()");
106                double[][] ai = Matrix.inverse(this.getTransformationMatrix());
107                return new LinearMapping(ai);
108        }
109        
110        /**
111         * Concatenates this mapping A with another linear mapping B and returns
112         * a new mapping C, such that C(x) = B(A(x)).
113         * @param B the second mapping
114         * @return the concatenated mapping
115         */
116        public LinearMapping concat(LinearMapping B) {
117                double[][] a = this.getTransformationMatrix();
118                double[][] b = B.getTransformationMatrix();
119                return new LinearMapping(Matrix.multiply(b, a));
120        }
121        
122        /**
123         * Concatenates a sequence of linear mappings AA = (A1, A2, ..., An), with
124         * the result A(x) = A1(A2(...(An(x))...)). Thus, the mapping
125         * An is applied first and A1 last, with the associated transformation
126         * matrix a = a1 * a2 * ... * an.
127         * If AA is empty, the identity mapping is returned.
128         * If AA contains only a single mapping, a copy of this mapping is returned.
129         * 
130         * @param AA a (possibly empty) sequence of linear transformations
131         * @return the concatenated linear transformation 
132         */
133        public static LinearMapping concatenate(LinearMapping... AA) {
134                if (AA.length == 0) {
135                        return new LinearMapping();     // identity
136                }
137                else {
138                        double[][] a = AA[0].getTransformationMatrix();
139                        for (int i = 1; i < AA.length; i++) {
140                                a = Matrix.multiply(a, AA[i].getTransformationMatrix());
141                        }
142                        return new LinearMapping(a);
143                }
144        }
145        
146        /**
147         * Retrieves the transformation matrix for this mapping.
148         * @return the 3x3 transformation matrix
149         */
150        public double[][] getTransformationMatrix() {
151                return new double[][]
152                                {{a00, a01, a02},
153                                 {a10, a11, a12},
154                                 {a20, a21, a22}};
155        }
156
157        /**
158         * Returns a copy of this mapping.
159         * @return a new mapping of the same type
160         */
161        public LinearMapping duplicate() {
162                return new LinearMapping(this);
163        }
164
165        public String toString() {
166                return Matrix.toString(getTransformationMatrix());
167        }
168        
169        // -----------------------------------------------------------
170        
171        /**
172         * For testing only.
173         * @param args ignored
174         */
175        public static void main(String[] args) {
176                PrintPrecision.set(6);
177                double[][] A = 
178                                {{-1.230769, 2.076923, -1.769231}, 
179                                {-2.461538, 2.615385, -3.538462}, 
180                                {-0.307692, 0.230769, 1.000000}};
181                System.out.println("A = " + Matrix.toString(A));
182                
183                double[][] Ai = Matrix.inverse(A);
184                System.out.println("Ai = " + Matrix.toString(Ai));
185                
186                double[][] I = Matrix.multiply(A, Ai);
187                System.out.println("\ntest: should be the  identity matrix: = \n" + Matrix.toString(I));
188                
189                ProjectiveMapping pA = (new LinearMapping(A)).normalize();
190                System.out.println("mapping pA = \n" + pA.toString());
191                
192                ProjectiveMapping AA1 = pA.concat(pA);
193                System.out.println("mapping AA1 = \n" + AA1.toString());
194                
195                ProjectiveMapping AA2 = LinearMapping.concatenate(pA, pA).normalize();
196                System.out.println("mapping AA1 = \n" + AA2.toString());
197        }
198        
199}
