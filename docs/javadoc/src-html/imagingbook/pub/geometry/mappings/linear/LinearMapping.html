029        public LinearMapping() {
030                a00 = 1; a01 = 0; a02 = 0;
031                a10 = 0; a11 = 1; a12 = 0;
032                a20 = 0; a21 = 0; a22 = 1;
033        }
034        
035        public LinearMapping(double[][] A) {
036                a00 = A[0][0]; a01 = A[0][1]; a02 = A[0][2];
037                a10 = A[1][0]; a11 = A[1][1]; a12 = A[1][2];
038                a20 = A[2][0]; a21 = A[2][1]; a22 = A[2][2];
039        }

041        /**
042         * Creates an arbitrary linear mapping from the specified matrix elements.
043         * @param a00 matrix element A_00
044         * @param a01 matrix element A_01
045         * @param a02 matrix element A_02
046         * @param a10 matrix element A_10
047         * @param a11 matrix element A_11
048         * @param a12 matrix element A_12
049         * @param a20 matrix element A_20
050         * @param a21 matrix element A_21
051         * @param a22 matrix element A_22
052         */
053        public LinearMapping (
054                        double a00, double a01, double a02, 
055                        double a10, double a11, double a12,
056                        double a20, double a21, double a22) {
057                this.a00 = a00;  this.a01 = a01;  this.a02 = a02;
058                this.a10 = a10;  this.a11 = a11;  this.a12 = a12;
059                this.a20 = a20;  this.a21 = a21;  this.a22 = a22;
060//              isInverseFlag = inv;
061        }

069                this.a10 = lmap.a10;  this.a11 = lmap.a11;  this.a12 = lmap.a12;
070                this.a20 = lmap.a20;  this.a21 = lmap.a21;  this.a22 = lmap.a22;
071//              this.isInverseFlag = lmap.isInverseFlag;
072        }

076        /**
077         * Scales the transformation matrix such that a22 becomes 1.
078         * Any linear mapping can be normalized and thereby be converted to
079         * a projective mapping (see {@link ProjectiveMapping}.
080         * @return the normalized linear mapping (i.e., a projective mapping)
081         */
082        public ProjectiveMapping normalize() {
083                // TODO: check a22 for zero value and throw exception, move to ProjectiveMapping, make constructor?
084                double b00 = a00/a22;   double b01 = a01/a22;   double b02 = a02/a22;
085                double b10 = a10/a22;   double b11 = a11/a22;   double b12 = a12/a22;
086                double b20 = a20/a22;   double b21 = a21/a22;
087                return new ProjectiveMapping(b00, b01, b02, b10, b11, b12, b20, b21);
088        }

092        @Override
093        public double[] applyTo (double x, double y) {
094                double h =  (a20 * x + a21 * y + a22);
095                double x1 = (a00 * x + a01 * y + a02) / h;
096                double y1 = (a10 * x + a11 * y + a12) / h;
097                // pnt.setLocation(x1, y1);
098                return new double[] {x1, y1};
099        }

116                double b11 = (a00*a22 - a02*a20) / det;

119                double b21 = (a01*a20 - a00*a21) / det; 
120                double b22 = (a00*a11 - a01*a10) / det;
121                return new LinearMapping(b00, b01, b02, b10, b11, b12, b20, b21, b22);
122        }

155        /**
156         * Concatenates this mapping A with another linear mapping B and returns
157         * a new mapping C, such that C(x) = B(A(x)).
158         * @param B the second mapping
159         * @return the concatenated mapping
160         */
161        public LinearMapping concat(LinearMapping B) {
162                double b00 = B.a00*a00 + B.a01*a10 + B.a02*a20;
163                double b01 = B.a00*a01 + B.a01*a11 + B.a02*a21;
164                double b02 = B.a00*a02 + B.a01*a12 + B.a02*a22;
165                
166                double b10 = B.a10*a00 + B.a11*a10 + B.a12*a20;
167                double b11 = B.a10*a01 + B.a11*a11 + B.a12*a21;
168                double b12 = B.a10*a02 + B.a11*a12 + B.a12*a22;
169                
170                double b20 = B.a20*a00 + B.a21*a10 + B.a22*a20;
171                double b21 = B.a20*a01 + B.a21*a11 + B.a22*a21;
172                double b22 = B.a20*a02 + B.a21*a12 + B.a22*a22;
173                return new LinearMapping(b00, b01, b02, b10, b11, b12, b20, b21, b22);
174        }

201         * Retrieves the transformation matrix for this mapping.
202         * @return the 3x3 transformation matrix
203         */

205                return new double[][]
206                                {{a00, a01, a02},
207                                 {a10, a11, a12},
208                                 {a20, a21, a22}};
209        }

211        /**
212         * Returns a copy of this mapping.

214         */

216                return new LinearMapping(this);
217        }
