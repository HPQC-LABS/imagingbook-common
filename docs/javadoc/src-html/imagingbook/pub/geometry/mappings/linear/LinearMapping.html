010package imagingbook.pub.geometry.mappings.linear;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.pub.geometry.mappings.Mapping;
014
015/**
016 * This class represents an arbitrary linear transformation in 2D.
017 * TODO: make fields final.
018 */
019public class LinearMapping extends Mapping {
020        
021        protected double 
022                a00 = 1, a01 = 0, a02 = 0,
023                a10 = 0, a11 = 1, a12 = 0,
024                a20 = 0, a21 = 0, a22 = 1;
025
026        /**
027         * Creates a new identity mapping.
028         */
029        public LinearMapping() {
030        }
031
032        /**
033         * Creates an arbitrary linear mapping from the specified matrix elements.
034         * @param a00 matrix element A_00
035         * @param a01 matrix element A_01
036         * @param a02 matrix element A_02
037         * @param a10 matrix element A_10
038         * @param a11 matrix element A_11
039         * @param a12 matrix element A_12
040         * @param a20 matrix element A_20
041         * @param a21 matrix element A_21
042         * @param a22 matrix element A_22
043         * @param inv set true if this mapping represents an inverse transformation
044         */
045        public LinearMapping (
046                        double a00, double a01, double a02, 
047                        double a10, double a11, double a12,
048                        double a20, double a21, double a22, boolean inv) {
049                this.a00 = a00;  this.a01 = a01;  this.a02 = a02;
050                this.a10 = a10;  this.a11 = a11;  this.a12 = a12;
051                this.a20 = a20;  this.a21 = a21;  this.a22 = a22;
052                isInverseFlag = inv;
053        }
054        
055        /**
056         * Creates a new linear mapping from an existing linear mapping.
057         * @param lmap a given linear mapping
058         */
059        public LinearMapping (LinearMapping lmap) {
060                this.a00 = lmap.a00;  this.a01 = lmap.a01;  this.a02 = lmap.a02;
061                this.a10 = lmap.a10;  this.a11 = lmap.a11;  this.a12 = lmap.a12;
062                this.a20 = lmap.a20;  this.a21 = lmap.a21;  this.a22 = lmap.a22;
063                this.isInverseFlag = lmap.isInverseFlag;
064        }
065        
066        // TODO: add a constructor that accepts a sequence of linear mappings, make concatenation protected.
067        
068        // ----------------------------------------------------------
069        
070        @Override
071        public double[] applyTo (double x, double y) {
072                double h =  (a20 * x + a21 * y + a22);
073                double x1 = (a00 * x + a01 * y + a02) / h;
074                double y1 = (a10 * x + a11 * y + a12) / h;
075                // pnt.setLocation(x1, y1);
076                return new double[] {x1, y1};
077        }
078        
079        // TODO: should be implemented differently (generics)
080        public LinearMapping getInverse() {
081                return (LinearMapping) super.getInverse();
082//              if (isInverseFlag)
083//                      return this;
084//              else {
085//                      return this.invert();
086//              }
087        }
088        
089
090        /**
091         * Calculates and returns the inverted mapping.
092         */
093        public LinearMapping invert() {
094                LinearMapping lm = new LinearMapping(this);
095                lm.invertDestructive();
096                return lm;
097        }
098        
099        /**
100         * Invertes this mapping destructively ("in-place") i.e.,
101         * without creating a new instance.
102         */
103        protected void invertDestructive() {
104                double det = a00*a11*a22 + a01*a12*a20 + a02*a10*a21 - 
105                                         a00*a12*a21 - a01*a10*a22 - a02*a11*a20;
106                double b11 = (a11*a22 - a12*a21) / det; 
107                double b12 = (a02*a21 - a01*a22) / det; 
108                double b13 = (a01*a12 - a02*a11) / det; 
109                double b21 = (a12*a20 - a10*a22) / det; 
110                double b22 = (a00*a22 - a02*a20) / det; 
111                double b23 = (a02*a10 - a00*a12) / det;
112                double b31 = (a10*a21 - a11*a20) / det; 
113                double b32 = (a01*a20 - a00*a21) / det; 
114                double b33 = (a00*a11 - a01*a10) / det;
115                a00 = b11;              a01 = b12;              a02 = b13;
116                a10 = b21;              a11 = b22;              a12 = b23;
117                a20 = b31;              a21 = b32;              a22 = b33;
118                isInverseFlag = !isInverseFlag;
119        }
120        
121        /**
122         * Concatenates this mapping A with another linear mapping B and returns
123         * a new mapping C, such that C(x) = B(A(x)).
124         * @param B the second mapping
125         * @return the concatenated mapping
126         */
127        public LinearMapping concat(LinearMapping B) {
128                LinearMapping A = new LinearMapping(this);
129                A.concatDestructive(B);
130                return A;
131        }
132        
133        /**
134         * Concatenates this mapping A destructively with another linear mapping B,
135         * such that A(x) becomes B(A(x)).
136         * TODO: This should not be public!
137         * @param B the second mapping
138         */
139        public void concatDestructive(LinearMapping B) {
140                double b11 = B.a00*a00 + B.a01*a10 + B.a02*a20;
141                double b12 = B.a00*a01 + B.a01*a11 + B.a02*a21;
142                double b13 = B.a00*a02 + B.a01*a12 + B.a02*a22;
143                
144                double b21 = B.a10*a00 + B.a11*a10 + B.a12*a20;
145                double b22 = B.a10*a01 + B.a11*a11 + B.a12*a21;
146                double b23 = B.a10*a02 + B.a11*a12 + B.a12*a22;
147                
148                double b31 = B.a20*a00 + B.a21*a10 + B.a22*a20;
149                double b32 = B.a20*a01 + B.a21*a11 + B.a22*a21;
150                double b33 = B.a20*a02 + B.a21*a12 + B.a22*a22;
151                a00 = b11;              a01 = b12;              a02 = b13;
152                a10 = b21;              a11 = b22;              a12 = b23;
153                a20 = b31;              a21 = b32;              a22 = b33;
154        }
155        
156        /**
157         * Retrieves the transformation matrix for this mapping.
158         * @return the 3x3 transformation matrix
159         */
160        public double[][] getTransformationMatrix() {
161                return new double[][]
162                                {{a00, a01, a02},
163                                 {a10, a11, a12},
164                                 {a20, a21, a22}};
165        }
166
167        /**
168         * Returns a copy of this mapping.
169         * @return a new linear mapping
170         */
171        public LinearMapping duplicate() {      // TODO: use generics?
172//              return (LinearMapping) this.clone();
173                return new LinearMapping(this);
174        }
175
176        public String toString() {
177                return Matrix.toString(getTransformationMatrix());
178        }
179        
180}
