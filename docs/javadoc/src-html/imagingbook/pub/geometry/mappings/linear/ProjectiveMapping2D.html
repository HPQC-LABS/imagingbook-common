019 * This class represents a projective transformation in 2D (also known
020 * as a "homography"). It can be specified uniquely by four pairs of corresponding
021 * points.
022 * It can be assumed that every instance of this class is indeed a projective mapping.

028        /**
029         * Creates a projective 2D mapping from two sequences of corresponding points.
030         * If 4 point pairs are specified, the mapping is exact, otherwise a
031         * minimum least-squares fit is calculated.
032         * @param P the source points
033         * @param Q the target points
034         * @return a new projective mapping for the two point sets
035         */
036        public static ProjectiveMapping2D fromPoints(Point[] P, Point[] Q) {
037                ProjectiveFit2D fit = new ProjectiveFit2D(P, Q);
038                return new ProjectiveMapping2D(fit.getTransformationMatrix());
039        }

169        //  constructors -----------------------------------------------------
170        
171        /**
172         * Creates the identity mapping.
173         */
174        public ProjectiveMapping2D() {
175                super();
176        }
177        
178        /**
179         * Creates a projective mapping from a transformation matrix A,
180         * which must be at least of size 3 x 3.
181         * If A is larger than 3 x 3, the remaining elements are ignored.
182         * @param A a 2 x 3(or larger) matrix
183         */
184        public ProjectiveMapping2D(double[][] A) {
185                this(new LinearMapping2D(A));
186        }
187
188        
189        /**
190         * Creates a projective mapping from the specified matrix elements.
191         * @param a00 matrix element A_00
192         * @param a01 matrix element A_01
193         * @param a02 matrix element A_02
194         * @param a10 matrix element A_10
195         * @param a11 matrix element A_11
196         * @param a12 matrix element A_12
197         * @param a20 matrix element A_20
198         * @param a21 matrix element A_21
199         */
200        public ProjectiveMapping2D(
201                        double a00, double a01, double a02, 
202                        double a10, double a11, double a12, 
203                        double a20, double a21) {
204                super(a00, a01, a02, a10, a11, a12, a20, a21, 1);
205        }
206        
207        /**
208         * Creates a projective mapping from any linear mapping.
209         * The transformation matrix gets normalized to a22 = 1.
210         * @param m a linear mapping
211         */
212        public ProjectiveMapping2D(LinearMapping2D m) {
213                this(m.normalize());
214        }
215        
216        /**
217         * Creates a projective mapping from an existing instance.
218         * @param m a projective mapping
219         */
220        public ProjectiveMapping2D(ProjectiveMapping2D m) {
221                //this(m.getTransformationMatrix());
222                this(m.a00, m.a01, m.a02, m.a10, m.a11, m.a12, m.a20, m.a21);
223        }
224        
225        // ----------------------------------------------------------
226        
227        /**
228         * Concatenates this mapping A with another linear mapping B and returns
229         * a new mapping C, such that C(x) = B(A(x)).
230         * @param B the second mapping
231         * @return the concatenated mapping
232         */
233        public ProjectiveMapping2D concat(ProjectiveMapping2D B) {
234                LinearMapping2D C = LinearMapping2D.concatenate(B, this);
235                return new ProjectiveMapping2D(C);
236        }
237        
238        /**
239         * {@inheritDoc}
240         * @return a new projective mapping
241         */
242        public ProjectiveMapping2D duplicate() {
243                return new ProjectiveMapping2D(this);
244        }
245        
246        /**
247         * {@inheritDoc}
248         * Note that the inverse A' of a projective transformation matrix A is again a linear transformation
249         * but its a'2' element is generally not 1. Scaling A' to A'' = A' / a22' yields a projective transformation
250         * that reverses A. While A * A' = I, the result of A * A'' is a scaled identity matrix.
251         * @return the inverse projective transformation
252         */
253        public ProjectiveMapping2D getInverse() {
254                return new ProjectiveMapping2D(super.getInverse());
255        }
256        
257        @Override
258        public double[][] getJacobian(Point xy) {
259                // see Baker 2003 "20 Years" Part 1, Eq. 99 (p. 46)
260                final double x = xy.getX();
261                final double y = xy.getY();
262                final double a = a00 * x + a01 * y + a02;       // = alpha
263                final double b = a10 * x + a11 * y + a12;       // = beta
264                final double c = a20 * x + a21 * y + 1;         // = gamma
265                if (Arithmetic.isZero(c)) {
266                        throw new ArithmeticException("getJacobian(): division by zero!");
267                }
268                final double cc = c * c;
269                return new double[][]
270                        {{x/c, y/c, 0,   0,   -(x*a)/cc, -(y*a)/cc, 1/c, 0  },
271                         {0,   0,   x/c, y/c, -(x*b)/cc, -(y*b)/cc, 0,   1/c}};
272        }

282
283                // book example:
284                Point[] P = {
285                                Point.create(2, 5),
286                                Point.create(4, 6),
287                                Point.create(7, 9),
288                                Point.create(5, 9),
289                                Point.create(5.2, 9.1)  // 5 points, overdetermined!
290                                };
291                
292                Point[] Q = {
293                                Point.create(4, 3),
294                                Point.create(5, 2),
295                                Point.create(9, 3),
296                                Point.create(7, 5),
297                                Point.create(7, 4.9)    // 5 points, overdetermined!
298                                };
299                
300                ProjectiveMapping2D pm = ProjectiveMapping2D.fromPoints(P, Q);
301                
302                System.out.println("\nprojective mapping = \n" + pm.toString());
303                
304                for (int i = 0; i < P.length; i++) {
305                        Point Bi = pm.applyTo(P[i]);
306                        System.out.println(P[i].toString() + " -> " + Bi.toString());
307                }
308                
309                System.out.println("pm is of class " + pm.getClass().getName());
310                ProjectiveMapping2D pmi = pm.getInverse();
311                pmi = pmi.normalize();
312                System.out.println("\ninverse projective mapping (normalized) = \n" + pmi.toString());
313                
314                for (int i = 0; i < Q.length; i++) {
315                        Point Ai = pmi.applyTo(Q[i]);
316                        System.out.println(Q[i].toString() + " -> " + Ai.toString());
317                }
318                
319                ProjectiveMapping2D testId = pm.concat(pmi);
320                System.out.println("\ntest: should be a scaled identity matrix: = \n" + testId.toString());
321        }
