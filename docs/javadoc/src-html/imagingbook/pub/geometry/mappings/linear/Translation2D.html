010package imagingbook.pub.geometry.mappings.linear;
011
012import imagingbook.pub.geometry.basic.Point;
013
014/**
015 * This class represents a pure 2D translation (as a special case of 
016 * affine transformation).
017 * It can be assumed that every instance of this class is indeed a translation.
018 */
019public class Translation2D extends AffineMapping2D {
020
021        /**
022         * Constructor. Creates a new translation mapping.
023         * @param tx translation in x
024         * @param ty translation in y
025         */
026        public Translation2D(double tx, double ty){
027                super(1, 0, tx, 0, 1, ty);
028        }
029
030        /** 
031         * Constructor. Creates a new translation instance from a given translation.
032         * @param m a translation
033         */
034        public Translation2D(Translation2D m) {
035                this(m.a02, m.a12);
036        }
037        
038        /**
039         * Creates a new translation that maps between the two given points.
040         * @param p the first point
041         * @param q the second point
042         * @return a new translation instance
043         */
044        public static Translation2D fromPoints(Point p, Point q) {
045                return new Translation2D(q.getX() - p.getX(), q.getY() - p.getY());
046        }
047        
048        // ----------------------------------------------------------
049
050        @Override
051        public Translation2D duplicate() {
052                return new Translation2D(this);
053        }
054        
055        /**
056         * Concatenates this translation A with another translation B and returns
057         * a new translation C, such that C(x) = B(A(x)).
058         * @param B the second translation
059         * @return the concatenated translations
060         */
061        public Translation2D concat(Translation2D B) {
062                return new Translation2D(this.a02 + B.a02, this.a12 + B.a12);
063        }
064        
065        @Override
066        public Translation2D getInverse() {
067                return new Translation2D(-this.a02, -this.a12);
068        }
069
070        @Override
071        public double[][] getJacobian(Point xy) {
072                // this mapping has a constant Jacobian (indep. of xy)
073                return new double[][]
074                        {{1, 0},
075                         {0, 1}};
076        }
077}
