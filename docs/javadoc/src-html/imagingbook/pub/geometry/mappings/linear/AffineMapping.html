010package imagingbook.pub.geometry.mappings.linear;
011
012import java.awt.geom.Point2D;
013
014/**
015 * This class represents aa affine transformation in 2D, which can be defined 
016 * by three pairs of corresponding points.
017 */
018public class AffineMapping extends ProjectiveMapping {
019
020        /**
021         * Creates the identity mapping.
022         */
023        public AffineMapping() {
024                super();
025        }
026
027        /**
028         * Creates an affine mapping from the specified matrix elements.
029         * @param a00 matrix element A_00
030         * @param a01 matrix element A_01
031         * @param a02 matrix element A_02
032         * @param a10 matrix element A_10
033         * @param a11 matrix element A_11
034         * @param a12 matrix element A_12
035         * @param inv set true if this mapping represents an inverse transformation
036         */
037        public AffineMapping (
038                        double a00, double a01, double a02, 
039                        double a10, double a11, double a12, 
040                        boolean inv) {
041                super(a00, a01, a02, a10, a11, a12, 0, 0, inv);
042        }
043
044        /**
045         * Creates a new affine mapping from an existing affine mapping.
046         * @param am a given projective mapping
047         */
048        public AffineMapping(AffineMapping am) {
049                super(am);
050                //              a20 = 0;
051                //              a21 = 0;
052                //              a22 = 1;
053        }
054
055        /**
056         * Creates a projective mapping between arbitrary triangles A, B.
057         * @param A1 point 1 of source triangle A
058         * @param A2 point 2 of source triangle A
059         * @param A3 point 3 of source triangle A
060         * @param B1 point 1 of source triangle B
061         * @param B2 point 2 of source triangle B
062         * @param B3 point 3 of source triangle B
063         */
064        public AffineMapping(Point2D A1, Point2D A2, Point2D A3, Point2D B1, Point2D B2, Point2D B3) {
065                super();
066                double ax1 = A1.getX(), ax2 = A2.getX(), ax3 = A3.getX();
067                double ay1 = A1.getY(), ay2 = A2.getY(), ay3 = A3.getY();
068                double bx1 = B1.getX(), bx2 = B2.getX(), bx3 = B3.getX();
069                double by1 = B1.getY(), by2 = B2.getY(), by3 = B3.getY();
070
071                double S = ax1 * (ay3 - ay2) + ax2 * (ay1 - ay3) + ax3 * (ay2 - ay1); // TODO: check S for zero value and throw exception!
072                a00 = (ay1 * (bx2 - bx3) + ay2 * (bx3 - bx1) + ay3 * (bx1 - bx2)) / S;
073                a01 = (ax1 * (bx3 - bx2) + ax2 * (bx1 - bx3) + ax3 * (bx2 - bx1)) / S;
074                a10 = (ay1 * (by2 - by3) + ay2 * (by3 - by1) + ay3 * (by1 - by2)) / S;
075                a11 = (ax1 * (by3 - by2) + ax2 * (by1 - by3) + ax3 * (by2 - by1)) / S;
076                a02 = (ax1*(ay3*bx2-ay2*bx3) + ax2*(ay1*bx3-ay3*bx1) + ax3*(ay2*bx1-ay1*bx2)) / S;
077                a12 = (ax1*(ay3*by2-ay2*by3) + ax2*(ay1*by3-ay3*by1) + ax3*(ay2*by1-ay1*by2)) / S;
078        }
079
080        /**
081         * Creates a projective mapping between arbitrary triangles A, B.
082         * @param A points of the source triangle
083         * @param B points of the target triangle
084         */
085        public AffineMapping(Point2D[] A, Point2D[] B) {
086                this(A[0], A[1], A[2], B[0], B[1], B[2]);
087        }
088
089        /**
090         * Concatenates this mapping A with another linear mapping B and returns
091         * a new mapping C, such that C(x) = B(A(x)).
092         * @param B the second mapping
093         * @return the concatenated mapping
094         */
095        public AffineMapping concat(AffineMapping B) {  // TODO: check if needed! Generics?
096                AffineMapping A = new AffineMapping(this);
097                A.concatDestructive(B);
098                return A;
099        }
100
101        @Override
102        public AffineMapping invert() {
103                AffineMapping pm = new AffineMapping(this);
104                pm.invertDestructive();
105                return pm;
106        }
107
108        @Override
109        public AffineMapping duplicate() {
110                return new AffineMapping(this);
111        }
112
113        // warp parameter support (used in Lucas-Kanade-matcher) --------------------------
114
115        @Override
116        public int getWarpParameterCount() {
117                return 6;
118        }
119
120        @Override
121        public double[] getWarpParameters() {
122                double[] p = new double[] {
123                                a00 - 1, 
124                                a01,
125                                a10,
126                                a11 - 1,
127                                a02,
128                                a12 };
129                return p;
130        }
131
132        @Override
133        public void setWarpParameters(double[] p) {
134                a00 = p[0] + 1;
135                a01 = p[1];
136                a10 = p[2];
137                a11 = p[3] + 1;
138                a02 = p[4];
139                a12 = p[5];
140        }
141
142        @Override
143        public double[][] getWarpJacobian(double[] xy) {
144                final double x = xy[0];
145                final double y = xy[1];
146                return new double[][]
147                                {{x, y, 0, 0, 1, 0},
148                        {0, 0, x, y, 0, 1}};
149        }
150
151}
