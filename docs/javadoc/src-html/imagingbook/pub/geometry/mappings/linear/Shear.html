010package imagingbook.pub.geometry.mappings.linear;
011
012
013public class Shear extends AffineMapping {
014
015        /**
016         * Creates a new {@link AffineMapping} that represents a shear
017         * transform.
018         * @param bx shear factor in x-direction
019         * @param by shear factor in y-direction
020         */
021        public Shear(double bx, double by) {
022                super( // calls constructor of AffineMapping
023                        1,  bx, 0,
024                        by, 1,  0);
025        }
026        
027        public Shear(Shear sh) {
028                super(sh);
029        }
030        
031        /**
032         * {@inheritDoc}
033         * @return a new shear mapping
034         */
035        public Shear duplicate() {
036                return new Shear(this);
037        }
038        
039}
