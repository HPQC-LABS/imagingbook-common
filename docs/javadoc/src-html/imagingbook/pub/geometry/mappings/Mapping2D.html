001package imagingbook.pub.geometry.mappings;
002
003import imagingbook.pub.geometry.basic.Point;
004
005public interface Mapping2D extends Cloneable {
006        
007        /**
008         * Applies this mapping to a single 2D point.
009         * 
010         * @param pnt the original point
011         * @return the transformed point
012         */
013        Point applyTo (Point pnt);
014        
015        default Point[] applyTo(Point[] pnts) {
016                Point[] outPnts = new Point[pnts.length];
017                for (int i = 0; i < pnts.length; i++) {
018                        outPnts[i] = applyTo(pnts[i]);
019                }
020                return outPnts;
021        }
022        
023        /**
024         * The inverse of this mapping is calculated (if possible)
025         * and returned. Implementing classes are supposed to
026         * override this default method.
027         * 
028         * @return the inverse mapping
029         */
030        default Mapping2D getInverse() {
031                throw new UnsupportedOperationException("Cannot invert mapping " + this.toString());
032        }
033
034        /**
035         * Returns the Jacobian matrix for this mapping, evaluated at
036         * the given 2D point.
037         * This method is only implemented for selected mappings.
038         * @param xy the 2D position to calculate the Jacobian for
039         * @return the Jacobian matrix
040         */
041        default double[][] getJacobian(Point xy) {
042                throw new UnsupportedOperationException("No Jacobian available for this mapping");
043        }
044}
