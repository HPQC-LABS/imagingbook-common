010package imagingbook.pub.geometry.mappings.nonlinear;
011
012import imagingbook.pub.geometry.basic.Point;
013import imagingbook.pub.geometry.mappings.Mapping2D;
014
015/**
016 * A non-linear mapping that produces a ripple effect.
017 * The transformation is implicitly inverted, i.e., maps target to source image
018 * coordinates.
019 * 
020 * @author WB
021 *
022 */
023public class RippleMapping implements Mapping2D {
024        final double xWavel; // = 20;
025        final double yWavel ; // = 100;
026        final double xAmpl ; // = 0;
027        final double yAmpl ; // = 10;
028   
029        public RippleMapping (double xWavel, double xAmpl, double yWavel, double yAmpl) {
030                this.xWavel = xWavel;
031                this.yWavel = yWavel;
032                this.xAmpl = xAmpl;
033                this.yAmpl = yAmpl;
034        }
035
036        @Override
037        public Point applyTo(Point pnt) {
038                final double x = pnt.getX();
039                final double y = pnt.getY();
040                double xx = x + xAmpl * Math.sin(y / xWavel);
041                double yy = y + yAmpl * Math.sin(x / yWavel);
042                return Point.create(xx, yy);
043        }
044}
