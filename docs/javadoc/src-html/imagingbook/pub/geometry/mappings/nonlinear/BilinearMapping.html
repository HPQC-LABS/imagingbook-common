010package imagingbook.pub.geometry.mappings.nonlinear;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.pub.geometry.basic.Point;
014import imagingbook.pub.geometry.mappings.Mapping2D;
015
016/**
017 * This class represents a bilinear transformation in 2D with 8
018 * parameters a0, ..., a3, b0, ..., b3 of the form
019 * <pre>
020 * x' = a0 * x + a1 * y + a2 * x * y + a3
021 * y' = b0 * x + b1 * y + b2 * x * y + b3
022 * </pre>
023 * Note that this is a non-linear transformation because of the mixed term.
024 */
025public class BilinearMapping implements Mapping2D {
026        
027        private final double a0, a1, a2, a3;
028        private final double b0, b1, b2, b3;
029        
030        public BilinearMapping(
031                        double a0, double a1, double a2, double a3,
032                        double b0, double b1, double b2, double b3) {
033                this.a0 = a0;   this.a1 = a1;   this.a2 = a2;   this.a3 = a3;
034                this.b0 = b0;   this.b1 = b1;   this.b2 = b2;   this.b3 = b3;           
035        }
036        
037        /**
038         * Calculates and returns the bilinear mapping M between two point
039         * sets P, Q, with 4 points each, such that q_i = M(p_i).
040         * The inverse mapping is obtained by simply swapping the two point sets.
041         * @param P the first point sequence
042         * @param Q the second point sequence
043         * @return a new bilinear mapping
044         */
045        public static BilinearMapping fromPoints(Point[] P, Point[] Q) {        
046                //define column vectors x, y
047                double[] x = {Q[0].getX(), Q[1].getX(), Q[2].getX(), Q[3].getX()};
048                double[] y = {Q[0].getY(), Q[1].getY(), Q[2].getY(), Q[3].getY()};              
049                // mount matrix M
050                double[][] M = new double[][]
051                        {{P[0].getX(), P[0].getY(), P[0].getX() * P[0].getY(), 1},
052                         {P[1].getX(), P[1].getY(), P[1].getX() * P[1].getY(), 1},
053                         {P[2].getX(), P[2].getY(), P[2].getX() * P[2].getY(), 1},
054                         {P[3].getX(), P[3].getY(), P[3].getX() * P[3].getY(), 1}};
055                double[] a = Matrix.solve(M, x);                // solve x = M * a = x (a is unknown)
056                double[] b = Matrix.solve(M, y);                // solve y = M * b = y (b is unknown)           
057                double a1 = a[0];               double b1 = b[0];
058                double a2 = a[1];               double b2 = b[1];
059                double a3 = a[2];               double b3 = b[2];
060                double a4 = a[3];               double b4 = b[3];
061                return new BilinearMapping(a1, a2, a3, a4, b1, b2, b3, b4);
062        }
063
064        @Override
065        public Point applyTo(Point pnt) {
066                final double x = pnt.getX();
067                final double y = pnt.getY();
068                double xx = a0 * x + a1 * y + a2 * x * y + a3;
069                double yy = b0 * x + b1 * y + b2 * x * y + b3;
070                return Point.create(xx, yy);
071        }
072        
073        public String toString() {
074                return String.format(
075                                "BilinearMapping[A = (%.3f, %.3f, %.3f, %.3f) / B = (%.3f, %.3f, %.3f, %.3f)]",
076                                a0, a1, a2, a3, b0, b1, b2, b3);
077        }
078        
079        // ------------------------------------------------------------------------
080        
081        /**
082         * For testing only.
083         * @param args ignored
084         */
085        public static void main(String[] args) {
086                Point[] P = {
087                                Point.create(2,5),
088                                Point.create(4,6),
089                                Point.create(7,9),
090                                Point.create(5,9),
091                                };
092                
093                Point[] Q = {
094                                Point.create(4,3),
095                                Point.create(5,2),
096                                Point.create(9,3),
097                                Point.create(7,5),
098                                };
099                
100                BilinearMapping bm = fromPoints(P, Q);
101                System.out.println("\nbilinear mapping = \n" + bm.toString());
102                
103                for (int i = 0; i < P.length; i++) {
104                        Point Qi = bm.applyTo(P[i]);
105                        System.out.println(P[i].toString() + " -> " + Qi.toString());
106                }
107                
108        }
109
