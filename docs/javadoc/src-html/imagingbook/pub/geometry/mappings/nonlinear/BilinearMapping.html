010package imagingbook.pub.geometry.mappings.nonlinear;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.pub.geometry.basic.Point;
014import imagingbook.pub.geometry.mappings.Mapping2D;
015
016/**
017 * This class represents a bilinear transformation in 2D with 8
018 * parameters a0, ..., a3, b0, ..., b3 of the form
019 * <pre>
020 * x' = a0 * x + a1 * y + a2 * x * y + a3
021 * y' = b0 * x + b1 * y + b2 * x * y + b3
022 * </pre>
023 * Note that this is a non-linear transformation because of the mixed term.
024 */
025public class BilinearMapping implements Mapping2D {
026        
027        private final double a0, a1, a2, a3;
028        private final double b0, b1, b2, b3;
029        
030        public BilinearMapping(
031                        double a0, double a1, double a2, double a3,
032                        double b0, double b1, double b2, double b3) {
033                this.a0 = a0;   this.a1 = a1;   this.a2 = a2;   this.a3 = a3;
034                this.b0 = b0;   this.b1 = b1;   this.b2 = b2;   this.b3 = b3;           
035        }
036        
037        public static BilinearMapping fromQuads(Point[] P, Point[] Q) {
038                return fromQuads(P[0], P[1], P[2], P[3], Q[0], Q[1], Q[2], Q[3]);
039        }
040
041        /**
042         * Calculates and returns the bilinear mapping M between two point
043         * sets P, Q, with 4 points each, such that q = M(p).
044         * The inverse mapping can be obtained by simply swapping the two point sets.
045         * @param P1 point P1
046         * @param P2 point P2
047         * @param P3 point P3
048         * @param P4 point P4
049         * @param Q1 point Q1
050         * @param Q2 point Q2
051         * @param Q3 point Q3
052         * @param Q4 point Q4
053         * @return a new bilinear mapping
054         */
055        public static BilinearMapping fromQuads(
056                        Point P1, Point P2, Point P3, Point P4, // source quad
057                        Point Q1, Point Q2, Point Q3, Point Q4) // target quad
058                {       
059                //define column vectors x, y
060                double[] x = {Q1.getX(), Q2.getX(), Q3.getX(), Q4.getX()};
061                double[] y = {Q1.getY(), Q2.getY(), Q3.getY(), Q4.getY()};
062                
063                //define matrix M
064                double[][] M = new double[][]
065                        {{P1.getX(), P1.getY(), P1.getX() * P1.getY(), 1},
066                         {P2.getX(), P2.getY(), P2.getX() * P2.getY(), 1},
067                         {P3.getX(), P3.getY(), P3.getX() * P3.getY(), 1},
068                         {P4.getX(), P4.getY(), P4.getX() * P4.getY(), 1}};
069                double[] a = Matrix.solve(M, x);                // solve x = M * a = x (a is unknown)
070                double[] b = Matrix.solve(M, y);                // solve y = M * b = y (b is unknown)           
071                double a1 = a[0];               double b1 = b[0];
072                double a2 = a[1];               double b2 = b[1];
073                double a3 = a[2];               double b3 = b[2];
074                double a4 = a[3];               double b4 = b[3];
075                return new BilinearMapping(a1, a2, a3, a4, b1, b2, b3, b4);
076        }
077        
078
079        @Override
080        public Point applyTo(Point pnt) {
081                final double x = pnt.getX();
082                final double y = pnt.getY();
083                double xx = a0 * x + a1 * y + a2 * x * y + a3;
084                double yy = b0 * x + b1 * y + b2 * x * y + b3;
085                return Point.create(xx, yy);
086        }
087

093//      }       
094        
095        public String toString() {
096                return String.format(
097                                "A = (%.3f, %.3f, %.3f, %.3f) / B = (%.3f, %.3f, %.3f, %.3f)",
098                                a0, a1, a2, a3, b0, b1, b2, b3);
099        }
100        
101        // ------------------------------------------------------------------------
102        
103        /**
104         * For testing only.
105         * @param args ignored
106         */
107        public static void main(String[] args) {
108                Point[] P = {
109                                Point.create(2,5),
110                                Point.create(4,6),
111                                Point.create(7,9),
112                                Point.create(5,9),
113                                };
114                
115                Point[] Q = {
116                                Point.create(4,3),
117                                Point.create(5,2),
118                                Point.create(9,3),
119                                Point.create(7,5),
120                                };
121                
122                BilinearMapping bm = fromQuads(P, Q);
123                System.out.println("\nbilinear mapping = \n" + bm.toString());
124                
125                for (int i = 0; i < P.length; i++) {
126                        Point Qi = bm.applyTo(P[i]);
127                        System.out.println(P[i].toString() + " -> " + Qi.toString());
128                }
129                
130        }
