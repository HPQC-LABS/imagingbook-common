010package imagingbook.pub.geometry.mappings.nonlinear;
011
012import imagingbook.pub.geometry.basic.Point;
013import imagingbook.pub.geometry.mappings.Mapping2D;
014
015/**
016 * A non-linear mapping that produces a spherical distortion.
017 * The transformation is implicitly inverted, i.e., maps target to source image
018 * coordinates.
019 * 
020 * @author WB
021 *
022 */
023public class SphereMapping implements Mapping2D {
024        
025        static double DefaultRefIdx = 1.8;
026                        
027        private final double xc;                        // center of sphere
028        private final double yc;
029        private final double rad;                       // radius of sphere
030        private final double refIdx;            // refraction index
031   
032        public SphereMapping(double xc, double yc, double rad) {
033                this(xc, yc, rad, DefaultRefIdx);
034        }
035        
036        public SphereMapping(double xc, double yc, double rad, double refIdx) {
037                this.xc = xc;
038                this.yc = yc;
039                this.rad = rad;
040                this.refIdx = refIdx;
041        }
042
043        @Override
044        public Point applyTo(Point pnt) {
045                final double x = pnt.getX();
046                final double y = pnt.getY();
047                double dx = x - xc;
048                double dy = y - yc;
049                double dx2 = dx * dx;
050                double dy2 = dy * dy;
051                double rad2 = rad * rad;
052
053                double r2 = dx * dx + dy * dy;
054
055                if (r2 > 0 && r2 < rad2) {
056                        double z2 = rad2 - r2;
057                        double z = Math.sqrt(z2);
058
059                        double xAlpha = Math.asin(dx / Math.sqrt(dx2 + z2));
060                        double xBeta = xAlpha - xAlpha * (1 / refIdx);
061                        double x1 = x - z * Math.tan(xBeta);
062
063                        double yAlpha = Math.asin(dy / Math.sqrt(dy2 + z2));
064                        double yBeta = yAlpha - yAlpha * (1 / refIdx);
065                        double y1 = y - z * Math.tan(yBeta);
066                        return Point.create(x1, y1);
067                }
068                else { // otherwise leave point unchanged
069                        return Point.create(x, y);
070                }
071        }
072}
