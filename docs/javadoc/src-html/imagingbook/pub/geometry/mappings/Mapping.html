010package imagingbook.pub.geometry.mappings;
011
012import java.awt.geom.Point2D;
013
014/**
015 * Abstract class which represents a general 2D transformation.
016 * @version 2019/01/01
017 */
018public abstract class Mapping implements Cloneable {
019        
020        /** 
021         * Duplicates this mapping by using {@link Object}'s shallow clone method.
022         * Sub-classes are supposed to override this method.
023         * @return a copy of this mapping
024         */
025        public Mapping duplicate() {
026                Mapping copy = null;
027                try {
028                        copy = (Mapping) super.clone();
029                } 
030                catch (CloneNotSupportedException e) {
031                        throw new UnsupportedOperationException("Cannot duplicate mapping " + this.toString());}
032                return copy;
033        }
034        
035        /**
036         * Applies this transformation to the given 2D point.
037         * All sub-classes of {@link Mapping} must implement this method.
038         * 
039         * @param x the x-coordinate of the point to be mapped
040         * @param y the y-coordinate of the point to be mapped
041         * @return the transformed 2D coordinate
042         */
043        public abstract double[] applyTo (double x, double y);
044        
045        /**
046         * Applies this transformation to a 2D point given as
047         * a {@code double} array.
048         * 
049         * @param xy the point to be mapped
050         * @return the transformed 2D coordinate
051         */
052        public double[] applyTo (double[] xy) {
053                return applyTo(xy[0], xy[1]);
054        }
055
056        /**
057         * The inverse of this mapping is calculated (if possible)
058         * and returned.
059         * 
060         * @return the inverse mapping
061         */
062        public Mapping getInverse() {
063                throw new UnsupportedOperationException("Cannot invert mapping " + this.toString());
064        }
065        
066        /**
067         * Applies this mapping to a single 2D point.
068         * 
069         * @param pnt the original point
070         * @return the transformed point
071         */
072        public Point2D applyTo(Point2D pnt) {
073                double[] xy = applyTo(new double[] {pnt.getX(), pnt.getY()});
074                return new Point2D.Double(xy[0], xy[1]);
075        }
076        
077        /**
078         * Applies this mapping to all points in the array of 2D points
079         * (convenience method).
080         * 
081         * @param pnts the original points
082         * @return the transformed points
083         */
084        public Point2D[] applyTo(Point2D[] pnts) {
085                Point2D[] outPnts = new Point2D[pnts.length];
086                for (int i = 0; i < pnts.length; i++) {
087                        outPnts[i] = applyTo(pnts[i]);
088                }
089                return outPnts;
090        }
091}
