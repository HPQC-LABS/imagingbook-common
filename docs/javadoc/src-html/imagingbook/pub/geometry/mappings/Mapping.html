010package imagingbook.pub.geometry.mappings;
011
012import ij.process.ImageProcessor;
013import imagingbook.lib.image.ImageAccessor;
014import imagingbook.lib.interpolation.InterpolationMethod;
015
016import java.awt.Point;
017import java.awt.geom.Point2D;
018
019/**
020 * Abstract class which represents a general 2D transformation.
021 * @version 2019/01/01
022 */
023public abstract class Mapping implements Cloneable {
024        
025        protected boolean isInverseFlag = false;
026        
027        /**
028         * Applies this transformation to the given 2D point.
029         * All sub-classes of {@link Mapping} must implement this method.
030         * 
031         * @param x the x-coordinate of the point to be mapped
032         * @param y the y-coordinate of the point to be mapped
033         * @return the transformed 2D coordinate
034         */
035        public abstract double[] applyTo (double x, double y);
036        
037        /**
038         * Applies this transformation to a 2D point given as
039         * a {@code double} array.
040         * 
041         * @param xy the point to be mapped
042         * @return the transformed 2D coordinate
043         */
044        public double[] applyTo (double[] xy) {
045                return applyTo(xy[0], xy[1]);
046        }
047
048        /**
049         * Returns true if this mapping represents an inverse transformation,
050         * otherwise false. Inverse transformations are needed for target-to-source
051         * image mappings.
052         * 
053         * @return true if this mapping represents an inverse transformation, otherwise false
054         */
055        public boolean isInverse() {
056                return isInverseFlag;
057        }
058
059        /**
060         * If this mapping is an inverse transformation already, a copy of this
061         * mapping is returned. Otherwise its inverse is calculated (if possible)
062         * and returned.
063         * 
064         * @return a copy of this mapping if inverse already, otherwise a newly calculated inverse mapping
065         */
066        public Mapping getInverse() {
067                if (isInverseFlag)
068                        return this.duplicate();
069                else {
070                        return this.invert(); // only linear mappings invert
071                }
072        }
073        
074        /**
075         * Tries to invert this mapping and returns the result. An {@link UnsupportedOperationException}
076         * is thrown if the mapping cannot be inverted.
077         * 
078         * @return the inverse mapping
079         */
080        protected Mapping invert() {
081                throw new UnsupportedOperationException("mapping cannot be inverted");
082        }
083        
084        /**
085         * Applies this mapping to a single 2D point.
086         * 
087         * @param pnt the original point
088         * @return the transformed point
089         */
090        public Point2D applyTo(Point2D pnt) {
091                double[] xy = applyTo(new double[] {pnt.getX(), pnt.getY()});
092                return new Point2D.Double(xy[0], xy[1]);
093        }
094        
095        /**
096         * Applies this mapping to all points in the array of 2D points.
097         * 
098         * @param pnts the original points
099         * @return the transformed points
100         */
101        public Point2D[] applyTo(Point2D[] pnts) {
102                Point2D[] outPnts = new Point2D[pnts.length];
103                for (int i = 0; i < pnts.length; i++) {
104                        outPnts[i] = applyTo(pnts[i]);
105                }
106                return outPnts;
107        }
108
109        /**
110         * Destructively transforms the passed image using this geometric
111         * mapping and the specified pixel interpolation method.
112         * TODO: this should not be here (geometry only)?
113         * 
114         * @param ip target image to which this mapping is applied
115         * @param im interpolation method
116         */
117        public void applyTo(ImageProcessor ip, InterpolationMethod im) {
118                // make a temporary copy of the image:
119                ImageProcessor source = ip.duplicate();
120                ImageProcessor target = ip;
121                applyTo(source, target, im);
122                source = null;
123        }
124
125        /**
126         * Transforms the source image to the target image using this geometric
127         * mapping and the specified pixel interpolation method. Note that source
128         * and target must be different images!
129         * 
130         * @param source input image (not modified)
131         * @param target output image (modified)
132         * @param im interpolation method
133         */
134        public void applyTo(ImageProcessor source, ImageProcessor target, InterpolationMethod im) {
135                if (target == source) {
136                        throw new IllegalArgumentException("Source and target image must not be the same!");
137                }
138                ImageAccessor sourceAcc = ImageAccessor.create(source, null, im);
139                ImageAccessor targetAcc = ImageAccessor.create(target);
140                applyTo(sourceAcc, targetAcc);
141        }
142
143
144        /**
145         * Transforms the source image to the target image using this geometric
146         * mapping and the specified pixel interpolation method.
147         * The two images are passed as instances of {@link ImageAccessor}.
148         * Note that source and target reference different images!
149         * 
150         * @param sourceAcc accessor to the source image
151         * @param targetAcc accessor to the target image
152         */
153        public void applyTo(ImageAccessor sourceAcc, ImageAccessor targetAcc) {
154                if (targetAcc.getProcessor() == sourceAcc.getProcessor()) {
155                        throw new IllegalArgumentException("Source and target image must not be the same!");
156                }
157                Mapping invMap = this.getInverse(); // get inverse mapping
158                ImageProcessor target = targetAcc.getProcessor();
159                final int w = target.getWidth();
160                final int h = target.getHeight();
161                for (int v = 0; v < h; v++) {
162                        for (int u = 0; u < w; u++) {
163                                Point2D sourcePt = invMap.applyTo(new Point(u, v));
164                                float[] val = sourceAcc.getPix(sourcePt.getX(),sourcePt.getY());
165                                targetAcc.setPix(u, v, val);
166                        }
167                }
168        }
169
170        /** 
171         * Duplicates this mapping. Sub-classes are supposed to override
172         * this method.
173         * TODO: Let this method throw an {@link UnsupportedOperationException} instead of returning null!
174         * 
175         * @return a copy of this mapping or null if it cannot be cloned
176         */
177        public Mapping duplicate() {
178                return this.clone();
179        }
180        
181        @Deprecated             // do we need this??
182        protected Mapping clone() {
183                Mapping copy = null;
184                try {
185                        copy = (Mapping) super.clone();
186                } 
187                catch (CloneNotSupportedException e) { }
188                return copy;
189        }
190
191}
