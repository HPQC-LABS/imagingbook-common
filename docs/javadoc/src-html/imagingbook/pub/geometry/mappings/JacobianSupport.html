010package imagingbook.pub.geometry.mappings;
011
012/**
013 * This interfaces defines methods for the parameterization of mappings
014 * with 1D double vectors (used in Lucas-Kanade matcher etc.).
015 * Currently only implemented by mapping classes {@code ProjectiveMapping}, 
016 * {@code AffineMapping} and {@code Translation}.
017 * This functionality is used, e.g., in package {@link imagingbook.pub.lucaskanade}.
018 * 
019 * <p> Use of an interface allows implementation by other (non-linear) mappings
020 * that ore not sub-classes of {@code ProjectiveMapping}.</p>
021 */
022public interface JacobianSupport {
023        
024        /**
025         * Returns the parameters of the associated geometric transformation
026         * as a vector.
027         * @return parameter vector
028         */
029        public double[] getParameters();
030        
031        /**
032         * Creates a new instance of this mapping from a parameter vector.
033         * Note that this is a non-static method, i.e., must be called
034         * on an instance of this mapping (which is usually available).
035         * @param p parameter vector
036         * @return a new instance of this mapping
037         */
038        public Mapping fromParameters(double[] p);
039        
040        /** Returns the Jacobian matrix of this geometric transformation,
041         * evaluated for the given 2D position.
042         * @param xy 2D position
043         * @return the Jacobian matrix
044         */
045        public double[][] getJacobian(double[] xy);
046        
047}
