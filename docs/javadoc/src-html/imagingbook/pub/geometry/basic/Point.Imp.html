001package imagingbook.pub.geometry.basic;
002
003import static imagingbook.lib.math.Arithmetic.sqr;
004
005import java.awt.geom.Point2D;
006import java.util.Collection;
007
008/** 
009 * Interface specifying the behavior of a simple 2D point.
010 * It is used to accommodate different (legacy) point implementations 
011 * a common API.
012 * This file also defines a nested default implementation class which
013 * simply extends {@link Point2D.Double}.
014 * 
015 */
016public interface Point {
017        double getX();
018        double getY();
019        
020        static Point create(double x, double y) {
021                return new Imp(x, y);
022        }
023        
024        static Point create(double[] xy) {
025                return new Imp(xy[0], xy[1]);
026        }
027        
028        static Point create(Point p) {
029                return new Imp(p.getX(), p.getY());
030        }
031        
032        static Point create(Point2D p) {
033                return new Imp(p.getX(), p.getY());
034        }
035        
036        // ----------------------------------------------------------
037        
038        default double[] toArray() {
039                return new double[] {this.getX(), this.getY()};
040        }
041        
042        static Point[] toArray(Collection<Point> points) {
043                return points.toArray(new Point[0]);
044        }
045        
046        static double distance(Point p, Point q) {
047                return Math.sqrt(distance2(p, q));
048        }
049        
050        static double distance2(Point p, Point q) {
051                return sqr(q.getX() - p.getX()) + sqr(q.getY() - p.getY());
052        }
053        
054        default double distance2(Point other) {
055                return distance2(this, other);
056        }
057        
058        default double distance(Point other) {
059                return distance(this, other);
060        }
061        
062        
063        /**
064         * Simple fallback implementation of the {@link Point} interface.
065         * If dependency on AWT is to be avoided, just implement your own.
066         */
067        class Imp extends java.awt.geom.Point2D.Double implements Point {
068                private static final long serialVersionUID = 1L;
069
070                public Imp(double x, double y) {
071                        super(x, y);
072                }
073                
074                @Override
075                public String toString() {
076                        return String.format("Point[%.3f, %.3f]", this.getX(), this.getY());
077                }
078        }
079
080}
