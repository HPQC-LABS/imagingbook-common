001package imagingbook.pub.geometry.delaunay;
002
003import java.util.Collection;
004
005import imagingbook.pub.geometry.basic.Point;
006
007/**
008 * This class defines static utility methods used by the Delaunay implementations.
009 */
010public abstract class Utils {
011        
012        /**
013         * Creates a 2D triangle that is sufficiently large to be used as 
014         * an outer triangle for the Delaunay triangulation of the given
015         * point set.
016         * @param points the 2D point set
017         * @return a triangle as an array of 3 points
018         */
019        public static Point[] makeOuterTriangle(Collection<Point> points) {
020                double xmin = Double.POSITIVE_INFINITY;
021                double xmax = Double.NEGATIVE_INFINITY;
022                double ymin = xmin;
023                double ymax = xmax;
024                
025                for (Point p : points) {
026                        double x = p.getX();
027                        double y = p.getY();
028                        xmin = Math.min(x, xmin);
029                        xmax = Math.max(x, xmax);
030                        ymin = Math.min(y, ymin);
031                        ymax = Math.max(y, ymax);
032                }
033                return makeOuterTriangle(xmin, xmax, ymin, ymax);
034        }
035        
036        /**
037         * Creates a 2D triangle that is sufficiently large to be used as 
038         * an outer triangle for the Delaunay triangulation of points 
039         * inside the given bounding rectangle.
040         * @param xmin minimum x-coordinate of the bounding rectangle
041         * @param xmax maximum x-coordinate of the bounding rectangle
042         * @param ymin minimum y-coordinate of the bounding rectangle
043         * @param ymax maximum y-coordinate of the bounding rectangle
044         * @return a triangle as an array of 3 points
045         */
046        public static Point[] makeOuterTriangle(double xmin, double xmax, double ymin, double ymax) {
047                double width = xmax - xmin;
048                double height = ymax - ymin;
049                double diam = Math.max(width,  height);
050                double xc = xmin + width / 2;
051                double yc = ymin + height / 2;
052                double s = 50;
053                return new Point[] {
054                                Point.create(xc, yc + s * diam),
055                                Point.create(xc + s * diam, yc),
056                                Point.create(xc - s * diam, yc - s * diam)
057                };
058        }
059        
060        /**
061         * Creates a 2D triangle that is sufficiently large to be used as 
062         * an outer triangle for the Delaunay triangulation of points 
063         * inside the given bounding rectangle, anchored at (0,0).
064         * @param width the width of the bounding rectangle
065         * @param height the height of the bounding rectangle
066         * @return a triangle as an array of 3 points
067         */
068        public static Point[] makeOuterTriangle(double width, double height) {
069                return makeOuterTriangle(0, width, 0, height);
070        }
071
072}
