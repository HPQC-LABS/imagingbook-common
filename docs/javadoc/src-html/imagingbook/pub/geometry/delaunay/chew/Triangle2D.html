023import java.util.Arrays;
024import java.util.Collection;
025import java.util.Iterator;
026import java.util.NoSuchElementException;
027
028import imagingbook.pub.geometry.basic.Point;
029import imagingbook.pub.geometry.basic.Triangle;
030
031
032/**
033 * A Triangle is an immutable Set of exactly three Pnts.
034 *
035 * All Set operations are available. Individual vertices can be accessed via
036 * iterator() and also via triangle.get(index).
037 *
038 * Note that, even if two triangles have the same vertex set, they are
039 * *different* triangles. Methods equals() and hashCode() are consistent with
040 * this rule.
041 *
042 * @author Paul Chew
043 *
044 * Created December 2007. Replaced general simplices with geometric triangle.
045 *
046 * Adapted by W. Burger (2019)
047 * @version 2020-01-01
048 */
049public class Triangle2D extends ArraySet<Pnt> implements Triangle {
050
051    private int idNumber;                   // The id number
052    private Pnt circumcenter = null;        // The triangle's circumcenter
053
054    private static int idGenerator = 0;     // Used to create id numbers
055    public static boolean moreInfo = true; // True iff more info in toString
056
057    /**
058     * @param vertices the vertices of the Triangle.
059     * @throws IllegalArgumentException if there are not three distinct vertices
060     */
061    public Triangle2D (Pnt... vertices) {
062        this(Arrays.asList(vertices));
063    }
064
065    /**
066     * @param collection a Collection holding the Simplex vertices
067     * @throws IllegalArgumentException if there are not three distinct vertices
068     */
069    public Triangle2D (Collection<? extends Pnt> collection) {
070        super(collection);
071        idNumber = idGenerator++;
072        if (this.size() != 3)
073            throw new IllegalArgumentException("Triangle must have 3 vertices");
074    }
075    
076        public Triangle2D(Point[] points) {
077                this(new Pnt(points[0]), new Pnt(points[1]), new Pnt(points[2]));
078        }
079    
080    // ----------------------------------------------------------------------------
081
082    @Override
083    public String toString () {
084        if (!moreInfo) return "Triangle" + idNumber;
085        return "Triangle" + idNumber + super.toString();
086    }
087
088    /**
089     * Get arbitrary vertex of this triangle, but not any of the bad vertices.
090     * @param badVertices one or more bad vertices
091     * @return a vertex of this triangle, but not one of the bad vertices
092     * @throws NoSuchElementException if no vertex found
093     */
094    public Pnt getVertexButNot (Pnt... badVertices) {
095        Collection<Pnt> bad = Arrays.asList(badVertices);
096        for (Pnt v: this) if (!bad.contains(v)) return v;
097        throw new NoSuchElementException("No vertex found");
098    }
099
100    /**
101     * True iff triangles are neighbors. Two triangles are neighbors if they
102     * share a facet.
103     * @param triangle the other Triangle
104     * @return true iff this Triangle is a neighbor of triangle
105     */
106    public boolean isNeighbor (Triangle2D triangle) {
107        int count = 0;
108        for (Pnt vertex: this)
109            if (!triangle.contains(vertex)) count++;
110        return count == 1;
111    }
112
113    /**
114     * Report the facet opposite vertex.
115     * @param vertex a vertex of this Triangle
116     * @return the facet opposite vertex
117     * @throws IllegalArgumentException if the vertex is not in triangle
118     */
119    public ArraySet<Pnt> facetOpposite (Pnt vertex) {
120        ArraySet<Pnt> facet = new ArraySet<Pnt>(this);
121        if (!facet.remove(vertex))
122            throw new IllegalArgumentException("Vertex not in triangle");
123        return facet;
124    }
125
126    /**
127     * @return the triangle's circumcenter
128     */
129    public Pnt getCircumcenter () {
130        if (circumcenter == null)
131            circumcenter = Pnt.circumcenter(this.toArray(new Pnt[0]));
132        return circumcenter;
133    }
134
135    /* The following two methods ensure that a Triangle is immutable */
136
137    @Override
138    public boolean add (Pnt vertex) {
139        throw new UnsupportedOperationException();
140    }
141
142    @Override
143    public Iterator<Pnt> iterator () {
144        return new Iterator<Pnt>() {
145            private Iterator<Pnt> it = Triangle2D.super.iterator();
146            public boolean hasNext() {return it.hasNext();}
147            public Pnt next() {return it.next();}
148            public void remove() {throw new UnsupportedOperationException();}
149        };
150    }
151
152    /* The following two methods ensure that all triangles are different. */
153
154    @Override
155    public int hashCode () {
156        return (int)(idNumber^(idNumber>>>32));
157    }
158
159    @Override
160    public boolean equals (Object o) {
161        return (this == o);
162    }
163
164    // method required by {@link Triangle} interface:
165    
166        @Override
167        public Point[] getPoints() {
168                return this.toArray(new Point[0]);
169        }
170
171}
