023import java.util.AbstractSet;
024import java.util.ArrayList;
025import java.util.Collection;
026import java.util.Iterator;
027
028/**
029 * An ArrayList implementation of Set. An ArraySet is good for small sets; it
030 * has less overhead than a HashSet or a TreeSet.
031 *
032 * @author Paul Chew
033 *
034 * Created December 2007.  For use with Voronoi/Delaunay applet.
035 *
036 */
037public class ArraySet<E> extends AbstractSet<E> {
038
039    private ArrayList<E> items;            // Items of the set
040
041    /**
042     * Create an empty set (default initial capacity is 3).
043     */
044    public ArraySet () {
045        this(3);
046    }
047
048    /**
049     * Create an empty set with the specified initial capacity.
050     * @param initialCapacity the initial capacity
051     */
052    public ArraySet (int initialCapacity) {
053        items  = new ArrayList<E>(initialCapacity);
054    }
055
056    /**
057     * Create a set containing the items of the collection.  Any duplicate
058     * items are discarded.
059     * @param collection the source for the items of the small set
060     */
061    public ArraySet (Collection<? extends E> collection) {
062        items = new ArrayList<E>(collection.size());
063        for (E item: collection)
064            if (!items.contains(item)) items.add(item);
065    }
066
067    /**
068     * Get the item at the specified index.
069     * @param index where the item is located in the ListSet
070     * @return the item at the specified index
071     * @throws IndexOutOfBoundsException if the index is out of bounds
072     */
073    public E get (int index) throws IndexOutOfBoundsException {
074        return items.get(index);
075    }
076
077    /**
078     * True iff any member of the collection is also in the ArraySet.
079     * @param collection the Collection to check
080     * @return true iff any member of collection appears in this ArraySet
081     */
082    public boolean containsAny (Collection<?> collection) {
083        for (Object item: collection)
084            if (this.contains(item)) return true;
085        return false;
086    }
087
088    @Override
089    public boolean add(E item) {
090        if (items.contains(item)) return false;
091        return items.add(item);
092    }
093
094    @Override
095    public Iterator<E> iterator() {
096        return items.iterator();
097    }
098
099    @Override
100    public int size() {
101        return items.size();
102    }
103
104}
