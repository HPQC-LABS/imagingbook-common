067 * <p>
068 * <strong>Original description:</strong> The actual data structure here is a Delaunay Triangulation. The Voronoi Diagram is built on-the-fly
069 * from the Delaunay Triangulation. The Delaunay Triangulation is built within a large triangle whose vertices
070 * are well off-screen. That's why in the Delaunay Triangulation there are lines heading off-screen. 
071 * This technique makes the code simpler since otherwise additional code would be needed to handle new sites
072 * that are outside the convex hull of the previous sites.
073 * </p>
074 * <p>
075 * The algorithm: To insert a new site, we walk across the triangulation, starting from the most recently created
076 * triangle, until we find the triangle that contains the new site. This triangle and any adjacent triangles that
077 * contain this new site in their circumcircle are eliminated and the resulting empty spot is re-triangulated.
078 * The site-insertion part of this technique is commonly called the Bowyer-Watson Algorithm. The expected time to
079 * insert a new site is roughly O(n1/2) where n is the current number of sites.
080 * </p>
081 * 
082 * Adapted by W. Burger
083 * @version 2020-01-02
084 */
085public class TriangulationChew extends AbstractSet<Triangle2D> implements DelaunayTriangulation {

122    /**
123     * All sites must fall within the initial triangle.
124     * @param superTriangle the initial triangle
125     */
126    private TriangulationChew (Triangle2D superTriangle) {
127        this.superTriangle = superTriangle;
128        this.triGraph = new Graph<Triangle2D>();
129        this.triGraph.add(superTriangle);
130        this.mostRecent = superTriangle;
131        this.points = new ArrayList<>();
132    }

153    /**
154     * True iff triangle is a member of this triangulation.
155     * This method isn't required by AbstractSet, but it improves efficiency.
156     * @param triangle the object to check for membership
157     */
158    public boolean contains (Object triangle) {
159        return triGraph.nodeSet().contains(triangle);

162    /**
163     * Report neighbor opposite the given vertex of triangle.
164     * @param site a vertex of triangle
165     * @param triangle we want the neighbor of this triangle
166     * @return the neighbor opposite site in triangle; null if none
167     * @throws IllegalArgumentException if site is not in this triangle
168     */
169    public Triangle2D neighborOpposite (Pnt site, Triangle2D triangle) {
170        if (!triangle.contains(site))
171            throw new IllegalArgumentException("Bad vertex; not in triangle");
172        for (Triangle2D neighbor: triGraph.neighbors(triangle)) {
173            if (!neighbor.contains(site)) return neighbor;
174        }
175        return null;
176    }

178    /**
179     * Return the set of triangles adjacent to triangle.
180     * @param triangle the triangle to check
181     * @return the neighbors of triangle
182     */
183    public Set<Triangle2D> neighbors(Triangle2D triangle) {
184        return triGraph.neighbors(triangle);
185    }

187    /**
188     * Report triangles surrounding site in order (cw or ccw).
189     * @param site we want the surrounding triangles for this site
190     * @param triangle a "starting" triangle that has site as a vertex
191     * @return all triangles surrounding site in order (cw or ccw)
192     * @throws IllegalArgumentException if site is not in triangle
193     */
194    public List<Triangle2D> surroundingTriangles (Pnt site, Triangle2D triangle) {
195        if (!triangle.contains(site))
196            throw new IllegalArgumentException("Site not in triangle");
197        List<Triangle2D> list = new ArrayList<Triangle2D>();
198        Triangle2D start = triangle;
199        Pnt guide = triangle.getVertexButNot(site);        // Affects cw or ccw
200        while (true) {
201            list.add(triangle);
202            Triangle2D previous = triangle;
203            triangle = this.neighborOpposite(guide, triangle); // Next triangle
204            guide = previous.getVertexButNot(site, guide);     // Update guide
205            if (triangle == start) break;
206        }
207        return list;
208    }

210    /**
211     * Locate the triangle with point inside it or on its boundary.
212     * @param point the point to locate
213     * @return the triangle that holds point; null if no such triangle
214     */
215    public Triangle2D locate (Pnt point) {
216        Triangle2D triangle = mostRecent;
217        if (!this.contains(triangle)) {
218                triangle = null;
219        }
220        // Try a directed walk (this works fine in 2D, but can fail in 3D)
221        Set<Triangle2D> visited = new HashSet<Triangle2D>();
222        while (triangle != null) {
223            if (visited.contains(triangle)) { // This should never happen
224                System.out.println("Warning: Caught in a locate loop");
225                break;
226            }
227            visited.add(triangle);
228            // Corner opposite point
229            Pnt corner = point.isOutside(triangle.toArray(new Pnt[0]));
230            if (corner == null) return triangle;
231            triangle = this.neighborOpposite(corner, triangle);
232        }
233        // No luck; try brute force
234        System.out.println("Warning: Checking all triangles for " + point);
235        for (Triangle2D tri: this) {
236            if (point.isOutside(tri.toArray(new Pnt[0])) == null) {
237                return tri;
238            }
239        }
240        // No such triangle
241        System.out.println("Warning: No triangle holds " + point);
242        return null;
243    }

245    public void insertAll(Collection<Point> pointSet) {
246        for (Point p : pointSet) {
247                insert(p);
248        }
249    }
250
251    /**
252     * Place a new site into the DT.
253     * Nothing happens if the site matches an existing DT vertex.
254     * Uses straightforward scheme rather than best asymptotic time.
255     * Throws an {@link IllegalArgumentException} if p does not lie in any triangle.
256     * @param p the point to be inserted
257     */
258    public void insert(Point p) {
259        Pnt vec = new Pnt(p);
260        // Locate containing triangle
261        Triangle2D triangle = locate(vec);
262        // Give up if no containing triangle or if site is already in DT
263        if (triangle == null)
264            throw new IllegalArgumentException("No containing triangle");
265//        if (triangle.contains(vec)) 
266//              return;
267        if (!triangle.contains(vec)) {
268                // Determine the cavity and update the triangulation
269                Set<Triangle2D> cavity = getCavity(vec, triangle);
270                mostRecent = update(vec, cavity);
271        }
272        points.add(p);
273    }

275    /**
276     * Determine the cavity caused by site.
277     * @param site the site causing the cavity
278     * @param triangle the triangle containing site
279     * @return set of all triangles that have site in their circumcircle
280     */
281    private Set<Triangle2D> getCavity (Pnt site, Triangle2D triangle) {
282        Set<Triangle2D> encroached = new HashSet<Triangle2D>();
283        Queue<Triangle2D> toBeChecked = new LinkedList<Triangle2D>();
284        Set<Triangle2D> marked = new HashSet<Triangle2D>();
285        toBeChecked.add(triangle);
286        marked.add(triangle);
287        while (!toBeChecked.isEmpty()) {
288            triangle = toBeChecked.remove();
289            if (site.vsCircumcircle(triangle.toArray(new Pnt[0])) == 1)
290                continue; // Site outside triangle => triangle not in cavity
291            encroached.add(triangle);
292            // Check the neighbors
293            for (Triangle2D neighbor: triGraph.neighbors(triangle)){
294                if (marked.contains(neighbor)) continue;
295                marked.add(neighbor);
296                toBeChecked.add(neighbor);
297            }
298        }
299        return encroached;
300    }
301
302    /**
303     * Update the triangulation by removing the cavity triangles and then
304     * filling the cavity with new triangles.
305     * @param site the site that created the cavity
306     * @param cavity the triangles with site in their circumcircle
307     * @return one of the new triangles
308     */
309    private Triangle2D update (Pnt site, Set<Triangle2D> cavity) {
310        Set<Set<Pnt>> boundary = new HashSet<Set<Pnt>>();
311        Set<Triangle2D> theTriangles = new HashSet<Triangle2D>();
312
313        // Find boundary facets and adjacent triangles
314        for (Triangle2D triangle: cavity) {
315            theTriangles.addAll(neighbors(triangle));
316            for (Pnt vertex: triangle) {
317                Set<Pnt> facet = triangle.facetOpposite(vertex);
318                if (boundary.contains(facet)) boundary.remove(facet);
319                else boundary.add(facet);
320            }
321        }
322        theTriangles.removeAll(cavity);        // Adj triangles only
323
324        // Remove the cavity triangles from the triangulation
325        for (Triangle2D triangle: cavity) triGraph.remove(triangle);
326
327        // Build each new triangle and add it to the triangulation
328        Set<Triangle2D> newTriangles = new HashSet<Triangle2D>();
329        for (Set<Pnt> vertices: boundary) {
330            vertices.add(site);
331            Triangle2D tri = new Triangle2D(vertices);
332            triGraph.add(tri);
333            newTriangles.add(tri);
334        }
335
336        // Update the graph links for each new triangle
337        theTriangles.addAll(newTriangles);    // Adj triangle + new triangles
338        for (Triangle2D triangle: newTriangles)
339            for (Triangle2D other: theTriangles)
340                if (triangle.isNeighbor(other))
341                    triGraph.add(triangle, other);
342
343        // Return one of the new triangles
344        return newTriangles.iterator().next();
345    }

347    /**
348     * Checks if the given triangle shares a vertex with the triangulation's
349     * initial ('super') triangle.
350     * @param otherTriangle the other triangle
351     * @return {@code true} iff there is a common vertex
352     */
353    private boolean connectsToSuperTriangle(Triangle2D otherTriangle) {
354        boolean connects = false;
355        for (Pnt vertex : superTriangle) {
356            if (otherTriangle.contains(vertex)) {
357                connects = true;
358                break;
359            }
360        }
361        return connects;
362    }
363    
364        @Override
365        public List<Triangle> getTriangles() {
366                return this.stream().filter(t -> !connectsToSuperTriangle(t)).collect(Collectors.toList());
367        }
368        
369        @Override
370        public List<Point> getPoints() {
371                return Collections.unmodifiableList(points);
372        }
