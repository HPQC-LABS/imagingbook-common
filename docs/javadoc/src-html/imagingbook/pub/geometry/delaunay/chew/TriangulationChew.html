069 * <p>
070 * <strong>Original description:</strong> The actual data structure here is a Delaunay Triangulation. The Voronoi Diagram is built on-the-fly
071 * from the Delaunay Triangulation. The Delaunay Triangulation is built within a large triangle whose vertices
072 * are well off-screen. That's why in the Delaunay Triangulation there are lines heading off-screen. 
073 * This technique makes the code simpler since otherwise additional code would be needed to handle new sites
074 * that are outside the convex hull of the previous sites.
075 * </p>
076 * <p>
077 * The algorithm: To insert a new site, we walk across the triangulation, starting from the most recently created
078 * triangle, until we find the triangle that contains the new site. This triangle and any adjacent triangles that
079 * contain this new site in their circumcircle are eliminated and the resulting empty spot is re-triangulated.
080 * The site-insertion part of this technique is commonly called the Bowyer-Watson Algorithm. The expected time to
081 * insert a new site is roughly O(n1/2) where n is the current number of sites.
082 * </p>
083 * 
084 * Adapted by W. Burger
085 * @version 2020-01-02
086 */

124    /**
125     * All sites must fall within the initial triangle.
126     * @param superTriangle the initial triangle
127     */
128    private TriangulationChew (Triangle2D superTriangle) {
129        this.superTriangle = superTriangle;
130        this.triGraph = new Graph<Triangle2D>();
131        this.triGraph.add(superTriangle);
132        this.mostRecent = superTriangle;
133        this.points = new ArrayList<>();
134    }

168    /**
169     * Report neighbor opposite the given vertex of triangle.
170     * @param site a vertex of triangle
171     * @param triangle we want the neighbor of this triangle
172     * @return the neighbor opposite site in triangle; null if none
173     * @throws IllegalArgumentException if site is not in this triangle
174     */
175    public Triangle2D neighborOpposite (Pnt site, Triangle2D triangle) {
176        if (!triangle.contains(site))
177            throw new IllegalArgumentException("Bad vertex; not in triangle");
178        for (Triangle2D neighbor: triGraph.neighbors(triangle)) {
179            if (!neighbor.contains(site)) return neighbor;
180        }
181        return null;
182    }

184    /**
185     * Return the set of triangles adjacent to triangle.
186     * @param triangle the triangle to check
187     * @return the neighbors of triangle
188     */
189    public Set<Triangle2D> neighbors(Triangle2D triangle) {
190        return triGraph.neighbors(triangle);
191    }

193    /**
194     * Report triangles surrounding site in order (cw or ccw).
195     * @param site we want the surrounding triangles for this site
196     * @param triangle a "starting" triangle that has site as a vertex
197     * @return all triangles surrounding site in order (cw or ccw)
198     * @throws IllegalArgumentException if site is not in triangle
199     */
200    public List<Triangle2D> surroundingTriangles (Pnt site, Triangle2D triangle) {
201        if (!triangle.contains(site))
202            throw new IllegalArgumentException("Site not in triangle");
203        List<Triangle2D> list = new ArrayList<Triangle2D>();
204        Triangle2D start = triangle;
205        Pnt guide = triangle.getVertexButNot(site);        // Affects cw or ccw
206        while (true) {
207            list.add(triangle);
208            Triangle2D previous = triangle;
209            triangle = this.neighborOpposite(guide, triangle); // Next triangle
210            guide = previous.getVertexButNot(site, guide);     // Update guide
211            if (triangle == start) break;
212        }
213        return list;
214    }

216    /**
217     * Locate the triangle with point inside it or on its boundary.
218     * @param point the point to locate
219     * @return the triangle that holds point; null if no such triangle
220     */
221    public Triangle2D locate (Pnt point) {
222        Triangle2D triangle = mostRecent;
223        if (!this.contains(triangle)) {
224                triangle = null;
225        }
226        // Try a directed walk (this works fine in 2D, but can fail in 3D)
227        Set<Triangle2D> visited = new HashSet<Triangle2D>();
228        while (triangle != null) {
229            if (visited.contains(triangle)) { // This should never happen
230                System.out.println("Warning: Caught in a locate loop");
231                break;
232            }
233            visited.add(triangle);
234            // Corner opposite point
235            Pnt corner = point.isOutside(triangle.toArray(new Pnt[0]));
236            if (corner == null) return triangle;
237            triangle = this.neighborOpposite(corner, triangle);
238        }
239        // No luck; try brute force
240        System.out.println("Warning: Checking all triangles for " + point);
241        for (Triangle2D tri: this) {
242            if (point.isOutside(tri.toArray(new Pnt[0])) == null) {
243                return tri;
244            }
245        }
246        // No such triangle
247        System.out.println("Warning: No triangle holds " + point);
248        return null;
249    }

251    public void insertAll(Collection<Point> pointSet) {
252        for (Point p : pointSet) {
253                insert(p);
254        }
255    }

257    /**
258     * Place a new site into the DT.
259     * Nothing happens if the site matches an existing DT vertex.
260     * Uses straightforward scheme rather than best asymptotic time.
261     * Throws an {@link IllegalArgumentException} if p does not lie in any triangle.
262     * @param p the point to be inserted
263     */
264    public void insert(Point p) {
265        Pnt vec = new Pnt(p);
266        // Locate containing triangle
267        Triangle2D triangle = locate(vec);
268        // Give up if no containing triangle or if site is already in DT
269        if (triangle == null)
270            throw new IllegalArgumentException("No containing triangle");
271//        if (triangle.contains(vec)) 
272//              return;
273        if (!triangle.contains(vec)) {
274                // Determine the cavity and update the triangulation
275                Set<Triangle2D> cavity = getCavity(vec, triangle);
276                mostRecent = update(vec, cavity);
277        }
278        points.add(p);
279    }

281    /**
282     * Determine the cavity caused by site.
283     * @param site the site causing the cavity
284     * @param triangle the triangle containing site
285     * @return set of all triangles that have site in their circumcircle
286     */
287    private Set<Triangle2D> getCavity (Pnt site, Triangle2D triangle) {
288        Set<Triangle2D> encroached = new HashSet<Triangle2D>();
289        Queue<Triangle2D> toBeChecked = new LinkedList<Triangle2D>();
290        Set<Triangle2D> marked = new HashSet<Triangle2D>();
291        toBeChecked.add(triangle);
292        marked.add(triangle);
293        while (!toBeChecked.isEmpty()) {
294            triangle = toBeChecked.remove();
295            if (site.vsCircumcircle(triangle.toArray(new Pnt[0])) == 1)
296                continue; // Site outside triangle => triangle not in cavity
297            encroached.add(triangle);
298            // Check the neighbors
299            for (Triangle2D neighbor: triGraph.neighbors(triangle)){
300                if (marked.contains(neighbor)) continue;
301                marked.add(neighbor);
302                toBeChecked.add(neighbor);
303            }
304        }
305        return encroached;
306    }
307
308    /**
309     * Update the triangulation by removing the cavity triangles and then
310     * filling the cavity with new triangles.
311     * @param site the site that created the cavity
312     * @param cavity the triangles with site in their circumcircle
313     * @return one of the new triangles
314     */
315    private Triangle2D update (Pnt site, Set<Triangle2D> cavity) {
316        Set<Set<Pnt>> boundary = new HashSet<Set<Pnt>>();
317        Set<Triangle2D> theTriangles = new HashSet<Triangle2D>();
318
319        // Find boundary facets and adjacent triangles
320        for (Triangle2D triangle: cavity) {
321            theTriangles.addAll(neighbors(triangle));
322            for (Pnt vertex: triangle) {
323                Set<Pnt> facet = triangle.facetOpposite(vertex);
324                if (boundary.contains(facet)) boundary.remove(facet);
325                else boundary.add(facet);
326            }
327        }
328        theTriangles.removeAll(cavity);        // Adj triangles only
329
330        // Remove the cavity triangles from the triangulation
331        for (Triangle2D triangle: cavity) triGraph.remove(triangle);
332
333        // Build each new triangle and add it to the triangulation
334        Set<Triangle2D> newTriangles = new HashSet<Triangle2D>();
335        for (Set<Pnt> vertices: boundary) {
336            vertices.add(site);
337            Triangle2D tri = new Triangle2D(vertices);
338            triGraph.add(tri);
339            newTriangles.add(tri);
340        }
341
342        // Update the graph links for each new triangle
343        theTriangles.addAll(newTriangles);    // Adj triangle + new triangles
344        for (Triangle2D triangle: newTriangles)
345            for (Triangle2D other: theTriangles)
346                if (triangle.isNeighbor(other))
347                    triGraph.add(triangle, other);
348
349        // Return one of the new triangles
350        return newTriangles.iterator().next();
351    }

353    /**
354     * Checks if the given triangle shares a vertex with the triangulation's
355     * initial ('super') triangle.
356     * @param otherTriangle the other triangle
357     * @return {@code true} iff there is a common vertex
358     */
359    private boolean connectsToSuperTriangle(Triangle2D otherTriangle) {
360        boolean connects = false;
361        for (Pnt vertex : superTriangle) {
362            if (otherTriangle.contains(vertex)) {
363                connects = true;
364                break;
365            }
366        }
367        return connects;
368    }

370        @Override
371        public List<Triangle> getTriangles() {
372                return this.stream().filter(t -> !connectsToSuperTriangle(t)).collect(Collectors.toList());
373        }

375        @Override
376        public List<Point> getPoints() {
377                return Collections.unmodifiableList(points);
378        }
