001package imagingbook.pub.geometry.delaunay.impGuibas;
002
003import java.util.Arrays;
004
005import imagingbook.pub.geometry.delaunay.common.Point;
006import imagingbook.pub.geometry.delaunay.common.Triangle;
007
008
009/**
010 * Represents a 2D triangle, specified by three corner points.
011 * Instances of this class are immutable.
012 */
013public class Triangle2D implements Triangle {
014
015        protected final Vector2D a;
016        protected final Vector2D b;
017        protected final Vector2D c;
018        private final boolean isOrientedCCW;
019
020        /**
021         * Constructor of the 2D triangle class used to create a new triangle instance
022         * from three 2D vectors describing the triangle's vertices.
023         * 
024         * @param a The first vertex of the triangle
025         * @param b The second vertex of the triangle
026         * @param c The third vertex of the triangle
027         */
028        public Triangle2D(Vector2D a, Vector2D b, Vector2D c) {
029                this.a = a;
030                this.b = b;
031                this.c = c;
032                isOrientedCCW = findIfOrientedCCW();
033        }
034        
035        public Triangle2D(Point[] points) {
036                this(new Vector2D(points[0]), new Vector2D(points[1]), new Vector2D(points[2]));
037        }
038
039        /**
040         * Tests if a 2D point lies inside this 2D triangle. See See Christer Ericson, 
041         * Real-Time Collision Detection, CRC Press, 2004 (Ch. 5, p. 206).
042         * 
043         * @param point the point to be checked
044         * @return {@code true} iff the point lies inside this 2D triangle
045         */
046        public boolean containsPoint(Vector2D point) {
047                double pab = point.sub(a).cross(b.sub(a));
048                double pbc = point.sub(b).cross(c.sub(b));
049                if (!hasSameSign(pab, pbc)) {
050                        return false;
051                }
052                double pca = point.sub(c).cross(a.sub(c));
053                if (!hasSameSign(pab, pca)) {
054                        return false;
055                }
056                return true;
057        }
058
059        /**
060         * Tests if a given point lies in the circumcircle of this triangle. Let the
061         * triangle ABC appear in counterclockwise (CCW) order. Then when det &gt; 0,
062         * the point lies inside the circumcircle through the three points a, b and c.
063         * If instead det &lt; 0, the point lies outside the circumcircle. When det = 0,
064         * the four points are cocircular. If the triangle is oriented clockwise (CW)
065         * the result is reversed. 
066         * See Christer Ericson, Real-Time Collision Detection, CRC Press, 2004 (Ch. 3, p. 34).
067         * 
068         * @param point the point to be checked
069         * @return {@code true} iff the point lies inside the circumcircle through the
070         *         three points a, b, and c of the triangle
071         */
072        protected boolean isPointInCircumCircle(Vector2D point) {
073                final double a11 = a.getX() - point.getX();
074                final double a21 = b.getX() - point.getX();
075                final double a31 = c.getX() - point.getX();
076
077                final double a12 = a.getY() - point.getY();
078                final double a22 = b.getY() - point.getY();
079                final double a32 = c.getY() - point.getY();
080                
081                final double a13 = a11 * a11 + a12 * a12;
082                final double a23 = a21 * a21 + a22 * a22;
083                final double a33 = a31 * a31 + a32 * a32;
084
085                final double det = 
086                                a11 * a22 * a33 + a12 * a23 * a31 + 
087                                a13 * a21 * a32 - a13 * a22 * a31 - 
088                                a12 * a21 * a33 - a11 * a23 * a32;
089
090                return isOrientedCCW ? det > 0.0 : det < 0.0;
091        }
092
093        /**
094         * Tests if a given point lies in the circumcircle of this triangle. Let the
095         * triangle ABC appear in counterclockwise (CCW) order. Then when det &gt; 0,
096         * the point lies inside the circumcircle through the three points a, b and c.
097         * If instead det &lt; 0, the point lies outside the circumcircle. When det = 0,
098         * the four points are cocircular. If the triangle is oriented clockwise (CW)
099         * the result is reversed. 
100         * See Christer Ericson, Real-Time Collision Detection, CRC Press, 2004 (Ch. 3, p. 32).
101         * Since triangles are immutable, this property can be pre-calculated.
102         * 
103         * @return {@code true} iff the triangle abc is oriented counterclockwise (CCW)
104         */
105        private boolean findIfOrientedCCW() {
106                double a11 = a.getX() - c.getX();
107                double a21 = b.getX() - c.getX();
108                double a12 = a.getY() - c.getY();
109                double a22 = b.getY() - c.getY();
110                double det = a11 * a22 - a12 * a21;
111                return det > 0.0;
112        }
113
114        /**
115         * Test if this triangle is oriented counterclockwise (CCW).
116         * This property is pre-calculated.
117         * 
118         * @return {@code true} iff the triangle ABC is oriented counterclockwise (CCW)
119         */
120        protected boolean isOrientedCCW() {
121                return isOrientedCCW;
122        }
123
124        /**
125         * Returns {@code true} if this triangle contains the given edge.
126         * 
127         * @param edge the edge to be tested
128         * @return {@code true} iff this triangle contains the specified edge
129         */
130        public boolean containsEdge(Edge2D edge) {
131                return (a == edge.a || b == edge.a || c == edge.a) && (a == edge.b || b == edge.b || c == edge.b);
132        }
133
134        /**
135         * Returns the vertex of this triangle opposite to the specified edge.
136         * 
137         * @param edge the edge (which must be contained in this triangle)
138         * @return the triangle vertex opposite to the specified edge
139         */
140        public Vector2D getOppositeVertex(Edge2D edge) {
141                final Vector2D p1 = edge.a;
142                final Vector2D p2 = edge.b;
143                if ((a == p1 && b == p2) || (a == p2 && b == p1)) {
144                        return c;
145                }
146                if ((a == p1 && c == p2) || (a == p2 && c == p1)) {
147                        return b;
148                }
149                if ((b == p1 && c == p2) || (b == p2 && c == p1)) {
150                        return a;
151                } 
152                throw new IllegalArgumentException("Specified edge is not part of this triangle");
153        }
154
155        /**
156         * Checks if the given vertex is amongst the triangle's vertices.
157         * 
158         * @param vertex the vertex to be checked
159         * @return {@code true} if the vertex is one of the corners of this triangle
160         */
161        public boolean hasVertex(Vector2D vertex) {
162                return (a == vertex || b == vertex || c == vertex);
163        }
164
165        /**
166         * Calculates the minimum distance from the specified point to this triangle.
167         * The result is returned as an {@link Edge2D.Distance} instance, representing 
168         * the point's distance to the closest edge of this triangle.
169         * 
170         * @param point the point to be checked
171         * @return the edge of this triangle that is closest to the specified point
172         */
173        public Edge2D.Distance findMinEdgeDistance(Vector2D point) {
174                Edge2D.Distance[] distances = {
175                                new Edge2D(a, b).distanceFromPoint(point),
176                                new Edge2D(b, c).distanceFromPoint(point),
177                                new Edge2D(c, a).distanceFromPoint(point)       
178                        };
179                Arrays.sort(distances);
180                return distances[0];
181        }
182
183        /**
184         * Tests if the two arguments have the same sign.
185         * @param a first quantity
186         * @param b second quantity
187         * @return {@code true} iff both arguments have the same sign
188         */
189        private boolean hasSameSign(double a, double b) {
190                return Math.signum(a) == Math.signum(b);
191                //return a * b > 0;
192        }
193
194        @Override
195        public String toString() {
196                return Triangle2D.class.getSimpleName() + "[" + a + ", " + b + ", " + c + "]";
197        }
198
199        @Override
200        public Point[] getPoints() {
201                return new Point[] {a, b, c};
202        }
203
204}
