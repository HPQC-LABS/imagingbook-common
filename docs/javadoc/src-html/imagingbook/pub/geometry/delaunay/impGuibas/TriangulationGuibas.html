053public class TriangulationGuibas implements DelaunayTriangulation {
054
055        private final List<Vector2D> points;
056        private final List<Triangle2D> triangles;
057        private final Triangle2D outerTriangle;
058
059        /**
060         * Constructor. 
061         * @param pointSet the point set to be triangulated
062         * @param shuffle set {@code true} to randomly shuffle the input points
063         */
064        public TriangulationGuibas(List<Point> pointSet, boolean shuffle) {
065                if (pointSet == null || pointSet.size() < 3) {
066                        throw new IllegalArgumentException("Point set must contain at least 3 points.");
067                }
068                this.points = makePointset(pointSet, shuffle);
069                this.triangles = new ArrayList<>();
070                this.outerTriangle = new Triangle2D(Utils.makeOuterTriangle(pointSet));
071                triangulate();
072        }
073        
074        /**
075         * The simplest constructor. The supplied points will be inserted
076         * in their original order (without shuffling).
077         * @param pointSet the point set to be triangulated
078         */
079        public TriangulationGuibas(List<Point> pointSet) {
080                this(pointSet, false);
081        }
082        
083        // -----------------------------------------------------------------------------
084        
085        /** 
086         * Converts the incoming points (of unknown type but implementing the {@link Point} interface)
087         * to instances of the local implementation ({@link Vector2D}).
088         * 
089         * @param inPoints the input points (must implement {@link Point})
090         * @param shuffle if {@code true}, the input point sequence is randomly permuted
091         * @return the new point sequence
092         */
093        private List<Vector2D> makePointset(Collection<? extends Point> inPoints, boolean shuffle) {
094                Vector2D[] outPoints = new Vector2D[inPoints.size()];
095                int i = 0;
096                for (Point ip : inPoints) {
097                        outPoints[i] = new Vector2D(ip);
098                        i++;
099                }
100                List<Vector2D> outList = Arrays.asList(outPoints);
101                if (shuffle) {
102                        Collections.shuffle(outList);   // random permutation of pointset (in-place)
103                }
104                return outList;
105        }
106
107        private void triangulate() {
108                triangles.add(outerTriangle);
109                
110                for (Vector2D pnt : points) {
111
112                        Triangle2D triangle = this.findContainingTriangle(pnt);
113
114                        if (triangle == null) { // pnt is outside of any triangle
115                                /* If no containing triangle exists, then the vertex is not inside a triangle
116                                 * (this can also happen due to numerical errors) and lies on an edge. In order
117                                 * to find this edge we search all edges of the triangle soup and select the one
118                                 * which is nearest to the point we try to add. This edge is removed and four
119                                 * new edges are added.
120                                 */
121                                Edge2D edge = this.findNearestEdge(pnt);
122
123                                Triangle2D tr1 = this.findOneTriangleSharing(edge);
124                                Triangle2D tr2 = this.findNeighbour(tr1, edge);
125
126                                Vector2D noneEdgeVertex1 = tr1.getOppositeVertex(edge);
127                                Vector2D noneEdgeVertex2 = tr2.getOppositeVertex(edge);
128
129                                triangles.remove(tr1);
130                                triangles.remove(tr2);
131
132                                Triangle2D triangle1 = new Triangle2D(edge.a, noneEdgeVertex1, pnt);
133                                Triangle2D triangle2 = new Triangle2D(edge.b, noneEdgeVertex1, pnt);
134                                Triangle2D triangle3 = new Triangle2D(edge.a, noneEdgeVertex2, pnt);
135                                Triangle2D triangle4 = new Triangle2D(edge.b, noneEdgeVertex2, pnt);
136
137                                triangles.add(triangle1);
138                                triangles.add(triangle2);
139                                triangles.add(triangle3);
140                                triangles.add(triangle4);
141
142
143                                legalizeEdge(triangle1, new Edge2D(edge.a, noneEdgeVertex1), pnt);
144                                legalizeEdge(triangle2, new Edge2D(edge.b, noneEdgeVertex1), pnt);
145                                legalizeEdge(triangle3, new Edge2D(edge.a, noneEdgeVertex2), pnt);
146                                legalizeEdge(triangle4, new Edge2D(edge.b, noneEdgeVertex2), pnt);
147                        } 
148                        else { // pnt is inside the triangle <a,b,c>.
149                                Vector2D a = triangle.a;
150                                Vector2D b = triangle.b;
151                                Vector2D c = triangle.c;
152
153                                triangles.remove(triangle);
154
155                                Triangle2D triangle1 = new Triangle2D(a, b, pnt);
156                                Triangle2D triangle2 = new Triangle2D(b, c, pnt);
157                                Triangle2D triangle3 = new Triangle2D(c, a, pnt);
158
159                                triangles.add(triangle1);
160                                triangles.add(triangle2);
161                                triangles.add(triangle3);
162
163                                legalizeEdge(triangle1, new Edge2D(a, b), pnt);
164                                legalizeEdge(triangle2, new Edge2D(b, c), pnt);
165                                legalizeEdge(triangle3, new Edge2D(c, a), pnt);
166                        }
167                }
168
169                // Remove all triangles containing any vertex of the super triangle:
170                this.removeTrianglesUsing(outerTriangle.a);
171                this.removeTrianglesUsing(outerTriangle.b);
172                this.removeTrianglesUsing(outerTriangle.c);
173        }
174
175        /**
176         * This method legalizes edges by recursively flipping all illegal edges.
177         * 
178         * @param triangle  the triangle
179         * @param edge      the edge to be legalized
180         * @param newVertex the new vertex
181         */
182        private void legalizeEdge(Triangle2D triangle, Edge2D edge, Vector2D newVertex) {
183                Triangle2D neighbourTriangle = this.findNeighbour(triangle, edge);
184                // If the triangle has a neighbor, then legalize the edge
185                if (neighbourTriangle != null) {
186                        if (neighbourTriangle.isPointInCircumCircle(newVertex)) {
187                                triangles.remove(triangle);
188                                triangles.remove(neighbourTriangle);
189
190                                Vector2D noneEdgeVertex = neighbourTriangle.getOppositeVertex(edge);
191
192                                Triangle2D triangle1 = new Triangle2D(noneEdgeVertex, edge.a, newVertex);
193                                Triangle2D triangle2 = new Triangle2D(noneEdgeVertex, edge.b, newVertex);
194
195                                triangles.add(triangle1);
196                                triangles.add(triangle2);
197
198                                legalizeEdge(triangle1, new Edge2D(noneEdgeVertex, edge.a), newVertex);
199                                legalizeEdge(triangle2, new Edge2D(noneEdgeVertex, edge.b), newVertex);
200                        }
201                }
202        }

204        /**
205         * Returns the point set in form of a vector of 2D vectors.
206         * 
207         * @return Returns the points set.
208         */
209        public List<Vector2D> getPointSet() {
210                return points;
211        }
212
213        /**
214         * Returns the triangles of the triangulation in form of a vector of 2D
215         * triangles. The initial 'superTriangle' is removed.
216         * The resulting triangles should form a convex structure.
217         * 
218         * @return the triangles of this triangulation.
219         */
220        @Override
221        public List<Triangle> getTriangles() {
222                return Collections.unmodifiableList(triangles);
223        }
224
225        @Override
226        public List<Point> getPoints() {
227                return Collections.unmodifiableList(points);
228        }
229        
230        // triangle-related methods ---------------------------
231
232        /**
233         * Returns the triangle that contains the specified point or null if no
234         * such triangle exists.
235         * @param point the query point
236         * @return the containing triangle or {@code null} if none was found
237         */
238        public Triangle2D findContainingTriangle(Vector2D point) {
239                for (Triangle2D triangle : triangles) {
240                        if (triangle.containsPoint(point)) {
241                                return triangle;
242                        }
243                }
244                return null;
245        }
246
247        /**
248         * Returns the neighboring triangle of the specified triangle sharing the same edge
249         * as specified. If no neighbor sharing the same edge exists {@code null} is returned.
250         * TODO: searching over ALL triangles seems to be unnecessarily expensive
251         * 
252         * @param tri1 the triangle
253         * @param edge the edge
254         * @return the triangle's neighboring triangle sharing the same edge or {@code null} if no
255         *         such triangle exists
256         */
257        public Triangle2D findNeighbour(Triangle2D tri1, Edge2D edge) {
258                for (Triangle2D tri2 : triangles) {
259                        if (tri2.containsEdge(edge) && tri2 != tri1) {
260                                return tri2;
261                        }
262                }
263                return null;
264        }
265
266        /**
267         * Returns one of the possible triangles sharing the specified edge. Based on
268         * the ordering of the triangles in this triangle soup the returned triangle may
269         * differ. To find the other triangle that shares this edge use the
270         * {@code findNeighbour(Triangle2D triangle, Edge2D edge)} method.
271         * @param edge the edge
272         * @return the triangle that shares the specified edge or {@code null} if none exists
273         */
274        public Triangle2D findOneTriangleSharing(Edge2D edge) {
275                for (Triangle2D triangle : triangles) {
276                        if (triangle.containsEdge(edge)) {
277                                return triangle;
278                        }
279                }
280                return null;
281        }
282        
283        /**
284         * Returns the triangle edge nearest to the specified point.
285         * @param point the query point
286         * @return the triangle edge nearest to the specified point
287         */
288        private Edge2D findNearestEdge(Vector2D point) {
289                Edge2D minEdge = null;
290                double minDist = Double.POSITIVE_INFINITY;
291                for (Triangle2D tri : triangles) {
292                        Edge2D.Distance ed = tri.findMinEdgeDistance(point);
293                        double dist = ed.getDistance();
294                        if (dist < minDist) {
295                                minDist = dist;
296                                minEdge = ed.getEdge();
297                        }
298                }
299                return minEdge;
300        }
301
302        /**
303         * Removes all triangles that contain the specified corner point.
304         * @param vertex the corner point
305         */
306        public void removeTrianglesUsing(Vector2D vertex) {
307                List<Triangle2D> trianglesToBeRemoved = new LinkedList<>();
308                for (Triangle2D triangle : triangles) {
309                        if (triangle.hasVertex(vertex)) {
310                                trianglesToBeRemoved.add(triangle);
311                        }
312                }
313                triangles.removeAll(trianglesToBeRemoved);
314        }
