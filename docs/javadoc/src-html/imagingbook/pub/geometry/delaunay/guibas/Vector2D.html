001package imagingbook.pub.geometry.delaunay.guibas;
002
003import imagingbook.pub.geometry.basic.Point;
004
005/**
006 * Implements a 2D point or vector with {@code double} coordinates.
007 * Instances of this class are immutable.
008 */
009public class Vector2D implements Point {
010
011    private final double x;
012    private final double y;
013    
014        /**
015         * Constructor from any object that implements the generic {@link Point} 
016         * interface.
017         * @param pt original point object (of unknown class)
018         */
019        public Vector2D(Point pt) {
020                this.x = pt.getX();
021                this.y = pt.getY();
022        }
023        
024        /**
025         * Constructor.
026         * @param x the x coordinate 
027         * @param y the y coordinate
028         */
029    public Vector2D(double x, double y) {
030        this.x = x;
031        this.y = y;
032    }
033
034        @Override
035        public double getX() {
036                return x;
037        }
038
039        @Override
040        public double getY() {
041                return y;
042        }
043
044        /**
045         * Subtracts the given vector from this vector.
046         * 
047         * @param vector the vector to be subtracted from this vector
048         * @return a new instance holding the result of the vector subtraction
049         */
050    public Vector2D sub(Vector2D vector) {
051        return new Vector2D(this.x - vector.x, this.y - vector.y);
052    }
053
054        /**
055         * Adds the given vector to this vector.
056         * 
057         * @param vector the vector to be added to this vector
058         * @return a new instance holding the result of the vector addition
059         */
060    public Vector2D add(Vector2D vector) {
061        return new Vector2D(this.x + vector.x, this.y + vector.y);
062    }
063
064        /**
065         * Multiplies this vector by the given scalar.
066         * 
067         * @param scalar the scalar to be multiplied by this
068         * @return a new instance holding the result of the multiplication
069         */
070    public Vector2D mult(double scalar) {
071        return new Vector2D(this.x * scalar, this.y * scalar);
072    }
073
074        /**
075         * Computes the magnitude or length of this vector.
076         * 
077         * @return the magnitude of this vector
078         */
079    public double mag() {
080        return Math.sqrt(this.x * this.x + this.y * this.y);
081    }
082
083    /**
084         * Computes the dot product of this vector and the given vector.
085         * 
086         * @param vector the vector to be multiplied by this vector
087         * @return a new instance holding the result of the multiplication
088         */
089    public double dot(Vector2D vector) {
090        return this.x * vector.x + this.y * vector.y;
091    }
092
093        /**
094         * Computes the 2D pseudo cross product Dot(Perp(this), vector) of this vector
095         * and the given vector.
096         * 
097         * @param vector the vector to be multiplied to the perpendicular vector of this vector
098         * @return a new instance holding the result of the pseudo cross product
099         */
100    public double cross(Vector2D vector) {
101        return this.y * vector.x - this.x * vector.y;
102    }
103
104    @Override
105    public String toString() {
106        return Vector2D.class.getSimpleName() + "[" + x + ", " + y + "]";
107    }
108
109}
