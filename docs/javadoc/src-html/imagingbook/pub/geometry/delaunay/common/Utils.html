001package imagingbook.pub.geometry.delaunay.common;
002
003import java.util.Collection;
004
005public abstract class Utils {
006        
007        public static Point[] makeOuterTriangle(Collection<Point> points) {
008                double xmin = Double.POSITIVE_INFINITY;
009                double xmax = Double.NEGATIVE_INFINITY;
010                double ymin = xmin;
011                double ymax = xmax;
012                
013                for (Point p : points) {
014                        double x = p.getX();
015                        double y = p.getY();
016                        xmin = Math.min(x, xmin);
017                        xmax = Math.max(x, xmax);
018                        ymin = Math.min(y, ymin);
019                        ymax = Math.max(y, ymax);
020                }
021                return makeOuterTriangle(xmin, xmax, ymin, ymax);
022        }
023        
024        public static Point[] makeOuterTriangle(double xmin, double xmax, double ymin, double ymax) {
025                double width = xmax - xmin;
026                double height = ymax - ymin;
027                double diam = Math.max(width,  height);
028                double xc = xmin + width / 2;
029                double yc = ymin + height / 2;
030                double s = 50;
031                return new Point[] {
032                                Point.create(xc, yc + s * diam),
033                                Point.create(xc + s * diam, yc),
034                                Point.create(xc - s * diam, yc - s * diam)
035                };
036        }
037        
038        public static Point[] makeOuterTriangle(double width, double height) {
039                return makeOuterTriangle(0, width, 0, height);
040        }
041
042}
