001package imagingbook.pub.geometry.delaunay.impGuibasXX;
002
003/**
004 * 2D edge class implementation.
005 * 
006 * @author Johannes Diemke
007 */
008public class Edge2D {
009
010        protected final Vector2D a;
011        protected final Vector2D b;
012
013        /**
014         * Constructor of the 2D edge class used to create a new edge instance from two
015         * 2D vectors describing the edge's vertices.
016         * 
017         * @param a first vertex of the edge
018         * @param b second vertex of the edge
019         */
020        protected Edge2D(Vector2D a, Vector2D b) {
021                this.a = a;
022                this.b = b;
023        }
024
025        protected double minDistance(Vector2D point) {
026                return getClosestPoint(point).sub(point).mag();
027        }
028
029        private Vector2D getClosestPoint(Vector2D point) {
030                Vector2D ab = b.sub(a);
031                double t = point.sub(a).dot(ab) / ab.dot(ab); // TODO: check for zero denominator?
032                if (t < 0.0) {
033                        t = 0.0;
034                } else if (t > 1.0) {
035                        t = 1.0;
036                }
037                return a.add(ab.mult(t));
038        }
039
040        protected Distance getEdgeDistance(Vector2D point) {
041                return new Distance(point);
042        }
043
044        /**
045         * Edge distance pack class implementation used to describe the distance to a
046         * given edge.
047         * 
048         * @author Johannes Diemke
049         */
050        protected class Distance implements Comparable<Distance> {
051
052                private final double distance;
053
054                protected Distance(double dist) {
055                        this.distance = dist;
056                }
057
058                protected Distance(Vector2D point) {
059                        this(Edge2D.this.minDistance(point));
060                }
061
062                // ---------------------------------------------------
063
064                protected double getDistance() {
065                        return distance;
066                }
067
068                protected Edge2D getEdge() {
069                        return Edge2D.this;
070                }
071
072                @Override
073                public int compareTo(Distance o) {
074                        return Double.compare(this.distance, o.distance);
075                }
076
077        }
078
079}
