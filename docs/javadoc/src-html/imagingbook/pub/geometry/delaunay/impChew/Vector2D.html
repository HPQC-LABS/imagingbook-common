025/**
026 * Points in Euclidean space, implemented as double[].
027 *
028 * Includes simple geometric operations.
029 * Uses matrices; a matrix is represented as an array of Pnts.
030 * Uses simplices; a simplex is represented as an array of Pnts.
031 *
032 * @author Paul Chew

034 * Created July 2005.  Derived from an earlier, messier version.

042public class Vector2D implements Point {

044    private double[] coordinates;          // The point's coordinates
045
046    /**
047     * Constructor.
048     * @param coords the coordinates
049     */
050    public Vector2D (double... coords) {
051        // Copying is done here to ensure that Pnt's coords cannot be altered.
052        // This is necessary because the double... notation actually creates a
053        // constructor with double[] as its argument.
054//        coordinates = new double[coords.length];
055//        System.arraycopy(coords, 0, coordinates, 0, coords.length);
056        coordinates = coords.clone();
057    }
058    
059    public Vector2D (Point p) {
060        this(p.getX(), p.getY());
061    }

065    @Override
066    public String toString () {
067        if (coordinates.length == 0) return "Pnt()";
068        String result = "Pnt(" + coordinates[0];
069        for (int i = 1; i < coordinates.length; i++)
070            result = result + "," + coordinates[i];
071        result = result + ")";
072        return result;
073    }
074
075    @Override
076    public boolean equals (Object other) {
077        if (!(other instanceof Vector2D)) return false;
078        Vector2D p = (Vector2D) other;
079        if (this.coordinates.length != p.coordinates.length) return false;
080        for (int i = 0; i < this.coordinates.length; i++)
081            if (this.coordinates[i] != p.coordinates[i]) return false;
082        return true;
083    }
084
085    @Override
086    public int hashCode () {
087        int hash = 0;
088        for (double c: this.coordinates) {
089            long bits = Double.doubleToLongBits(c);
090            hash = (31*hash) ^ (int)(bits ^ (bits >> 32));
091        }
092        return hash;
093    }

108    public int dimension () {
109        return coordinates.length;

112    /**
113     * Check that dimensions match.
114     * @param p the Pnt to check (against this Pnt)
115     * @return the dimension of the Pnts
116     * @throws IllegalArgumentException if dimension fail to match
117     */
118    public int dimCheck (Vector2D p) {
119        int len = this.coordinates.length;
120        if (len != p.coordinates.length)
121            throw new IllegalArgumentException("Dimension mismatch");
122        return len;
123    }
124
125    /**
126     * Create a new Pnt by adding additional coordinates to this Pnt.
127     * @param coords the new coordinates (added on the right end)
128     * @return a new Pnt with the additional coordinates
129     */
130    public Vector2D extend (double... coords) {
131        double[] result = new double[coordinates.length + coords.length];
132        System.arraycopy(coordinates, 0, result, 0, coordinates.length);
133        System.arraycopy(coords, 0, result, coordinates.length, coords.length);
134        return new Vector2D(result);
135    }

137    /**
138     * Dot product.
139     * @param p the other Pnt
140     * @return dot product of this Pnt and p
141     */
142    public double dot (Vector2D p) {
143        int len = dimCheck(p);
144        double sum = 0;
145        for (int i = 0; i < len; i++)
146            sum += this.coordinates[i] * p.coordinates[i];
147        return sum;
148    }
149
150    /**
151     * Magnitude (as a vector).
152     * @return the Euclidean length of this vector
153     */
154    public double magnitude () {
155        return Math.sqrt(this.dot(this));
156    }
157
158    /**
159     * Subtract.
160     * @param p the other Pnt
161     * @return a new Pnt = this - p
162     */
163    public Vector2D subtract (Vector2D p) {
164        int len = dimCheck(p);
165        double[] coords = new double[len];
166        for (int i = 0; i < len; i++)
167            coords[i] = this.coordinates[i] - p.coordinates[i];
168        return new Vector2D(coords);
169    }
170
171    /**
172     * Add.
173     * @param p the other Pnt
174     * @return a new Pnt = this + p
175     */
176    public Vector2D add (Vector2D p) {
177        int len = dimCheck(p);
178        double[] coords = new double[len];
179        for (int i = 0; i < len; i++)
180            coords[i] = this.coordinates[i] + p.coordinates[i];
181        return new Vector2D(coords);
182    }

184    /**
185     * Angle (in radians) between two Pnts (treated as vectors).
186     * @param p the other Pnt
187     * @return the angle (in radians) between the two Pnts
188     */
189    public double angle (Vector2D p) {
190        return Math.acos(this.dot(p) / (this.magnitude() * p.magnitude()));
191    }
192
193    /**
194     * Perpendicular bisector of two Pnts.
195     * Works in any dimension.  The coefficients are returned as a Pnt of one
196     * higher dimension (e.g., (A,B,C,D) for an equation of the form
197     * Ax + By + Cz + D = 0).
198     * @param point the other point
199     * @return the coefficients of the perpendicular bisector
200     */
201    public Vector2D bisector (Vector2D point) {
202        dimCheck(point);
203        Vector2D diff = this.subtract(point);
204        Vector2D sum = this.add(point);
205        double dot = diff.dot(sum);
206        return diff.extend(-dot / 2);
207    }

223    /**
224     * Compute the determinant of a matrix (array of Pnts).
225     * This is not an efficient implementation, but should be adequate
226     * for low dimension.
227     * @param matrix the matrix as an array of Pnts
228     * @return the determinnant of the input matrix
229     * @throws IllegalArgumentException if dimensions are wrong
230     */
231    public static double determinant (Vector2D[] matrix) {
232        if (matrix.length != matrix[0].dimension())
233            throw new IllegalArgumentException("Matrix is not square");
234        boolean[] columns = new boolean[matrix.length];
235        for (int i = 0; i < matrix.length; i++) columns[i] = true;
236        try {return determinant(matrix, 0, columns);}
237        catch (ArrayIndexOutOfBoundsException e) {
238            throw new IllegalArgumentException("Matrix is wrong shape");
239        }
240    }
241
242    /**
243     * Compute the determinant of a submatrix specified by starting row
244     * and by "active" columns.
245     * @param matrix the matrix as an array of Pnts
246     * @param row the starting row
247     * @param columns a boolean array indicating the "active" columns
248     * @return the determinant of the specified submatrix
249     * @throws ArrayIndexOutOfBoundsException if dimensions are wrong
250     */
251    private static double determinant(Vector2D[] matrix, int row, boolean[] columns){
252        if (row == matrix.length) return 1;
253        double sum = 0;
254        int sign = 1;
255        for (int col = 0; col < columns.length; col++) {
256            if (!columns[col]) continue;
257            columns[col] = false;
258            sum += sign * matrix[row].coordinates[col] *
259                   determinant(matrix, row+1, columns);
260            columns[col] = true;
261            sign = -sign;
262        }
263        return sum;
264    }
265
266    /**
267     * Compute generalized cross-product of the rows of a matrix.
268     * The result is a Pnt perpendicular (as a vector) to each row of
269     * the matrix.  This is not an efficient implementation, but should
270     * be adequate for low dimension.
271     * @param matrix the matrix of Pnts (one less row than the Pnt dimension)
272     * @return a Pnt perpendicular to each row Pnt
273     * @throws IllegalArgumentException if matrix is wrong shape
274     */
275    public static Vector2D cross (Vector2D[] matrix) {
276        int len = matrix.length + 1;
277        if (len != matrix[0].dimension())
278            throw new IllegalArgumentException("Dimension mismatch");
279        boolean[] columns = new boolean[len];
280        for (int i = 0; i < len; i++) columns[i] = true;
281        double[] result = new double[len];
282        int sign = 1;
283        try {
284            for (int i = 0; i < len; i++) {
285                columns[i] = false;
286                result[i] = sign * determinant(matrix, 0, columns);
287                columns[i] = true;
288                sign = -sign;
289            }
290        } catch (ArrayIndexOutOfBoundsException e) {
291            throw new IllegalArgumentException("Matrix is wrong shape");
292        }
293        return new Vector2D(result);
294    }

298    /**
299     * Determine the signed content (i.e., area or volume, etc.) of a simplex.
300     * @param simplex the simplex (as an array of Pnts)
301     * @return the signed content of the simplex
302     */
303    public static double content (Vector2D[] simplex) {
304        Vector2D[] matrix = new Vector2D[simplex.length];
305        for (int i = 0; i < matrix.length; i++)
306            matrix[i] = simplex[i].extend(1);
307        int fact = 1;
308        for (int i = 1; i < matrix.length; i++) fact = fact*i;
309        return determinant(matrix) / fact;
310    }

312    /**
313     * Relation between this Pnt and a simplex (represented as an array of
314     * Pnts). Result is an array of signs, one for each vertex of the simplex,
315     * indicating the relation between the vertex, the vertex's opposite facet,
316     * and this Pnt.
317     *
318     * <pre>
319     *   -1 means Pnt is on same side of facet
320     *    0 means Pnt is on the facet
321     *   +1 means Pnt is on opposite side of facet
322     * </pre>
323     *
324     * @param simplex an array of Pnts representing a simplex
325     * @return an array of signs showing relation between this Pnt and simplex
326     * @throws IllegalArgumentExcpetion if the simplex is degenerate
327     */
328    public int[] relation (Vector2D[] simplex) {
329        /* In 2D, we compute the cross of this matrix:
330         *    1   1   1   1
331         *    p0  a0  b0  c0
332         *    p1  a1  b1  c1
333         * where (a, b, c) is the simplex and p is this Pnt. The result is a
334         * vector in which the first coordinate is the signed area (all signed
335         * areas are off by the same constant factor) of the simplex and the
336         * remaining coordinates are the *negated* signed areas for the
337         * simplices in which p is substituted for each of the vertices.
338         * Analogous results occur in higher dimensions.
339         */
340        int dim = simplex.length - 1;
341        if (this.dimension() != dim)
342            throw new IllegalArgumentException("Dimension mismatch");
343
344        /* Create and load the matrix */
345        Vector2D[] matrix = new Vector2D[dim+1];
346        /* First row */
347        double[] coords = new double[dim+2];
348        for (int j = 0; j < coords.length; j++) coords[j] = 1;
349        matrix[0] = new Vector2D(coords);
350        /* Other rows */
351        for (int i = 0; i < dim; i++) {
352            coords[0] = this.coordinates[i];
353            for (int j = 0; j < simplex.length; j++)
354                coords[j+1] = simplex[j].coordinates[i];
355            matrix[i+1] = new Vector2D(coords);
356        }
357
358        /* Compute and analyze the vector of areas/volumes/contents */
359        Vector2D vector = cross(matrix);
360        double content = vector.coordinates[0];
361        int[] result = new int[dim+1];
362        for (int i = 0; i < result.length; i++) {
363            double value = vector.coordinates[i+1];
364            if (Math.abs(value) <= 1.0e-6 * Math.abs(content)) result[i] = 0;
365            else if (value < 0) result[i] = -1;
366            else result[i] = 1;
367        }
368        if (content < 0) {
369            for (int i = 0; i < result.length; i++)
370                result[i] = -result[i];
371        }
372        if (content == 0) {
373            for (int i = 0; i < result.length; i++)
374                result[i] = Math.abs(result[i]);
375        }
376        return result;
377    }

379    /**
380     * Test if this Pnt is outside of simplex.
381     * @param simplex the simplex (an array of Pnts)
382     * @return simplex Pnt that "witnesses" outsideness (or null if not outside)
383     */
384    public Vector2D isOutside (Vector2D[] simplex) {
385        int[] result = this.relation(simplex);
386        for (int i = 0; i < result.length; i++) {
387            if (result[i] > 0) return simplex[i];
388        }
389        return null;
390    }

392    /**
393     * Test if this Pnt is on a simplex.
394     * @param simplex the simplex (an array of Pnts)
395     * @return the simplex Pnt that "witnesses" on-ness (or null if not on)
396     */
397    public Vector2D isOn (Vector2D[] simplex) {
398        int[] result = this.relation(simplex);
399        Vector2D witness = null;
400        for (int i = 0; i < result.length; i++) {
401            if (result[i] == 0) witness = simplex[i];
402            else if (result[i] > 0) return null;
403        }
404        return witness;
405    }

407    /**
408     * Test if this Pnt is inside a simplex.
409     * @param simplex the simplex (an arary of Pnts)
410     * @return true iff this Pnt is inside simplex.
411     */
412    public boolean isInside (Vector2D[] simplex) {
413        int[] result = this.relation(simplex);
414        for (int r: result) if (r >= 0) return false;
415        return true;
416    }

418    /**
419     * Test relation between this Pnt and circumcircle of a simplex.
420     * @param simplex the simplex (as an array of Pnts)
421     * @return -1, 0, or +1 for inside, on, or outside of circumcircle
422     */
423    public int vsCircumcircle (Vector2D[] simplex) {
424        Vector2D[] matrix = new Vector2D[simplex.length + 1];
425        for (int i = 0; i < simplex.length; i++)
426            matrix[i] = simplex[i].extend(1, simplex[i].dot(simplex[i]));
427        matrix[simplex.length] = this.extend(1, this.dot(this));
428        double d = determinant(matrix);
429        int result = (d < 0)? -1 : ((d > 0)? +1 : 0);
430        if (content(simplex) < 0) result = - result;
431        return result;
432    }

434    /**
435     * Circumcenter of a simplex.
436     * @param simplex the simplex (as an array of Pnts)
437     * @return the circumcenter (a Pnt) of simplex
438     */
439    public static Vector2D circumcenter (Vector2D[] simplex) {
440        int dim = simplex[0].dimension();
441        if (simplex.length - 1 != dim)
442            throw new IllegalArgumentException("Dimension mismatch");
443        Vector2D[] matrix = new Vector2D[dim];
444        for (int i = 0; i < dim; i++)
445            matrix[i] = simplex[i].bisector(simplex[i+1]);
446        Vector2D hCenter = cross(matrix);      // Center in homogeneous coordinates
447        double last = hCenter.coordinates[dim];
448        double[] result = new double[dim];
449        for (int i = 0; i < dim; i++) result[i] = hCenter.coordinates[i] / last;
450        return new Vector2D(result);
451    }
