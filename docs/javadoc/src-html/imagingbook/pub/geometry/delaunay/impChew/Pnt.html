025/**
026 * Points in Euclidean space, implemented as double[].
027 *
028 * Includes simple geometric operations.
029 * Uses matrices; a matrix is represented as an array of Pnts.
030 * Uses simplices; a simplex is represented as an array of Pnts.
031 *
032 * @author Paul Chew
033 *
034 * Created July 2005.  Derived from an earlier, messier version.
035 *
036 * Modified Novemeber 2007.  Minor clean up.
037 * 
038 * Adapted by W. Burger (2019)
039 * @version 2020-01-01
040 * 
041 */
042public class Pnt implements Point {
043
044    private double[] coordinates;          // The point's coordinates
045
046    /**
047     * Constructor.
048     * @param coords the coordinates
049     */
050    public Pnt (double... coords) {
051        // Copying is done here to ensure that Pnt's coords cannot be altered.
052        // This is necessary because the double... notation actually creates a
053        // constructor with double[] as its argument.
054//        coordinates = new double[coords.length];
055//        System.arraycopy(coords, 0, coordinates, 0, coords.length);
056        coordinates = coords.clone();
057    }
058    
059    public Pnt (Point p) {
060        this(p.getX(), p.getY());
061    }

065    @Override
066    public String toString () {
067        if (coordinates.length == 0) return "Pnt()";
068        String result = "Pnt(" + coordinates[0];
069        for (int i = 1; i < coordinates.length; i++)
070            result = result + "," + coordinates[i];
071        result = result + ")";
072        return result;
073    }
074
075    @Override
076    public boolean equals (Object other) {
077        if (!(other instanceof Pnt)) return false;
078        Pnt p = (Pnt) other;
079        if (this.coordinates.length != p.coordinates.length) return false;
080        for (int i = 0; i < this.coordinates.length; i++)
081            if (this.coordinates[i] != p.coordinates[i]) return false;
082        return true;
083    }
084
085    @Override
086    public int hashCode () {
087        int hash = 0;
088        for (double c: this.coordinates) {
089            long bits = Double.doubleToLongBits(c);
090            hash = (31*hash) ^ (int)(bits ^ (bits >> 32));
091        }
092        return hash;
093    }

097    /**
098     * Returns the specified coordinate of this point/vector.
099     * @param i the coordinate index (0, 1, ...)
100     * @return as described
101     */
102    public double coord (int i) {
103        return this.coordinates[i];
104    }

106    /**
107     * @return this Pnt's dimension.
108     */
109    public int dimension () {
110        return coordinates.length;
111    }
112
113    /**
114     * Check that dimensions match.
115     * @param p the Pnt to check (against this Pnt)
116     * @return the dimension of the Pnts
117     * @throws IllegalArgumentException if dimension fail to match
118     */
119    public int dimCheck (Pnt p) {
120        int len = this.coordinates.length;
121        if (len != p.coordinates.length)
122            throw new IllegalArgumentException("Dimension mismatch");
123        return len;
124    }
125
126    /**
127     * Create a new Pnt by adding additional coordinates to this Pnt.
128     * @param coords the new coordinates (added on the right end)
129     * @return a new Pnt with the additional coordinates
130     */
131    public Pnt extend (double... coords) {
132        double[] result = new double[coordinates.length + coords.length];
133        System.arraycopy(coordinates, 0, result, 0, coordinates.length);
134        System.arraycopy(coords, 0, result, coordinates.length, coords.length);
135        return new Pnt(result);
136    }
137
138    /**
139     * Dot product.
140     * @param p the other Pnt
141     * @return dot product of this Pnt and p
142     */
143    public double dot (Pnt p) {
144        int len = dimCheck(p);
145        double sum = 0;
146        for (int i = 0; i < len; i++)
147            sum += this.coordinates[i] * p.coordinates[i];
148        return sum;
149    }
150
151    /**
152     * Magnitude (as a vector).
153     * @return the Euclidean length of this vector
154     */
155    public double magnitude () {
156        return Math.sqrt(this.dot(this));
157    }
158
159    /**
160     * Subtract.
161     * @param p the other Pnt
162     * @return a new Pnt = this - p
163     */
164    public Pnt subtract (Pnt p) {
165        int len = dimCheck(p);
166        double[] coords = new double[len];
167        for (int i = 0; i < len; i++)
168            coords[i] = this.coordinates[i] - p.coordinates[i];
169        return new Pnt(coords);
170    }
171
172    /**
173     * Add.
174     * @param p the other Pnt
175     * @return a new Pnt = this + p
176     */
177    public Pnt add (Pnt p) {
178        int len = dimCheck(p);
179        double[] coords = new double[len];
180        for (int i = 0; i < len; i++)
181            coords[i] = this.coordinates[i] + p.coordinates[i];
182        return new Pnt(coords);
183    }
184
185    /**
186     * Angle (in radians) between two Pnts (treated as vectors).
187     * @param p the other Pnt
188     * @return the angle (in radians) between the two Pnts
189     */
190    public double angle (Pnt p) {
191        return Math.acos(this.dot(p) / (this.magnitude() * p.magnitude()));
192    }
193
194    /**
195     * Perpendicular bisector of two Pnts.
196     * Works in any dimension.  The coefficients are returned as a Pnt of one
197     * higher dimension (e.g., (A,B,C,D) for an equation of the form
198     * Ax + By + Cz + D = 0).
199     * @param point the other point
200     * @return the coefficients of the perpendicular bisector
201     */
202    public Pnt bisector (Pnt point) {
203        dimCheck(point);
204        Pnt diff = this.subtract(point);
205        Pnt sum = this.add(point);
206        double dot = diff.dot(sum);
207        return diff.extend(-dot / 2);
208    }

214     * @param matrix the matrix (an array of Pnts)

217    public static String toString (Pnt[] matrix) {
218        StringBuilder buf = new StringBuilder("{");
219        for (Pnt row: matrix) buf.append(" " + row);
220        buf.append(" }");
221        return buf.toString();

224    /**
225     * Compute the determinant of a matrix (array of Pnts).
226     * This is not an efficient implementation, but should be adequate
227     * for low dimension.
228     * @param matrix the matrix as an array of Pnts
229     * @return the determinnant of the input matrix
230     * @throws IllegalArgumentException if dimensions are wrong
231     */
232    public static double determinant (Pnt[] matrix) {
233        if (matrix.length != matrix[0].dimension())
234            throw new IllegalArgumentException("Matrix is not square");
235        boolean[] columns = new boolean[matrix.length];
236        for (int i = 0; i < matrix.length; i++) columns[i] = true;
237        try {return determinant(matrix, 0, columns);}
238        catch (ArrayIndexOutOfBoundsException e) {
239            throw new IllegalArgumentException("Matrix is wrong shape");
240        }
241    }
242
243    /**
244     * Compute the determinant of a submatrix specified by starting row
245     * and by "active" columns.
246     * @param matrix the matrix as an array of Pnts
247     * @param row the starting row
248     * @param columns a boolean array indicating the "active" columns
249     * @return the determinant of the specified submatrix
250     * @throws ArrayIndexOutOfBoundsException if dimensions are wrong
251     */
252    private static double determinant(Pnt[] matrix, int row, boolean[] columns){
253        if (row == matrix.length) return 1;
254        double sum = 0;
255        int sign = 1;
256        for (int col = 0; col < columns.length; col++) {
257            if (!columns[col]) continue;
258            columns[col] = false;
259            sum += sign * matrix[row].coordinates[col] *
260                   determinant(matrix, row+1, columns);
261            columns[col] = true;
262            sign = -sign;
263        }
264        return sum;
265    }
266
267    /**
268     * Compute generalized cross-product of the rows of a matrix.
269     * The result is a Pnt perpendicular (as a vector) to each row of
270     * the matrix.  This is not an efficient implementation, but should
271     * be adequate for low dimension.
272     * @param matrix the matrix of Pnts (one less row than the Pnt dimension)
273     * @return a Pnt perpendicular to each row Pnt
274     * @throws IllegalArgumentException if matrix is wrong shape
275     */
276    public static Pnt cross (Pnt[] matrix) {
277        int len = matrix.length + 1;
278        if (len != matrix[0].dimension())
279            throw new IllegalArgumentException("Dimension mismatch");
280        boolean[] columns = new boolean[len];
281        for (int i = 0; i < len; i++) columns[i] = true;
282        double[] result = new double[len];
283        int sign = 1;
284        try {
285            for (int i = 0; i < len; i++) {
286                columns[i] = false;
287                result[i] = sign * determinant(matrix, 0, columns);
288                columns[i] = true;
289                sign = -sign;
290            }
291        } catch (ArrayIndexOutOfBoundsException e) {
292            throw new IllegalArgumentException("Matrix is wrong shape");
293        }
294        return new Pnt(result);
295    }

299    /**
300     * Determine the signed content (i.e., area or volume, etc.) of a simplex.
301     * @param simplex the simplex (as an array of Pnts)
302     * @return the signed content of the simplex
303     */
304    public static double content (Pnt[] simplex) {
305        Pnt[] matrix = new Pnt[simplex.length];
306        for (int i = 0; i < matrix.length; i++)
307            matrix[i] = simplex[i].extend(1);
308        int fact = 1;
309        for (int i = 1; i < matrix.length; i++) fact = fact*i;
310        return determinant(matrix) / fact;
311    }

313    /**
314     * Relation between this Pnt and a simplex (represented as an array of
315     * Pnts). Result is an array of signs, one for each vertex of the simplex,
316     * indicating the relation between the vertex, the vertex's opposite facet,
317     * and this Pnt.
318     * Throws a {@link IllegalArgumentException} if the simplex is degenerate.
319     *
320     * <pre>
321     *   -1 means Pnt is on same side of facet
322     *    0 means Pnt is on the facet
323     *   +1 means Pnt is on opposite side of facet
324     * </pre>
325     *
326     * @param simplex an array of Pnts representing a simplex
327     * @return an array of signs showing relation between this Pnt and simplex
328     */
329    public int[] relation (Pnt[] simplex) {
330        /* In 2D, we compute the cross of this matrix:
331         *    1   1   1   1
332         *    p0  a0  b0  c0
333         *    p1  a1  b1  c1
334         * where (a, b, c) is the simplex and p is this Pnt. The result is a
335         * vector in which the first coordinate is the signed area (all signed
336         * areas are off by the same constant factor) of the simplex and the
337         * remaining coordinates are the *negated* signed areas for the
338         * simplices in which p is substituted for each of the vertices.
339         * Analogous results occur in higher dimensions.
340         */
341        int dim = simplex.length - 1;
342        if (this.dimension() != dim)
343            throw new IllegalArgumentException("Dimension mismatch");
344
345        /* Create and load the matrix */
346        Pnt[] matrix = new Pnt[dim+1];
347        /* First row */
348        double[] coords = new double[dim+2];
349        for (int j = 0; j < coords.length; j++) coords[j] = 1;
350        matrix[0] = new Pnt(coords);
351        /* Other rows */
352        for (int i = 0; i < dim; i++) {
353            coords[0] = this.coordinates[i];
354            for (int j = 0; j < simplex.length; j++)
355                coords[j+1] = simplex[j].coordinates[i];
356            matrix[i+1] = new Pnt(coords);
357        }
358
359        /* Compute and analyze the vector of areas/volumes/contents */
360        Pnt vector = cross(matrix);
361        double content = vector.coordinates[0];
362        int[] result = new int[dim+1];
363        for (int i = 0; i < result.length; i++) {
364            double value = vector.coordinates[i+1];
365            if (Math.abs(value) <= 1.0e-6 * Math.abs(content)) result[i] = 0;
366            else if (value < 0) result[i] = -1;
367            else result[i] = 1;
368        }
369        if (content < 0) {
370            for (int i = 0; i < result.length; i++)
371                result[i] = -result[i];
372        }
373        if (content == 0) {
374            for (int i = 0; i < result.length; i++)
375                result[i] = Math.abs(result[i]);
376        }
377        return result;
378    }

380    /**
381     * Test if this Pnt is outside of simplex.
382     * @param simplex the simplex (an array of Pnts)
383     * @return simplex Pnt that "witnesses" outsideness (or null if not outside)
384     */
385    public Pnt isOutside (Pnt[] simplex) {
386        int[] result = this.relation(simplex);
387        for (int i = 0; i < result.length; i++) {
388            if (result[i] > 0) return simplex[i];
389        }
390        return null;
391    }
392
393    /**
394     * Test if this Pnt is on a simplex.
395     * @param simplex the simplex (an array of Pnts)
396     * @return the simplex Pnt that "witnesses" on-ness (or null if not on)
397     */
398    public Pnt isOn (Pnt[] simplex) {
399        int[] result = this.relation(simplex);
400        Pnt witness = null;
401        for (int i = 0; i < result.length; i++) {
402            if (result[i] == 0) witness = simplex[i];
403            else if (result[i] > 0) return null;
404        }
405        return witness;
406    }

408    /**
409     * Test if this Pnt is inside a simplex.
410     * @param simplex the simplex (an arary of Pnts)
411     * @return true iff this Pnt is inside simplex.
412     */
413    public boolean isInside (Pnt[] simplex) {
414        int[] result = this.relation(simplex);
415        for (int r: result) if (r >= 0) return false;
416        return true;
417    }

419    /**
420     * Test relation between this Pnt and circumcircle of a simplex.
421     * @param simplex the simplex (as an array of Pnts)
422     * @return -1, 0, or +1 for inside, on, or outside of circumcircle
423     */
424    public int vsCircumcircle (Pnt[] simplex) {
425        Pnt[] matrix = new Pnt[simplex.length + 1];
426        for (int i = 0; i < simplex.length; i++)
427            matrix[i] = simplex[i].extend(1, simplex[i].dot(simplex[i]));
428        matrix[simplex.length] = this.extend(1, this.dot(this));
429        double d = determinant(matrix);
430        int result = (d < 0)? -1 : ((d > 0)? +1 : 0);
431        if (content(simplex) < 0) result = - result;
432        return result;
433    }

435    /**
436     * Circumcenter of a simplex.
437     * @param simplex the simplex (as an array of Pnts)
438     * @return the circumcenter (a Pnt) of simplex
439     */
440    public static Pnt circumcenter (Pnt[] simplex) {
441        int dim = simplex[0].dimension();
442        if (simplex.length - 1 != dim)
443            throw new IllegalArgumentException("Dimension mismatch");
444        Pnt[] matrix = new Pnt[dim];
445        for (int i = 0; i < dim; i++)
446            matrix[i] = simplex[i].bisector(simplex[i+1]);
447        Pnt hCenter = cross(matrix);      // Center in homogeneous coordinates
448        double last = hCenter.coordinates[dim];
449        double[] result = new double[dim];
450        for (int i = 0; i < dim; i++) result[i] = hCenter.coordinates[i] / last;
451        return new Pnt(result);
452    }
