023import java.util.Collections;
024import java.util.HashMap;
025import java.util.Map;
026import java.util.Set;
027
028/**
029 * Straightforward undirected graph implementation.
030 * Nodes are generic type N.
031 *
032 * @author Paul Chew
033 *
034 * Created November, December 2007.  For use in Delaunay/Voronoi code.
035 *
036 */
037public class Graph<N> {
038
039    private Map<N, Set<N>> theNeighbors =    // Node -> adjacent nodes
040        new HashMap<N, Set<N>>();
041    private Set<N> theNodeSet =              // Set view of all nodes
042        Collections.unmodifiableSet(theNeighbors.keySet());
043
044    /**
045     * Add a node.  If node is already in graph then no change.
046     * @param node the node to add
047     */
048    public void add (N node) {
049        if (theNeighbors.containsKey(node)) return;
050        theNeighbors.put(node, new ArraySet<N>());
051     }
052
053    /**
054     * Add a link. If the link is already in graph then no change.
055     * @param nodeA one end of the link
056     * @param nodeB the other end of the link
057     * @throws NullPointerException if either endpoint is not in graph
058     */
059    public void add (N nodeA, N nodeB) throws NullPointerException {
060        theNeighbors.get(nodeA).add(nodeB);
061        theNeighbors.get(nodeB).add(nodeA);
062    }
063
064    /**
065     * Remove node and any links that use node. If node not in graph, nothing
066     * happens.
067     * @param node the node to remove.
068     */
069    public void remove (N node) {
070        if (!theNeighbors.containsKey(node)) return;
071        for (N neighbor: theNeighbors.get(node))
072            theNeighbors.get(neighbor).remove(node);    // Remove "to" links
073        theNeighbors.get(node).clear();                 // Remove "from" links
074        theNeighbors.remove(node);                      // Remove the node
075    }
076
077    /**
078     * Remove the specified link. If link not in graph, nothing happens.
079     * @param nodeA one end of the link
080     * @param nodeB the other end of the link
081     * @throws NullPointerException if either endpoint is not in graph
082     */
083    public void remove (N nodeA, N nodeB) throws NullPointerException {
084        theNeighbors.get(nodeA).remove(nodeB);
085        theNeighbors.get(nodeB).remove(nodeA);
086    }
087
088    /**
089     * Report all the neighbors of node.
090     * @param node the node
091     * @return the neighbors of node
092     * @throws NullPointerException if node does not appear in graph
093     */
094    public Set<N> neighbors (N node) throws NullPointerException {
095        return Collections.unmodifiableSet(theNeighbors.get(node));
096    }
097
098    /**
099     * Returns an unmodifiable Set view of the nodes contained in this graph.
100     * The set is backed by the graph, so changes to the graph are reflected in
101     * the set.
102     * @return a Set view of the graph's node set
103     */
104    public Set<N> nodeSet () {
105        return theNodeSet;
106    }
107
108}
