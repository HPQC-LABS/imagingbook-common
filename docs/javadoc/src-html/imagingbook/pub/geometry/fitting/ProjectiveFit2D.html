001package imagingbook.pub.geometry.fitting;
002
003import org.apache.commons.math3.linear.DecompositionSolver;
004import org.apache.commons.math3.linear.MatrixUtils;
005import org.apache.commons.math3.linear.RealMatrix;
006import org.apache.commons.math3.linear.RealVector;
007import org.apache.commons.math3.linear.SingularValueDecomposition;
008
009import imagingbook.pub.geometry.basic.Point;
010import imagingbook.pub.geometry.mappings.linear.LinearMapping2D;
011
012public class ProjectiveFit2D implements LinearFit2D {
013        
014        private final RealMatrix A;             // the calculated transformation matrix
015        private final double err;               // the calculated error
016        
017        /**
018         * Constructor.
019         * Fits two sequences of 2D points using a projective transformation model.
020         * At least 4 point pairs are required. For 4 point pairs, the solution
021         * is an exact fit, otherwise a least-squares fit is found.
022         * @param P the source points
023         * @param Q the target points
024         */
025        public ProjectiveFit2D(Point[] P, Point[] Q) {
026                checkSize(P, Q);
027                final int n = P.length;
028                
029                double[] ba = new double[2 * n];
030                double[][] Ma = new double[2 * n][];
031                for (int i = 0; i < n; i++) {
032                        double px = P[i].getX();
033                        double py = P[i].getY();
034                        double qx = Q[i].getX();
035                        double qy = Q[i].getY();
036                        ba[2 * i + 0] = qx;
037                        ba[2 * i + 1] = qy;
038                        Ma[2 * i + 0] = new double[] { px, py, 1, 0, 0, 0, -qx * px, -qx * py };
039                        Ma[2 * i + 1] = new double[] { 0, 0, 0, px, py, 1, -qy * px, -qy * py };
040                }
041                
042                RealMatrix M = MatrixUtils.createRealMatrix(Ma);
043                RealVector b = MatrixUtils.createRealVector(ba);
044                DecompositionSolver solver = new SingularValueDecomposition(M).getSolver();
045                RealVector h = solver.solve(b);
046                A = MatrixUtils.createRealMatrix(3, 3);
047                A.setEntry(0, 0, h.getEntry(0));
048                A.setEntry(0, 1, h.getEntry(1));
049                A.setEntry(0, 2, h.getEntry(2));
050                A.setEntry(1, 0, h.getEntry(3));
051                A.setEntry(1, 1, h.getEntry(4));
052                A.setEntry(1, 2, h.getEntry(5));
053                A.setEntry(2, 0, h.getEntry(6));
054                A.setEntry(2, 1, h.getEntry(7));
055                A.setEntry(2, 2, 1.0);
056                
057                err = this.calculateError(P, Q, A);
058        }
059
060        @Override
061        public double[][] getTransformationMatrix() {
062                return A.getData();
063        }
064
065        @Override
066        public double getError() {
067                return err;
068        }
069        
070        // ------------------------------------------------------------------------------
071        
072        protected static void checkSize(Point[] P, Point[] Q) {
073                if (P.length < 4 || Q.length < 4) {
074                        throw new IllegalArgumentException("At least 4 point pairs are required to calculate this fit");
075                }
076        }
077        
078        /**
079         * Calculates and returns the cumulative distance error
080         * between the two point sequences under the transformation A,
081         * i.e., {@code e = sum_i (||p_i * A - q_i||)}.
082         * @param P     the first point sequence
083         * @param Q the second point sequence
084         * @param A     a (3 x 3) projective transformation matrix
085         * @return the error {@code e}
086         */
087        private double calculateError(Point[] P, Point[] Q, RealMatrix A) {
088                int m = Math.min(P.length,  Q.length);
089                LinearMapping2D map = new LinearMapping2D(A.getData());
090                double errSum = 0;
091                for (int i = 0; i < m; i++) {
092                        Point p = P[i];
093                        Point q = Q[i];
094                        Point pp = map.applyTo(p);
095                        errSum = errSum + Point.distance(q, pp);
096                }
097                return errSum;
098        }
099
100}
