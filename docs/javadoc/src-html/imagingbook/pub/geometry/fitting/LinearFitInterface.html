001package imagingbook.pub.geometry.fitting;
002
003import java.awt.geom.Point2D;
004import java.util.List;
005
006import org.apache.commons.math3.linear.RealMatrix;
007
008/**
009 * Describes a fitter based on a linear transformation model.
010 * @author WB
011 * @deprecated
012 */
013public interface LinearFitInterface {
014        
015        /**
016         * Calculates the least-squares fit for the supplied point sequences,
017         * which must have the same length and order (i.e., points must be in 
018         * correspondence).
019         * @param X Sequence of n-dimensional points
020         * @param Y Sequence of n-dimensional points (reference)
021         */
022        public void fit(List<double[]> X, List<double[]> Y);
023        
024        /**
025         * Calculates the least-squares fit for the supplied point sequences,
026         * which must have the same length and order (i.e., points must be in 
027         * correspondence). Convenience method for fitting 2D points.
028         * @param X Sequence of 2-dimensional points
029         * @param Y Sequence of 2-dimensional points (reference)
030         */
031        public void fitPoints(List<Point2D> X, List<Point2D> Y);
032        
033        /**
034         * Retrieves the (n) x (n+1) transformation matrix A, such that
035         * y_i = A * x_i (with x_i in homogeneous coordinates)
036         */
037        public RealMatrix getTransformationMatrix();
038        
039        /**
040         * Retrieves the total (squared) error for the estimated fit.
041         * @return The total error for the estimated fit.
042         */
043        public double getError();
044
045}
