001package imagingbook.pub.geometry.fitting;
002
003import org.apache.commons.math3.linear.RealMatrix;
004
005import imagingbook.pub.geometry.basic.Point;
006import imagingbook.pub.geometry.mappings.linear.LinearMapping2D;
007
008/**
009 * Describes a fitter based on a linear transformation model.
010 * @author WB
011 */
012public interface LinearFit2D {
013        
014        /**
015         * Returns the (3,3) or (2,3) transformation matrix A for this fit, such that
016         * {@code y_i ~ A * x_i} (with {@code x_i} in homogeneous coordinates).
017         * 
018         * @return the transformation matrix for this fit
019         */
020        RealMatrix getTransformationMatrix();
021        
022        /**
023         * Retrieves the total (squared) error for the estimated fit.
024         * @return the fitting error
025         */
026        double getError();
027        
028        
029        default public double calculateError(Point[] P, Point[] Q, RealMatrix A) {
030                final int m = Math.min(P.length,  Q.length);
031                LinearMapping2D map = new LinearMapping2D(A.getData());
032                double errSum = 0;
033                for (int i = 0; i < m; i++) {
034                        Point p = P[i];
035                        Point q = Q[i];
036                        Point pp = map.applyTo(p);
037                        double e = Point.distance(q, pp);
038                        errSum = errSum + e * e;
039                }
040                return Math.sqrt(errSum);
041        }
042        
043}
