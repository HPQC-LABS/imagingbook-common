001package imagingbook.pub.geometry.fitting;
002
003import java.util.List;
004
005import org.apache.commons.math3.linear.ArrayRealVector;
006import org.apache.commons.math3.linear.DecompositionSolver;
007import org.apache.commons.math3.linear.MatrixUtils;
008import org.apache.commons.math3.linear.RealMatrix;
009import org.apache.commons.math3.linear.RealVector;
010import org.apache.commons.math3.linear.SingularValueDecomposition;
011
012import ij.IJ;
013
014public class AffineFit extends LinearFit {
015        
016        private int m;                  // number of samples
017        
018        private RealMatrix A = null;
019        
020        public AffineFit() {
021                this(2);        // 2D by default
022        }
023        
024        public AffineFit(int n) {
025                super(n);
026        }
027
028        @Override
029        public void fit(List<double[]> X, List<double[]> Y) {   // fits n-dimensional data sets with affine model
030                if (X.size() != Y.size())
031                        throw new IllegalArgumentException("point sequences X, Y must have same length");
032                if (X.get(0).length < n)
033                        throw new IllegalArgumentException("dimensionality of samples must be >= " + n);
034                this.m = X.size();
035                
036                RealMatrix M = MatrixUtils.createRealMatrix(2 * m, 2 * (n + 1));
037                RealVector b = new ArrayRealVector(2 * m);
038                
039                IJ.log("M = " + M.getRowDimension() + " / " + M.getColumnDimension());
040                
041                // mount matrix M:      TODO: needs to be checked for n > 2!
042                int row = 0;
043                for (double[] x : X) {
044                        for (int j = 0; j < n; j++) {
045                                M.setEntry(row, j, x[j]);
046                                M.setEntry(row, n, 1);
047                        }
048                        row++;
049                        for (int j = 0; j < n; j++) {
050                                M.setEntry(row, j + n + 1, x[j]);
051                                M.setEntry(row, 2 * n + 1, 1);
052                        }
053                        row++;
054                }
055                //IJ.log("M = \n" + Matrix.toString(M.getData()));
056                
057                // mount vector b
058                row = 0;
059                for (double[] y : Y) {
060                        for (int j = 0; j < n; j++) {
061                                b.setEntry(row, y[j]);
062                                row++;
063                        }
064                }
065                
066                //IJ.log("b = \n" + Matrix.toString(b.toArray()));
067                
068                SingularValueDecomposition svd = new SingularValueDecomposition(M);
069                DecompositionSolver solver = svd.getSolver();
070                RealVector a = solver.solve(b);
071                A = makeTransformationMatrix(a);
072        }
073
074        // creates a n x (n+1) transformation matrix from the elements of a
075        private RealMatrix makeTransformationMatrix(RealVector a) {
076                RealMatrix A = MatrixUtils.createRealMatrix(n, n + 1);
077                int i = 0;
078                for (int row = 0; row < n; row++) {
079                        // get (n+1) elements from a and place in row
080                        for (int j = 0; j <= n; j++) {
081                                A.setEntry(row, j, a.getEntry(i));
082                                i++;
083                        }
084                }
085                //A.setEntry(n - 1, n, 1);
086                return A;
087        }
088
089        @Override
090        public RealMatrix getTransformationMatrix() {
091                return A;
092        }
093
094        @Override
095        public double getError() {
096                // TODO Auto-generated method stub
097                return 0;
098        }
099
100}
