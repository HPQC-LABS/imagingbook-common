001package imagingbook.pub.geometry.fitting;
002
003import static imagingbook.lib.math.Arithmetic.sqr;
004
005import org.apache.commons.math3.linear.ArrayRealVector;
006import org.apache.commons.math3.linear.DecompositionSolver;
007import org.apache.commons.math3.linear.MatrixUtils;
008import org.apache.commons.math3.linear.RealMatrix;
009import org.apache.commons.math3.linear.RealVector;
010import org.apache.commons.math3.linear.SingularValueDecomposition;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.pub.geometry.basic.Point;
014
015/**
016 * TODO: see ProjectiveMapping2D.fromNPoints(Point[] P, Point[] Q) -- remove 1 version (either fit or mapping)!
017 *
018 */
019public class AffineFit2D implements LinearFit2D {
020        
021        private final RealMatrix A;             // the calculated transformation matrix
022        private final double err;               // the calculated error
023        
024        /**
025         * Constructor.
026         * Fits two sequences of 2D points using an affine transformation model.
027         * At least 3 point pairs are required. For 3 point pairs, the solution
028         * is an exact fit, otherwise a least-squares fit is found.
029         * @param P the source points
030         * @param Q the target points
031         */
032        public AffineFit2D(Point[] P, Point[] Q) {      // 
033                checkSize(P, Q);
034                int m = P.length;
035                
036                RealMatrix M = MatrixUtils.createRealMatrix(2 * m, 6);
037                RealVector b = new ArrayRealVector(2 * m);
038                
039                // mount the matrix M
040                int row = 0;
041                for (Point p : P) {
042                        M.setEntry(row, 0, p.getX());
043                        M.setEntry(row, 1, p.getY());
044                        M.setEntry(row, 2, 1);
045                        row++;
046                        M.setEntry(row, 3, p.getX());
047                        M.setEntry(row, 4, p.getY());
048                        M.setEntry(row, 5, 1);
049                        row++;
050                }
051                
052                // mount vector b
053                row = 0;
054                for (Point q : Q) {
055                        b.setEntry(row, q.getX());
056                        row++;
057                        b.setEntry(row, q.getY());
058                        row++;
059                }
060                
061                // solve M * a = b (for the unknown parameter vector a):
062                DecompositionSolver solver = new SingularValueDecomposition(M).getSolver();
063                RealVector a = solver.solve(b);
064                A = makeTransformationMatrix(a);
065                err = getError(P, Q, A);
066        }
067
068        // creates a n x (n+1) transformation matrix from the elements of a
069        private RealMatrix makeTransformationMatrix(RealVector a) {
070                RealMatrix A = MatrixUtils.createRealMatrix(2, 3);
071                int i = 0;
072                for (int row = 0; row < 2; row++) {
073                        // get (n+1) elements from a and place in row
074                        for (int j = 0; j < 3; j++) {
075                                A.setEntry(row, j, a.getEntry(i));
076                                i++;
077                        }
078                }
079                return A;
080        }
081        
082        // --------------------------------------------------------
083        
084        protected static void checkSize(Point[] P, Point[] Q) {
085                if (P.length < 3 || Q.length < 3) {
086                        throw new IllegalArgumentException("At least 3 point pairs are required to calculate this fit");
087                }
088        }
089        
090        /**
091         * Calculates and returns the cumulative distance error
092         * between the two point sequences under the transformation A,
093         * i.e., {@code e = sum_i (||p_i * A - q_i||)}.
094         * @param P     the first point sequence
095         * @param Q the second point sequence
096         * @param A     a (2 x 3) affine transformation matrix
097         * @return the error {@code e}
098         */
099        protected static double getError(Point[] P, Point[] Q, RealMatrix A) {
100                int m = Math.min(P.length,  Q.length);
101                double errSum = 0;
102                for (int i = 0; i < m; i++) {
103                        Point p = P[i];
104                        Point q = Q[i];
105                        Point p2 = Point.create(A.operate(Matrix.toHomogeneous(p.toArray())));
106                        errSum = errSum + 
107                                        Math.sqrt(sqr(q.getX() - p2.getX()) + sqr(q.getY() - p2.getY()));
108                }
109                return errSum;
110        }
111        
112        // --------------------------------------------------------
113
114        @Override
115        public RealMatrix getTransformationMatrix() { // TODO: should return a mapping!?
116                return A;
117        }
118
119        @Override
120        public double getError() {
121                return err;
122        }
123
124}
