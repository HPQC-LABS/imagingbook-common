001package imagingbook.pub.geometry.fitting;
002
003import org.apache.commons.math3.linear.ArrayRealVector;
004import org.apache.commons.math3.linear.DecompositionSolver;
005import org.apache.commons.math3.linear.MatrixUtils;
006import org.apache.commons.math3.linear.RealMatrix;
007import org.apache.commons.math3.linear.RealVector;
008import org.apache.commons.math3.linear.SingularValueDecomposition;
009
010import imagingbook.pub.geometry.basic.Point;
011import imagingbook.pub.geometry.mappings.linear.AffineMapping2D;
012
013public class AffineFit2D implements LinearFit2D {
014        
015        private final RealMatrix A;             // the calculated transformation matrix
016        private final double err;               // the calculated error
017        
018        /**
019         * Constructor.
020         * Fits two sequences of 2D points using an affine transformation model.
021         * At least 3 point pairs are required. For 3 point pairs, the solution
022         * is an exact fit, otherwise a least-squares fit is found.
023         * @param P the source points
024         * @param Q the target points
025         */
026        public AffineFit2D(Point[] P, Point[] Q) {      // 
027                checkSize(P, Q);
028                int m = P.length;
029                
030                RealMatrix M = MatrixUtils.createRealMatrix(2 * m, 6);
031                RealVector b = new ArrayRealVector(2 * m);
032                
033                // mount the matrix M
034                int row = 0;
035                for (Point p : P) {
036                        M.setEntry(row, 0, p.getX());
037                        M.setEntry(row, 1, p.getY());
038                        M.setEntry(row, 2, 1);
039                        row++;
040                        M.setEntry(row, 3, p.getX());
041                        M.setEntry(row, 4, p.getY());
042                        M.setEntry(row, 5, 1);
043                        row++;
044                }
045                
046                // mount vector b
047                row = 0;
048                for (Point q : Q) {
049                        b.setEntry(row, q.getX());
050                        row++;
051                        b.setEntry(row, q.getY());
052                        row++;
053                }
054                
055                // solve M * a = b (for the unknown parameter vector a):
056                DecompositionSolver solver = new SingularValueDecomposition(M).getSolver();
057                RealVector a = solver.solve(b);
058                A = makeTransformationMatrix(a);
059                err = calculateError(P, Q, A);
060        }
061
062        // creates a (2 x 3) transformation matrix from the elements of a
063        private RealMatrix makeTransformationMatrix(RealVector a) {
064                RealMatrix A = MatrixUtils.createRealMatrix(2, 3);
065                int i = 0;
066                for (int row = 0; row < 2; row++) {
067                        // get (n+1) elements from a and place in row
068                        for (int j = 0; j < 3; j++) {
069                                A.setEntry(row, j, a.getEntry(i));
070                                i++;
071                        }
072                }
073                return A;
074        }
075        
076        // --------------------------------------------------------
077        
078        private void checkSize(Point[] P, Point[] Q) {
079                if (P.length < 3 || Q.length < 3) {
080                        throw new IllegalArgumentException("At least 3 point pairs are required to calculate this fit");
081                }
082        }
083        
084        /**
085         * Calculates and returns the cumulative distance error
086         * between the two point sequences under the transformation A,
087         * i.e., {@code e = sum_i (||p_i * A - q_i||)}.
088         * @param P     the first point sequence
089         * @param Q the second point sequence
090         * @param A     a 2 x 3 (affine) transformation matrix
091         * @return the error {@code e}
092         */
093        protected static double calculateError(Point[] P, Point[] Q, RealMatrix A) {
094                int m = Math.min(P.length,  Q.length);
095                AffineMapping2D map = new AffineMapping2D(A.getData());
096                double errSum = 0;
097                for (int i = 0; i < m; i++) {
098                        Point p = P[i];
099                        Point q = Q[i];
100                        Point pp = map.applyTo(p);
101                        errSum = errSum + Point.distance(q, pp);
102                }
103                return errSum;
104        }
105        
106        // --------------------------------------------------------
107
108        @Override
109        public double[][] getTransformationMatrix() {
110                return A.getData();
111        }
112
113        @Override
114        public double getError() {
115                return err;
116        }
117
118}
