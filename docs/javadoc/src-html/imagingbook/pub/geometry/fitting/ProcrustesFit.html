017/**
018 * Implements a 2-dimensional Procrustes fit, using the algorithm described in 
019 * Shinji Umeyama, "Least-squares estimation of transformation parameters 
020 * between two point patterns", IEEE Transactions on Pattern Analysis and 
021 * Machine Intelligence 13.4 (Apr. 1991), pp. 376â€“380.
022 * Usage example (also see the {@code main()} method of this class):
023 * <pre>
024 * Point[] P = ... // create sequence of 2D source points
025 * Point[] Q = ... // create sequence of 2D target points
026 * ProcrustesFit pf = new ProcrustesFit(P, Q);
027 * double err = pf.getError();
028 * RealMatrix R = pf.getR();
029 * RealVector t = pf.getT();
030 * double s = pf.getScale();
031 * double err = pf.getError();
032 * RealMatrix A = pf.getTransformationMatrix();
033 * </pre>
034 * 
035 * @author W. Burger
036 * @version 2020-01-05
037 */
038public class ProcrustesFit implements LinearFit2D {
039        
040        private RealMatrix R = null;                                    // orthogonal (rotation) matrix
041        private RealVector t = null;                                    // translation vector
042        private double s = 1;                                                   // scale
043        private double err = Double.POSITIVE_INFINITY;  // total (squared) error
044        
045        // --------------------------------------------------------------
046        
047        /**
048         * Convenience constructor, with
049         * parameters {@code allowTranslation}, {@code allowScaling} and {@code forceRotation}
050         * set to {@code true}.
051         * @param P the source points
052         * @param Q the target points
053         */
054        public ProcrustesFit(Point[] P, Point[] Q) {
055                this(P, Q, true, true, true);
056        }
057        
058        /**
059         * Full constructor.
060         * @param P the first point sequence
061         * @param Q the second point sequence
062         * @param allowTranslation if {@code true}, translation (t) between point sets is considered, 
063         *              otherwise zero translation is assumed
064         * @param allowScaling if {@code true}, scaling (s) between point sets is considered, 
065         *              otherwise unit scale assumed
066         * @param forceRotation if {@code true}, the orthogonal part of the transformation (Q)
067         *              is forced to a true rotation and no reflection is allowed
068         */
069        public ProcrustesFit(Point[] P, Point[] Q, boolean allowTranslation, boolean allowScaling, boolean forceRotation) {
070                AffineFit2D.checkSize(P, Q);
071                
072                double[] meanP = null;
073                double[] meanY = null;
074                
075                if (allowTranslation) {
076                        meanP = getMeanVec(P);
077                        meanY = getMeanVec(Q);
078                }
079                
080                RealMatrix vP = makeDataMatrix(P, meanP);
081                RealMatrix vQ = makeDataMatrix(Q, meanY);
082                MatrixUtils.checkAdditionCompatible(vP, vQ);    // P, Q of same dimensions?
083                
084                RealMatrix QPt = vQ.multiply(vP.transpose());
085                SingularValueDecomposition svd = new SingularValueDecomposition(QPt);
086                
087                RealMatrix U = svd.getU();
088                RealMatrix S = svd.getS();
089                RealMatrix V = svd.getV();
090                        
091                double d = (svd.getRank() >= 2) ? det(QPt) : det(U) * det(V);
092                
093                RealMatrix D = MatrixUtils.createRealIdentityMatrix(2);
094                if (d < 0 && forceRotation)
095                        D.setEntry(1, 1, -1);
096                
097                R = U.multiply(D).multiply(V.transpose());
098                
099                double normP = vP.getFrobeniusNorm();
100                double normQ = vQ.getFrobeniusNorm();
101                
102                s = (allowScaling) ? 
103                                S.multiply(D).getTrace() / sqr(normP) : 1.0;
104                
105                if (allowTranslation) {
106                        RealVector ma = MatrixUtils.createRealVector(meanP);
107                        RealVector mb = MatrixUtils.createRealVector(meanY);
108                        t = mb.subtract(R.scalarMultiply(s).operate(ma));
109                }
110                else {
111                        t = new ArrayRealVector(2);     // zero vector
112                }
113                
114                // TODO: sqrt() seems to be OK! result is same order of magnitude but ca. 50 % of euclidean error
115                err = Math.sqrt(sqr(normQ) - sqr(S.multiply(D).getTrace() / normP));    
116        }
117        
118        // -----------------------------------------------------------------
119        
120        /**
121         * Retrieves the estimated scale.
122         * @return The estimated scale (or 1 if {@code allowscaling = false}).
123         */
124        public double getScale() {
125                return s;
126        }
127        
128        /**
129         * Retrieves the estimated orthogonal (rotation) matrix.
130         * @return The estimated rotation matrix.
131         */
132        public RealMatrix getR() {
133                return R;
134        }
135        
136        /**
137         * Retrieves the estimated translation vector.
138         * @return The estimated translation vector.
139         */
140        public RealVector getT() {
141                return t;
142        }

144        // --------------------------------------------------------
145        
146        @Override
147        public RealMatrix getTransformationMatrix() {
148                RealMatrix cR = R.scalarMultiply(s);
149                RealMatrix M = MatrixUtils.createRealMatrix(2, 3);
150                M.setSubMatrix(cR.getData(), 0, 0);
151                M.setColumnVector(2, t);
152                return M;
153        }
154        
155        @Override
156        public double getError() {
157                return err;
158        }
159        
160        /**
161         * Calculates the total error for the estimated fit as
162         * the sum of the squared Euclidean distances between the 
163         * transformed point set X and the reference set Y.
164         * This method is provided for testing as an alternative to
165         * the quicker {@link getError} method.
166         * @param P Sequence of n-dimensional points.
167         * @param Q Sequence of n-dimensional points (reference).
168         * @return The total error for the estimated fit.
169         */
170        private double getEuclideanError(Point[] P, Point[] Q) {
171                int m = Math.min(P.length,  Q.length);
172                RealMatrix cR = R.scalarMultiply(s);
173                double errSum = 0;
174                for (int i = 0; i < m; i++) {
175                        RealVector ai = new ArrayRealVector(P[i].toArray());
176                        RealVector bi = new ArrayRealVector(Q[i].toArray());
177                        RealVector aiT = cR.operate(ai).add(t);
178                        double ei = aiT.subtract(bi).getNorm();
179                        errSum = errSum + ei;
180                }
181                return errSum;
182        }
183        
184        private double getEuclideanError2(Point[] P, Point[] Q) {       
185                return AffineFit2D.getError(P, Q, getTransformationMatrix());
186        }

205        private double det(RealMatrix M) {
206                return new LUDecomposition(M).getDeterminant();
207        }
208        
209        private double[] getMeanVec(Point[] points) {
210                //double[] sum = new double[X.get(0).length];
211                double sumX = 0;
212                double sumY = 0;
213                for (Point p : points) {
214                        sumX = sumX + p.getX();
215                        sumY = sumY + p.getY();
216                }
217                return new double[] {sumX / points.length, sumY / points.length};
218        }
219        
220        private RealMatrix makeDataMatrix(Point[] points, double[] meanX) {
221                RealMatrix M = MatrixUtils.createRealMatrix(2, points.length);
222                RealVector mean = MatrixUtils.createRealVector(meanX);
223                int i = 0;
224                for (Point p : points) {
225                        RealVector cv = MatrixUtils.createRealVector(p.toArray());
226                        if (meanX != null) {
227                                cv = cv.subtract(mean);
228                        }
229                        M.setColumnVector(i, cv);
230                        i++;
231                }
232                return M;
233        }

235        private void printSVD(SingularValueDecomposition svd) {
236                RealMatrix U = svd.getU();
237                RealMatrix S = svd.getS();
238                RealMatrix V = svd.getV();
239                System.out.println("------ SVD ---------------");
240                System.out.println("U = " + Matrix.toString(U.getData()));
241                System.out.println("S = " + Matrix.toString(S.getData()));
242                System.out.println("V = " + Matrix.toString(V.getData()));
243                System.out.println("--------------------------");
244        }

246        private static double roundToDigits(double x, int ndigits) {
247                int d = (int) Math.pow(10, ndigits);
248                return Math.rint(x * d) / d;
249        }

253        public static void main(String[] args) {
254                PrintPrecision.set(6);
255                int NDIGITS = 1;
256                
257                boolean allowTranslation = true;
258                boolean allowScaling = true;
259                boolean forceRotation = true;
260                
261                double a = 0.6;
262                double[][] R0data =
263                        {{ Math.cos(a), -Math.sin(a) },
264                         { Math.sin(a),  Math.cos(a) }};
265                
266                RealMatrix R0 = MatrixUtils.createRealMatrix(R0data);
267                double[] t0 = {4, -3};
268                double s = 3.5;
269                
270                System.out.format("original alpha: a = %.6f\n", a);
271                System.out.println("original rotation: R = \n" + Matrix.toString(R0.getData()));
272                System.out.println("original translation: t = " + Matrix.toString(t0));
273                System.out.format("original scale: s = %.6f\n", s);
274                System.out.println();
275                
276                Point[] P = {
277                                Point.create(2, 5),
278                                Point.create(7, 3),
279                                Point.create(0, 9),
280                                Point.create(5, 4)
281                };
282                
283                Point[] Q = new Point[P.length];
284                
285                for (int i = 0; i < P.length; i++) {
286                        Point q = Point.create(R0.operate(P[i].toArray()));
287                        // noise!
288                        double qx = roundToDigits(s * q.getX() + t0[0], NDIGITS);
289                        double qy = roundToDigits(s * q.getY() + t0[1], NDIGITS);
290                        Q[i] = Point.create(qx, qy);
291                }
292                
293                //P[0] = Point.create(2, 0);    // to provoke a large error
294                
295                ProcrustesFit pf = new ProcrustesFit(P, Q, allowTranslation, allowScaling, forceRotation);
296
297                System.out.format("estimated alpha: a = %.6f\n", Math.acos(pf.getR().getEntry(0, 0)));
298                System.out.println("estimated rotation: R = \n" + Matrix.toString(pf.getR().getData()));
299                System.out.println("estimated translation: t = " + Matrix.toString(pf.getT().toArray()));
300                System.out.format("estimated scale: s = %.6f\n", pf.getScale());
301                
302                System.out.println();
303                System.out.format("fitting error = %.6f\n", pf.getError());
304                System.out.format("euclidean error1 = %.6f\n", pf.getEuclideanError(P, Q));
305                System.out.format("euclidean error2 = %.6f\n", pf.getEuclideanError2(P, Q));
306                
307                RealMatrix M = pf.getTransformationMatrix();
308                System.out.println("transformation matrix: A = \n" + Matrix.toString(M.getData()));
309        }
