022        /**
023         * Creates and returns a 1D Gaussian filter kernel large enough
024         * to avoid truncation effects. The resulting array is odd-sized.
025         * @param sigma the width of the Gaussian
026         * @return a 1D, odd-sized filter kernel
027         */
028        public static float[] makeGaussKernel1D(double sigma) {
029                if (sigma < 0) {        // TODO: should check for larger sigma
030                        throw new IllegalArgumentException("Positive sigma required for Gaussian kernel!");
031                }
032                final int rad = (int) Math.ceil(3.5 * sigma);
033                int size = rad + rad + 1;
034                final float[] kernel = new float[size]; // odd size, center cell = kernel[rad]
035                final double sigma2 = sigma * sigma;
036                final double scale = 1.0 / Math.sqrt(2 * Math.PI * sigma2);
037                for (int i = 0; i < kernel.length; i++) {
038                        double x = rad - i;
039                        kernel[i] =  (float) (scale * Math.exp(-0.5 * (x * x) / sigma2));
040                }
041                return kernel;
042        }

077        /**
078         * Creates and returns a 2D Gaussian filter kernel large enough
079         * to avoid truncation effects. The resulting array is odd-sized.
080         * @param sigmaX the width of the Gaussian in x-direction
081         * @param sigmaY the width of the Gaussian in y-direction
082         * @return a 2D, odd-sized filter kernel
083         */
084        public static float[][] makeGaussKernel2D(double sigmaX, double sigmaY){
085                final int radX = (int) Math.ceil(3.5 * sigmaX);
086                final int radY = (int) Math.ceil(3.5 * sigmaY);
087                final int sizeX = radX + radX + 1;
088                final int sizeY = radY + radY + 1;
089
090                final float[][] kernel = new float[sizeX][sizeY]; //center cell = kernel[rad][rad]
091                final double sigmaX2 = (sigmaX > 0.1) ? sigmaX * sigmaX : 0.1;
092                final double sigmaY2 = (sigmaY > 0.1) ? sigmaY * sigmaY : 0.1;
093                
094                double sum = 0;
095                for (int i = 0; i < sizeX; i++) {
096                        double x = radX - i;
097                        for (int j = 0; j < sizeY; j++) {
098                                double y = radY - j;
099                                // IJ.log("x = " + x + " / " + "y = " + y);
100                                double g = (float) Math.exp(-((x * x) / (2 * sigmaX2) + (y * y) / (2 * sigmaY2)));
101                                // IJ.log("g = " + g);
102                                kernel[i][j] = (float) g;
103                                sum = sum + g;
104                        }
105                }
106
107                // normalize the kernel to sum 1
108                double scale = 1.0 / sum;
109                for (int i = 0; i < sizeX; i++) {
110                        for (int j = 0; j < sizeY; j++) {
111                                kernel[i][j] = (float) (kernel[i][j] * scale);
112                        }
113                }
114                return kernel;
115        }
