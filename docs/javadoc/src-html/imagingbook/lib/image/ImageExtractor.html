009package imagingbook.lib.image;
010
011import ij.process.ImageProcessor;
012import imagingbook.pub.geometry.basic.Point;
013import imagingbook.pub.geometry.mappings.linear.AffineMapping2D;
014import imagingbook.pub.geometry.mappings.linear.LinearMapping2D;
015import imagingbook.pub.geometry.mappings.linear.ProjectiveMapping2D;
016
017/**
018 * Use to exctract warped images for testing the Lucas-Kanade matcher.
019 * @author wilbur
020 *
021 */
022public class ImageExtractor {
023        
024

029        
030                
031        private int interpolationMethod = ImageProcessor.BILINEAR;
032        private final ImageProcessor I;
033        
034        /**
035         * Creates a new instance of {@link ImageExtractor} for the image {@code I}.
036         * @param I the target image.
037         */
038        public ImageExtractor(ImageProcessor I) {
039                this.I = I;
040        }
041        
042        public void setInterpolationMethod(int interpolationMethod) {
043                this.interpolationMethod = interpolationMethod;
044        }
045        
046        /**
047         * Extracts an image {@code R} of size {@code width} x {@code height} from the source image 
048         * {@code I} (referenced by {@code this} object).
049         * The image {@code R} is extracted from a quadrilateral patch of the source image,
050         * defined by the transformation of the boundary of {@code R} by {@code T(x)}.
051         * @param width the width of the target image {@code R}.
052         * @param height the height of the target image {@code R}.
053         * @param T a {@link LinearMapping2D} object.
054         * @return the extracted image {@code R}, which is of the same type as the source image.
055         */     
056
057        public ImageProcessor extractImage(int width, int height, LinearMapping2D T) {
058                ImageProcessor R = I.createProcessor(width, height);
059                extractImage(R, T);
060                return R;
061        }
062        
063        public ImageProcessor extractImage(int width, int height, Point[] sourcePnts) {
064                ImageProcessor R = I.createProcessor(width, height);
065                ProjectiveMapping2D T = getMapping(width, height, sourcePnts);
066                extractImage(R, T);
067                return R;
068        }
069        
070        /**
071         * Fills the image {@code R} from the source image 
072         * {@code I} (referenced by {@code this} object).
073         * The image {@code R} is extracted from a quadrilateral patch of the source image,
074         * defined by the transformation of the boundary of {@code R} by {@code T(x)}.
075         * Grayscale and color images my not be mixed (i.e., {@code R} must be of the same type as {@code I}).
076         * @param R the image to be filled.
077         * @param T a {@link LinearMapping2D} object.
078         */     
079        public void extractImage(ImageProcessor R, LinearMapping2D T) {
080                int prevInterpolationMethod = I.getInterpolationMethod();
081                // save current interpolation method
082                I.setInterpolationMethod(interpolationMethod);
083                
084                ImageAccessor iaI = ImageAccessor.create(I);
085                ImageAccessor iaR = ImageAccessor.create(R);
086        
087                int wT = R.getWidth();
088                int hT = R.getHeight();
089                for (int u = 0; u < wT; u++) {
090                        for (int v = 0; v < hT; v++) {
091                                Point uv = Point.create(u, v);
092                                Point xy = T.applyTo(uv);
093                                float[] val = iaI.getPix(xy.getX(), xy.getY());
094                                iaR.setPix(u, v, val);
095                        }
096                }
097                // restore interpolation method
098                I.setInterpolationMethod(prevInterpolationMethod);
099        }
100        
101        /**
102         * Extracts a warped sub-image of the associated target image I,
103         * defined by a sequence of 3 or 4 points. In the case of 3 
104         * points in sourcePnts, an {@link AffineMapping2D} is used; with 4 points,
105         * a {@link ProjectiveMapping2D} is used. The 3 or 4 points map clockwise to
106         * the corner points of the target image R, starting with the top-left corner.
107         * @param R the target image;
108         * @param sourcePnts an array of 3 or 4 {@link Point} objects.
109         */
110        public void extractImage(ImageProcessor R, Point[] sourcePnts) {
111                ProjectiveMapping2D T = getMapping(R.getWidth(), R.getHeight(), sourcePnts);
112                extractImage(R, T);
113        }
114        
115        private ProjectiveMapping2D getMapping(int w, int h, Point[] sourcePnts) {
116                Point[] targetPnts = {
117                                Point.create(0, 0), Point.create(w - 1, 0),
118                                Point.create(w - 1, h - 1), Point.create(0, h - 1)
119                        };
120                ProjectiveMapping2D T = null;
121                switch (sourcePnts.length) {
122                case (3) : T = AffineMapping2D.fromPoints(targetPnts, sourcePnts); break;
123                case (4) : T = ProjectiveMapping2D.fromPoints(targetPnts, sourcePnts); break;
124                default : throw new IllegalArgumentException("wrong number of source points");
125                }
126                return T;
127        }
128        
129}
