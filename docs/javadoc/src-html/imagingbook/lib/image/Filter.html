009package imagingbook.lib.image;
010
011import ij.plugin.filter.Convolver;
012import ij.process.ImageProcessor;
013
014/**
015 * Utility methods for filtering images. 
016 * None of the filter methods modifies the kernel, i.e., kernels are 
017 * used as supplied and never normalized.
018 * 
019 * @author WB
020 * @version 2020-02-19
021 */
022public abstract class Filter {
023
024        /**
025         * Applies a one-dimensional convolution kernel to the given image,
026         * which is modified. The 1D kernel is applied in horizontal direction only.
027         * @param fp the image to be filtered (modified)
028         * @param h the filter kernel
029         */
030        public static void convolveX (ImageProcessor fp, float[] h) {
031                Convolver conv = new Convolver();
032                conv.setNormalize(false);
033                conv.convolve(fp, h, h.length, 1);
034        }
035
036        /**
037         * Applies a one-dimensional convolution kernel to the given image,
038         * which is modified. The 1D kernel is applied in vertical direction only.
039         * @param fp the image to be filtered (modified)
040         * @param h the filter kernel
041         */
042        public static void convolveY (ImageProcessor fp, float[] h) {
043                Convolver conv = new Convolver();
044                conv.setNormalize(false);
045                conv.convolve(fp, h, 1, h.length);
046        }
047
048        /**
049         * Applies a one-dimensional convolution kernel to the given image,
050         * which is modified. The 1D kernel is applied twice in horizontal and
051         * vertical direction.
052         * @param fp the image to be filtered (modified)
053         * @param h the filter kernel
054         */
055        public static void convolveXY (ImageProcessor fp, float[] h) {
056                Convolver conv = new Convolver();
057                conv.setNormalize(false);
058                conv.convolve(fp, h, h.length, 1);
059                conv.convolve(fp, h, 1, h.length);
060//              convolveX(fp, h);
061//              convolveY(fp, h);
062        }
063
064}
