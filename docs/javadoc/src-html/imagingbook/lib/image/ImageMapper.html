001package imagingbook.lib.image;
002
003import java.awt.Point;
004import java.awt.geom.Point2D;
005
006import ij.process.ImageProcessor;
007import imagingbook.lib.interpolation.InterpolationMethod;
008import imagingbook.pub.geometry.mappings.Mapping;
009
010/**
011 * This class defines methods to perform arbitrary geometric transformations
012 * on images. The geometric transformation (mapping) must be specified at
013 * construction, optionally a pixel interpolation method can be specified.
014 * 
015 * @version 2019/01/04
016 *
017 */
018public class ImageMapper {
019        
020        /** Default pixel interpolation method (see {@link InterpolationMethod}). */
021        public static InterpolationMethod DefaultInterpolationMethod = InterpolationMethod.Bicubic;     
022        
023        private final InterpolationMethod im;
024        private final Mapping mp;
025        
026        /**
027         * Creates a new instance with the specified geometric mapping.
028         * The default pixel interpolation method is used
029         * (see {@link DefaultInterpolationMethod}).
030         * @param mp the geometric mapping
031         */
032        public ImageMapper(Mapping mp) {
033                this(mp, DefaultInterpolationMethod);
034        }
035
036        /**
037         * Creates a new instance with the specified geometric mapping
038         * and pixel interpolation method.
039         * @param mp the geometric mapping
040         * @param im the pixel interpolation method
041         */
042        public ImageMapper(Mapping mp, InterpolationMethod im) {
043                this.mp = mp;
044                this.im = im;
045        }
046        
047        // ---------------------------------------------------------------------
048        
049        /**
050         * Destructively transforms the passed image using this geometric
051         * mapping and the specified pixel interpolation method.
052         * The geometric mapping is supposed to be INVERTED, i.e. transforming
053         * target to source image coordinates!
054         * 
055         * @param ip target image to which this mapping is applied
056         */
057        public void map(ImageProcessor ip) {
058                // make a temporary copy of the image:
059                ImageProcessor source = ip.duplicate();
060                ImageProcessor target = ip;
061                map(source, target);
062                source = null;
063        }
064        
065        // ---------------------------------------------------------------------
066        
067        /**
068         * Transforms the source image to the target image using this geometric
069         * mapping and the specified pixel interpolation method. Note that source
070         * and target must be different images!
071         * The geometric mapping is supposed to be INVERTED, i.e. transforming
072         * target to source image coordinates!
073         * 
074         * @param source input image (not modified)
075         * @param target output image (modified)
076         */
077        public void map(ImageProcessor source, ImageProcessor target) {
078                if (target == source) {
079                        throw new IllegalArgumentException("Source and target image must not be the same!");
080                }
081                ImageAccessor sourceAcc = ImageAccessor.create(source, null, im);
082                ImageAccessor targetAcc = ImageAccessor.create(target);
083                map(sourceAcc, targetAcc);
084        }
085
086        
087        // ---------------------------------------------------------------------
088
089        /**
090         * Transforms the source image to the target image using this geometric
091         * mapping and the specified pixel interpolation method.
092         * The two images are passed as instances of {@link ImageAccessor}.
093         * Note that source and target reference different images!
094         * The geometric mapping is supposed to be INVERTED, i.e. transforming
095         * target to source image coordinates!
096         * 
097         * @param sourceAcc accessor to the source image
098         * @param targetAcc accessor to the target image
099         */
100        public void map(ImageAccessor sourceAcc, ImageAccessor targetAcc) {
101                if (targetAcc.getProcessor() == sourceAcc.getProcessor()) {
102                        throw new IllegalArgumentException("Source and target image must not be the same!");
103                }
104                Mapping invMap = mp;            // this always IS an inverse mapping!!
105                ImageProcessor target = targetAcc.getProcessor();
106                final int w = target.getWidth();
107                final int h = target.getHeight();
108                for (int v = 0; v < h; v++) {
109                        for (int u = 0; u < w; u++) {
110                                Point2D sourcePt = invMap.applyTo(new Point(u, v));
111                                float[] val = sourceAcc.getPix(sourcePt.getX(), sourcePt.getY());
112                                targetAcc.setPix(u, v, val);
113                        }
114                }
115        }
116
117}
