001package imagingbook.lib.image;
002
003import ij.process.ImageProcessor;
004import imagingbook.lib.interpolation.InterpolationMethod;
005import imagingbook.pub.geometry.basic.Point;
006import imagingbook.pub.geometry.mappings.Mapping2D;
007
008/**
009 * This class defines methods to perform arbitrary geometric transformations
010 * on images. The geometric transformation (mapping) must be specified at
011 * construction, optionally a pixel interpolation method can be specified.
012 * 
013 * @version 2019/01/04
014 *
015 */
016public class ImageMapper {
017        
018        /** Default pixel interpolation method (see {@link InterpolationMethod}). */
019        public static InterpolationMethod DefaultInterpolationMethod = InterpolationMethod.Bicubic;     
020        
021        private final InterpolationMethod im;
022        private final Mapping2D mp;
023        
024        /**
025         * Creates a new instance with the specified geometric mapping.
026         * The default pixel interpolation method is used
027         * (see {@link DefaultInterpolationMethod}).
028         * @param mp the geometric mapping
029         */
030        public ImageMapper(Mapping2D mp) {
031                this(mp, DefaultInterpolationMethod);
032        }
033
034        /**
035         * Creates a new instance with the specified geometric mapping
036         * and pixel interpolation method.
037         * @param mp the geometric mapping
038         * @param im the pixel interpolation method
039         */
040        public ImageMapper(Mapping2D mp, InterpolationMethod im) {
041                this.mp = mp;
042                this.im = im;
043        }
044        
045        // ---------------------------------------------------------------------
046        
047        /**
048         * Destructively transforms the passed image using this geometric
049         * mapping and the specified pixel interpolation method.
050         * The geometric mapping is supposed to be INVERTED, i.e. transforming
051         * target to source image coordinates!
052         * 
053         * @param ip target image to which this mapping is applied
054         */
055        public void map(ImageProcessor ip) {
056                // make a temporary copy of the image:
057                ImageProcessor source = ip.duplicate();
058                ImageProcessor target = ip;
059                map(source, target);
060                source = null;
061        }
062        
063        // ---------------------------------------------------------------------
064        
065        /**
066         * Transforms the source image to the target image using this geometric
067         * mapping and the specified pixel interpolation method. Note that source
068         * and target must be different images!
069         * The geometric mapping is supposed to be INVERTED, i.e. transforming
070         * target to source image coordinates!
071         * 
072         * @param source input image (not modified)
073         * @param target output image (modified)
074         */
075        public void map(ImageProcessor source, ImageProcessor target) {
076                if (target == source) {
077                        throw new IllegalArgumentException("Source and target image must not be the same!");
078                }
079                ImageAccessor sourceAcc = ImageAccessor.create(source, null, im);
080                ImageAccessor targetAcc = ImageAccessor.create(target);
081                map(sourceAcc, targetAcc);
082        }
083
084        
085        // ---------------------------------------------------------------------
086
087        /**
088         * Transforms the source image to the target image using this geometric
089         * mapping and the specified pixel interpolation method.
090         * The two images are passed as instances of {@link ImageAccessor}.
091         * Note that source and target reference different images!
092         * The geometric mapping is supposed to be INVERTED, i.e. transforming
093         * target to source image coordinates!
094         * 
095         * @param sourceAcc accessor to the source image
096         * @param targetAcc accessor to the target image
097         */
098        public void map(ImageAccessor sourceAcc, ImageAccessor targetAcc) {
099                if (targetAcc.getProcessor() == sourceAcc.getProcessor()) {
100                        throw new IllegalArgumentException("Source and target image must not be the same!");
101                }
102                Mapping2D invMap = mp;          // this always IS an inverse mapping!!
103                ImageProcessor target = targetAcc.getProcessor();
104                final int w = target.getWidth();
105                final int h = target.getHeight();
106                for (int v = 0; v < h; v++) {
107                        for (int u = 0; u < w; u++) {
108                                Point sourcePt = invMap.applyTo(Point.create(u, v));
109                                float[] val = sourceAcc.getPix(sourcePt.getX(), sourcePt.getY());
110                                targetAcc.setPix(u, v, val);
111                        }
112                }
113        }
114
115}
