001package imagingbook.lib.util;
002
003import java.io.BufferedInputStream;
004import java.io.BufferedOutputStream;
005import java.io.File;
006import java.io.FileInputStream;
007import java.io.FileOutputStream;
008import java.io.IOException;
009import java.io.InputStream;
010import java.io.ObjectInput;
011import java.io.ObjectInputStream;
012import java.io.ObjectOutput;
013import java.io.ObjectOutputStream;
014import java.io.OutputStream;
015
016/**
017 * Helper class providing simple static methods for writing/reading
018 * serialized data to/from files. It is recommended to serialize
019 * only data structures composed of standard Java types.
020 * Otherwise, if self-defined classes are reloaded, classes of 
021 * previously serialized objects may not match any more, causing
022 * a ClassNotFoundException to be thrown.
023 * 
024 * @author WB
025 * @version 2018/04/03
026 */
027public class SerializationHelper {
028        
029        // This class is not supposed to be instantiated.
030        private SerializationHelper() {
031        }
032        
033        /**
034         * Writes a serialized representation of an arbitrary Java object to 
035         * a file. Make sure the serialized object is composed of standard Java types 
036         * only to avoid class loader problems.
037         * @param obj The object to be serialized.
038         * @param fileName The file to write to.
039         * @return The full path of the written file.
040         * @deprecated
041         */
042        public static String writeObject(Object obj, String fileName) {
043                File file = new File(fileName);
044                return writeObject(obj, file);
045        }
046        
047        /**
048         * Writes a serialized representation of an arbitrary Java object to 
049         * a file.
050         * @param obj The object to be serialized.
051         * @param file The file to write to.
052         * @return The full path of the written file.
053         */
054        public static String writeObject(Object obj, File file) {
055                String path = file.getAbsolutePath();
056                try (FileOutputStream strm = new FileOutputStream(file);
057                         OutputStream buffer = new BufferedOutputStream(strm);
058                         ObjectOutput output = new ObjectOutputStream(buffer);) 
059                {
060                        output.writeObject(obj);
061                } catch (IOException e) {
062                        System.err.println(e.toString());
063                        return null;
064                }
065                return path;
066        }
067        
068        /**
069         * Reads an object (of known type) from a serialization file.
070         * The return value must be cast to the appropriate type, which
071         * must be known.
072         * @param fileName The file containing serialized data.
073         * @return The object reconstructed from the file representation or null if unsuccessful.
074         * @deprecated
075         */
076        public static Object readObject(String fileName) {
077                File file = new File(fileName);
078                return readObject(file);
079        }
080        
081
082        /**
083         * Reads an object (of known type) from a serialization file.
084         * The return value must be cast to the appropriate type, which
085         * must be known.
086         * @param file The file to read.
087         * @return The object reconstructed from the file representation or null if unsuccessful.
088         */
089        public static Object readObject(File file) {
090                Object obj = null;
091                try (InputStream strm = new FileInputStream(file);
092                         InputStream buffer = new BufferedInputStream(strm);
093                         ObjectInput input = new ObjectInputStream(buffer);) 
094                {
095                        obj = input.readObject();
096                } catch (ClassNotFoundException | IOException e) {
097                        System.err.println(e.toString());
098                        return null;
099                }
100                return obj;
101        }
102
103}
