001package imagingbook.lib.util;
002
003import java.lang.reflect.Field;
004
005public abstract class ClassUtils {
006        
007        /**
008         * Lists all accessible public fields of the given object and 
009         * returns the result as a string.
010         * @param obj a (non-null) object 
011         * @return a string listing the names and values of the object's fields
012         */
013        public static String listFields(Object obj) {
014                Class<?> clz = obj.getClass();
015                StringBuilder buf = new StringBuilder(clz.getName() + ":\n");
016                Field[] fields = clz.getFields();
017                for (Field f : fields) {
018                        String name = f.getName();
019                        String value = null;
020                        try {
021                                value = f.get(obj).toString();
022                        } catch (IllegalArgumentException | IllegalAccessException e1) {                        }
023                        buf.append(name + ": " + value + "\n");
024                }
025                return buf.toString();
026        }
027
028}
