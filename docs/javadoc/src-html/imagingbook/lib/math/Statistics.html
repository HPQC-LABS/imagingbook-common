010package imagingbook.lib.math;
011
012import org.apache.commons.math3.linear.MatrixUtils;
013//import org.apache.commons.math3.linear.MatrixUtils;
014import org.apache.commons.math3.stat.correlation.Covariance;
015
016/**
017 * Utility class defines a few statistical methods.
018 * @author WB
019 *
020 */
021public abstract class Statistics {
022        
023        /**
024         * Calculates the covariance matrix for a sequence of sample vectors.
025         * Takes a sequence of n data samples, each of dimension m.
026         * The data element samples[i][j] refers to the j-th component
027         * of sample i. No statistical bias correction is applied.
028         * @param samples Array of m-dimensional vectors (double[n][m]).
029         * @return The covariance matrix (of dimension double{m][m]).
030         */
031        public static double[][] covarianceMatrix(double[][] samples) {
032                return covarianceMatrix(samples, false);
033        }
034        
035        /**
036         * Calculates the covariance matrix for a sequence of sample vectors.
037         * Takes a sequence of n data samples, each of dimension m.
038         * The data element samples[i][j] refers to the j-th component
039         * of sample i. No statistical bias correction is applied.
040         * @param samples Array of m-dimensional vectors (double[n][m]).
041         * @param biasCorrect If {@code true}, statistical bias correction is applied.
042         * @return The covariance matrix (of dimension double{m][m]).
043         */
044        public static double[][] covarianceMatrix(double[][] samples, boolean biasCorrect) {
045                Covariance cov = new Covariance(samples, biasCorrect);
046                return cov.getCovarianceMatrix().getData();
047        }
048        
049        /** 
050         * example from UTICS-C Appendix:
051         * N = 4 samples
052         * K = 3 dimensions
053         * @param args ignored
054         */
055        public static void main(String[] args) {
056                
057                boolean BIAS_CORRECT = false;
058                
059                // example: n = 4 samples of dimension m = 3:
060                // samples[i][j], i = column (sample index), j = row (dimension index).
061                double[][] samples = { 
062                                {75, 37, 12},   // i = 0
063                                {41, 27, 20},   // i = 1
064                                {93, 81, 11},   // i = 2
065                                {12, 48, 52}    // i = 3
066                };
067                
068                // covariance matrix Cov (3x3)
069                double[][] cov = covarianceMatrix(samples, BIAS_CORRECT);
070                System.out.println("cov = \n" + Matrix.toString(cov));
071                
072                System.out.println();
073                
074                double[][] icov = Matrix.inverse(cov);
075                System.out.println("icov = \n" + Matrix.toString(icov));
076                
077                System.out.println();
078                
079                double trace = MatrixUtils.createRealMatrix(cov).getTrace();
080                System.out.println("trace(cov) = " + trace);
081                double Fnorm = MatrixUtils.createRealMatrix(cov).getFrobeniusNorm();
082                System.out.println("Fnorm(cov) = " + Fnorm);
083        }
084        
085/* Results (bias-corrected):
086cov = 
087{{1296.250, 442.583, -627.250}, 
088{442.583, 550.250, -70.917}, 
089{-627.250, -70.917, 370.917}}
090
091icov = 
092{{0.024, -0.014, 0.038}, 
093{-0.014, 0.011, -0.022}, 
094{0.038, -0.022, 0.063}}
095*/
096        
097/* verified with Mathematica
098X1 = {75, 37, 12}; X2 = {41, 27, 20}; X3 = {93, 81, 11}; X4 = {12, 48, 52};
099samples = {X1, X2, X3, X4}
100N[Covariance[samples]]
101-> {{1296.25, 442.583, -627.25}, {442.583, 550.25, -70.9167}, {-627.25, -70.9167, 370.917}}
102*/
103

105cov =

109
110icov = {{0.032, -0.019, 0.051}, 
111{-0.019, 0.014, -0.030}, 
112{0.051, -0.030, 0.083}}
113*/
