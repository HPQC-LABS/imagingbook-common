012import org.apache.commons.math3.linear.CholeskyDecomposition;
013import org.apache.commons.math3.linear.EigenDecomposition;
014import org.apache.commons.math3.linear.MatrixUtils;

018/**
019 * This class implements the Mahalanobis distance using the Apache commons math library.
020 * No statistical bias correction is used.
021 * TODO: Check/reuse methods for covariance matrix etc in class 'Statistics'!
022 */
023public class MahalanobisDistance { // extends VectorNorm 
024
025        /** The default minimum diagonal value used to condition the covariance matrix. */
026        public static final double DEFAULT_MIN_DIAGONAL_VALUE = 1.0E-15;
027        
028        /** Used for Cholesky decomposition. 
029         * See {@link CholeskyDecomposition#DEFAULT_RELATIVE_SYMMETRY_THRESHOLD} (1.0E-15), which seems too small). */
030        
031        public static double RELATIVE_SYMMETRY_THRESHOLD = 1.0E-6;
032        /** Used for Cholesky decomposition.
033         * See {@link CholeskyDecomposition#DEFAULT_ABSOLUTE_POSITIVITY_THRESHOLD}. */
034        
035        public static double ABSOLUTE_POSITIVITY_THRESHOLD = CholeskyDecomposition.DEFAULT_ABSOLUTE_POSITIVITY_THRESHOLD;
036
037        private final int M;                                    // feature dimension
038        private final double[] mean;                    // the distribution's mean vector (\mu)
039        private final double[][] cov;                   // covariance matrix of size n x n
040        private final double[][] icov;                  // inverse covariance matrix of size n x n
041
042        /**
043         * Create a new instance from an array of m-dimensional samples, e.g.,
044         * samples = {{x1,y1}, {x2,y2},...,{xn,yn}} for a vector of n two-dimensional
045         * samples. Uses {@link #DEFAULT_MIN_DIAGONAL_VALUE} to condition the 
046         * covariance matrix.
047         * @param samples a sequence of m-dimensional samples, i.e.,
048         *      samples[k][i] represents the i-th component of the k-th sample
049         */
050        public MahalanobisDistance(double[][] samples) {
051                this(samples, DEFAULT_MIN_DIAGONAL_VALUE);
052        }
053        
054        /**
055         * Create a new instance from an array of m-dimensional samples, e.g.
056         * samples = {{x1,y1}, {x2,y2},...,{xn,yn}} for a vector of n two-dimensional
057         * samples.
058         * @param samples a vector of length n with m-dimensional samples, i.e.,
059         *      samples[k][i] represents the i-th component of the k-th sample
060         * @param minDiagVal the minimum diagonal value used to condition the
061         *      covariance matrix to avoid singularity (see {@link #DEFAULT_MIN_DIAGONAL_VALUE})
062         */
063        public MahalanobisDistance(double[][] samples, double minDiagVal) {
064                this(makeCovarianceMatrix(samples, minDiagVal), makeMeanVector(samples));
065        }
066
067        /**
068         * Create a new instance from the m x m covariance matrix and the m-dimensional
069         * mean vector of a distribution of m-dimensional data. The covariance matrix is 
070         * assumed to be properly conditioned, i.e., has all-positive diagonal values.
071         * @param cov the covariance matrix of size m x m
072         * @param mean the mean vector of length m
073         */
074        public MahalanobisDistance(double[][] cov, double[] mean) {
075                this.cov = cov;
076                this.mean = mean;
077                this.M = mean.length;
078                RealMatrix S = MatrixUtils.createRealMatrix(cov);
079                // get the inverse covariance matrix
080                this.icov = MatrixUtils.inverse(S).getData();   // not always symmetric?
081        }
082
083        // 
084        /**
085         * Conditions the supplied covariance matrix by enforcing
086         * positive eigenvalues on its main diagonal. 
087         * @param S original covariance matrix
088         * @param minDiagVal the minimum positive value of diagonal elements
089         * @return conditioned covariance matrix
090         */
091        private static RealMatrix conditionCovarianceMatrix(RealMatrix S, double minDiagVal) {
092                EigenDecomposition ed = new EigenDecomposition(S);  // S  ->  V . D . V^T
093                RealMatrix V  = ed.getV();
094                RealMatrix D  = ed.getD();      // diagonal matrix of eigenvalues
095                RealMatrix VT = ed.getVT();
096                for (int i = 0; i < D.getRowDimension(); i++) {
097                        D.setEntry(i, i, Math.max(D.getEntry(i, i), minDiagVal));       // setting eigenvalues to zero is not enough!
098                }
099                return V.multiply(D).multiply(VT);
100        }

102        // --------------------------------------------------------------
103
104        private static double[] makeMeanVector(double[][] samples) {
105                int n = samples.length;
106                int m = samples[0].length;
107                double[] mu = new double[m];
108                for (int k = 0; k < n; k++) {
109                        for (int i = 0; i < m; i++) {
110                                mu[i] = mu[i] + samples[k][i];
111                        }
112                }
113                for (int i = 0; i < m; i++) {
114                        mu[i] = mu[i] / n;
115                }
116                return mu;
117        }
118
119        /**
120         * Calculates the covariance matrix for a sequence of m-dimensional samples, e.g.,
121         * samples = {{x1,y1}, {x2,y2},...,{xn,yn}} for n two-dimensional
122         * samples. The covariance matrix is conditioned to avoid negative or zero
123         * values on its main diagonal.
124         * @param samples a sequence of m-dimensional samples, i.e.,
125         *      samples[k][i] represents the i-th component of the k-th sample
126         * @param minDiagVal the minimum positive value of covariance matrix diagonal elements
127         *              (see {@link #DEFAULT_MIN_DIAGONAL_VALUE})
128         * 
129         * @return the covariance matrix for the supplied samples
130         */
131        public static double[][] makeCovarianceMatrix(double[][] samples, double minDiagVal) {
132                if (samples.length < 1)
133                        throw new IllegalArgumentException("number of samples must be > 0");
134                if (samples[0].length < 1)
135                        throw new IllegalArgumentException("sample dimension must be > 0");
136                RealMatrix S = new Covariance(samples, false).getCovarianceMatrix();
137                // condition the covariance matrix to avoid singularity:
138                S = conditionCovarianceMatrix(S, minDiagVal);
139                return S.getData();
140        }

143         * Returns the sample dimension (M) for this instance.
144         * @return the sample dimension
145         */
146        public int getDimension() {
147                return M;
148        }

170                return mean;

175        /**
176         * Returns the 'root' (U) of the inverse covariance matrix S^{-1},
177         * such that S^{-1} = U^T . U
178         * This matrix can be used to pre-transform the original sample
179         * vectors X (by X &#8594; U . X) to a space where distance (in the Mahalanobis
180         * sense) can be measured with the usual Euclidean norm.
181         * The matrix U is invertible in case the reverse transformation is required.
182         * 
183         * @return the m x m matrix for pre-transforming the original (m-dimensional) sample vectors
184         */
185        public double[][] getWhiteningTransformation() {
186                CholeskyDecomposition cd = 
187                                new CholeskyDecomposition(MatrixUtils.createRealMatrix(icov),
188                                                RELATIVE_SYMMETRY_THRESHOLD, ABSOLUTE_POSITIVITY_THRESHOLD);
189                RealMatrix U = cd.getLT();
190                return U.getData();
191        }

195        /**
196         * Returns the Mahalanobis distance between the given points.
197         * @param X first point
198         * @param Y second point
199         * @return the Mahalanobis distance
200         */
201        public double distance(double[] X, double[] Y) {
202                return Math.sqrt(distance2(X, Y));
203        }
204
205        /**
206         * Returns the squared Mahalanobis distance between the given points.
207         * @param X first point
208         * @param Y second point
209         * @return the squared Mahalanobis distance
210         */
211        public double distance2(double[] X, double[] Y) {
212                if (X.length != M || Y.length != M) {
213                        throw new IllegalArgumentException("vectors must be of length " + M);
214                }
215                // TODO: implement with methods from Matrix class
216                // d^2(X,Y) = (X-Y)^T . S^{-1} . (X-Y)
217                double[] dXY = new double[M]; // = (X-Y)
218                for (int k = 0; k < M; k++) {
219                        dXY[k] = X[k] - Y[k];
220                }
221                double[] iSdXY = new double[M]; // = S^{-1} . (X-Y)
222                for (int k = 0; k < M; k++) {
223                        for (int i = 0; i < M; i++) {
224                                iSdXY[k] += icov[i][k] * dXY[i];
225                        }
226                }
227                double d2 = 0;  // final sum
228                for (int k = 0; k < M; k++) {
229                        d2 += dXY[k] * iSdXY[k];
230                }                                               // d = (X-Y)^T . S^{-1} . (X-Y)
231                return d2;
232        }
233
234        /**
235         * Calculates the Mahalanobis distance between point x and
236         * the mean of the reference distribution.
237         * @param x an arbitrary m-dimensional vector
238         * @return the Mahalanobis distance between the point x and
239         *                      the mean of the reference distribution
240         */
241        public double distance(double[] x) {
242                return distance(x, mean);
243        }
244
245        /**
246         * Calculates the sqared Mahalanobis distance between point x and
247         * the mean of the reference distribution.
248         * @param x an arbitrary m-dimensional vector
249         * @return the squared Mahalanobis distance between the point x and
250         *                      the mean of the reference distribution
251         */
252        public double distance2(double[] x) {
253                return distance2(x, mean);
254        }
