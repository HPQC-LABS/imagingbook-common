010package imagingbook.lib.math;
011
012public abstract class Arithmetic {
013        
014        // machine accuracy for IEEE 754 float/double;
015        public static final float EPSILON_FLOAT         = 1e-7f;        // 1.19 x 10^-7
016        public static final double EPSILON_DOUBLE       = 2e-16;        // 2.22 x 10^-16
017
018        public static int sqr(int x) {
019                return x * x;
020        }
021        
022        public static float sqr(float x) {
023                return x * x;
024        }
025        
026        public static double sqr(double x) {
027                return x * x;
028        }
029        
030        /**
031         * Integer version of the modulus operator ({@code a mod b}).
032         * Also see <a href="http://en.wikipedia.org/wiki/Modulo_operation">here</a>.
033         * This implementation has the same behavior as the {@code Math.floorMod(a,b)},
034         * available in Java 8 and higher.
035         * An exception is thrown if b is zero.
036         * @param a the dividend
037         * @param b the divisor
038         * @return {@code a mod b}
039         */
040        public static int mod(int a, int b) {
041                return Math.floorMod(a, b);
042        }
043        
044        // original implementation (obsolete)
045//      public static int mod(int a, int b) {
046//              if (b == 0)
047//                      return a;
048//              // a, b have the same sign OR the remainder is zero
049//        if ((long)a * b >= 0 || a % b == 0)   // or (a / b) * b == a
050//              return a - b * (a / b);
051//        else
052//              return a - b * (a / b - 1);
053//      }
054
055        
056        /**
057         * Non-integer version of modulus operator, with results identical to Mathematica. 
058         * Calculates {@code a mod b}.
059         * Also see <a href="http://en.wikipedia.org/wiki/Modulo_operation">here</a>.
060         * An exception is thrown if b is zero.
061         * @param a dividend
062         * @param b divisor
063         * @return {@code a mod b}
064         */
065        public static double mod(double a, double b) {
066                if (isZero(b))
067                                throw new DivideByZeroException();
068                return a - b * Math.floor(a / b);
069        }
070        
071        /**
072         * Test for zero (float version).
073         * @param x quantity to be tested
074         * @return true if argument is close to zero
075         */
076        public static boolean isZero(float x) {
077                return Math.abs(x) < Arithmetic.EPSILON_FLOAT;
078        }
079        
080        /**
081         * Test for zero (double version).
082         * @param x quantity to be tested
083         * @return true if argument is close to zero
084         */
085        public static boolean isZero(double x) {
086                return Math.abs(x) < Arithmetic.EPSILON_DOUBLE;
087        }
088        
089        //--------------------------------------------------------------------------
090        
091        public static class DivideByZeroException extends ArithmeticException {
092                private static final long serialVersionUID = 1L;
093                private static String DefaultMessage = "zero denominator in division";
094                
095                public DivideByZeroException() {
096                        super(DefaultMessage);
097                }
098        }
099        
100        // -------------------------------------
101        
102        public static void main(String[] args) {
103                System.out.println(Arithmetic.mod(13, 4));
104                System.out.println(Arithmetic.mod(13, -4));
105                System.out.println(Arithmetic.mod(-13, 4));
106                System.out.println(Arithmetic.mod(-13, -4));
107                
108                int b = 7;
109                System.out.format("   i  -> floor |  mod\n");
110                for (int i = -25; i < 25; i++) {
111                        System.out.format("%4d  -> %4d  | %4d \n", i, Math.floorMod(i, b), Arithmetic.mod(i, b));
112                }
113        }
114
115}
