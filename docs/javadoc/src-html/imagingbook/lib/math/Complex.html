010package imagingbook.lib.math;
011
012/**
013 * This class represents complex numbers. Instances are immutable.
014 * Methods are mostly defined to be compatible with 
015 * org.apache.commons.math3.complex.Complex and (newer)
016 * org.apache.commons.numbers.complex.Complex.
017 * Arithmetic operations are generally more accurate than with the
018 * Apache implementation.
019 * 
020 * @author W. Burger
021 * @version 2018/03/18
022 */
023public class Complex {
024        
025        public static final Complex ONE = new Complex(1, 0);
026        public static final Complex ZERO = new Complex(0, 0);
027        public static final Complex I = new Complex(0, 1);
028
029        /** The real part of this complex number */
030        public final double re;
031        /** The imaginary part of this complex number */
032        public final double im;
033
034        /**
035         * Constructor.
036         * @param re real part
037         * @param im imaginary part
038         */
039        public Complex(double re, double im) {
040                this.re = re;
041                this.im = im;
042        }
043
044        /**
045         * Constructor
046         * @param z complex quantity, which is duplicated
047         */
048        public Complex(Complex z) {
049                this.re = z.re;
050                this.im = z.im;
051        }
052
053        /**
054         * Create a complex quantity on the unit circle with angle 'phi':
055         * e^{\i \phi} = \cos(\phi) + \i \cdot \sin(\phi)
056         * @param phi angle
057         */
058        public Complex(double phi) {
059                this.re = Math.cos(phi);
060                this.im = Math.sin(phi);
061        }
062
063        /**
064         * Returns the absolute value of this complex number, i.e.,
065         * its radius or distance from the origin.
066         * @return the absolute value
067         */
068        public double abs() {
069                return Math.sqrt(this.abs2());
070        }
071
072        /**
073         * Returns the squared absolute value of this complex number, i.e.,
074         * its squared radius or distance from the origin.
075         * @return the squared absolute value
076         */
077        public double abs2() {
078                return re * re + im * im;
079        }
080
081        /**
082         * Returns the 'argument' of this complex number, i.e.,
083         * its angle relative to the real axis. 
084         * @return the argument (in radians)
085         */
086        public double arg() {
087                return Math.atan2(im, re);
088        }
089
090        /**
091     * Returns the conjugate {@code z*} of this complex number, i.e,
092     * if {@code z = a + i b} then {@code z* = a - i b}.
093     *
094     * @return the complex conjugate
095     */
096        public Complex conjugate() {
097                return new Complex(this.re, -this.im);
098        }
099
100        /**
101         * Adds a complex quantity to this complex number and returns
102         * a new complex number.
103         * @param z complex value
104         * @return the sum of this complex number and {@code z}
105         */
106        public Complex add(Complex z) {
107                return new Complex(this.re + z.re,  this.im + z.im);
108        }
109
110        /**
111         * Rotates this complex number by the angle {@code phi} and returns
112         * a new complex number.
113         * @param phi the angle
114         * @return the rotated complex value
115         */
116        public Complex rotate(double phi) {
117                return this.multiply(new Complex(phi));
118        }
119
120        @Override
121    public String toString() {
122        return "(" + this.re + ", " + this.im + ")";
123    }
124        
125    /**
126     * Returns true if the real or imaginary component of this complex number
127     * is {@code NaN}.
128     * @return true if {@code NaN}, otherwise false
129     */
130        public boolean isNaN() {
131                return Double.isNaN(this.getReal()) || Double.isNaN(this.getImaginary());
132        }
133        
134        /**
135         * Returns the real part of this complex number.
136         * @return the real part
137         */
138        public double getReal() {
139                return this.re;
140        }
141        
142        /**
143         * Returns the imaginary part of this complex number.
144         * @return the imaginary part
145         */
146        public double getImaginary() {
147                return this.im;
148        }
149
150        /**
151         * Multiplies this complex number with another complex quantity {@code z2} and returns
152         * a new complex number.
153         * @param z2 a complex quantity
154         * @return this complex number multiplied by {@code z2}
155         */
156        public Complex multiply(Complex z2) {
157                // (x1 + i y1)(x2 + i y2) = (x1 x2 + y1 y2) + i (x1 y2 + y1 x2)
158                Complex z1 = this;
159                double x = z1.re * z2.re - z1.im * z2.im;
160                double y = z1.re * z2.im + z1.im * z2.re;
161                return new Complex(x, y);
162        }
163        
164        /**
165         * Multiplies this complex number with the scalar factor {@code s} and returns
166         * a new complex number.
167         * @param s a scalar factor
168         * @return this complex number multiplied by {@code s}
169         */
170        public Complex multiply(double s) {
171                return new Complex(this.re * s, this.im * s);
172        }
173        
174         /**
175     * Returns of value of this complex number raised to the power of {@code k}.
176     * @param k the exponent
177     * @return {@code z^k}
178     */
179        public Complex pow(int k) {
180                if (k < 0) throw new IllegalArgumentException("exponent k >= 0 expected");
181                Complex prod = new Complex(1, 0);
182                for (int i = 0; i < k; i++) {
183                        prod = prod.multiply(this);
184                }
185                return prod;
186        }
187
188        //------------ TESTING only ------------------------------
189
190        public static void main(String[] args) {
191                Complex z1 = new Complex(0.3, 0.6);
192                Complex z2 = new Complex(-1, 0.2);
193                System.out.println("z1 = " + z1);
194                System.out.println("z2 = " + z2);
195                Complex z3 = z1.multiply(z2);
196                System.out.println("z3 = " + z3);
197                Complex z4 = z2.multiply(z1);
198                System.out.println("z4 = " + z4);
199        }
200
201}
