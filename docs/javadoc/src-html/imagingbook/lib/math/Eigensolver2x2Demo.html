001package imagingbook.lib.math;
002
003import imagingbook.lib.math.Eigensolver2x2;
004import imagingbook.lib.math.Matrix;
005import imagingbook.lib.settings.PrintPrecision;
006
007/**
008 * This is a small demo program to show the use of the
009 * {@link Eigensolver2x2} class.
010 * It takes a 2x2 real-valued matrix and calculates its two
011 * eigenvalues and eigenvectors.
012 * Results are printed and checked.
013 * 
014 * @author WB
015 *@version 2020/03/25
016 */
017public class Eigensolver2x2Demo {
018        
019        public static void main(String[] args) {
020                
021                PrintPrecision.set(6);
022                
023                // specify a 2x2 matrix
024                double[][] M = {
025                                {3, -2},
026                                {-4, 1}};
027                
028                System.out.format("M =\n%s\n\n", Matrix.toString(M));
029                
030                Eigensolver2x2 solver = new Eigensolver2x2(M);
031                double[]   eigenvalues  = solver.getEigenvalues();
032                double[][] eigenvectors = solver.getEigenvectors();
033                
034                System.out.println("eigenvalues/eigenvectors: check if M*x = \u03BB*x \n");
035                
036                for (int i = 1; i <= 2; i++) {
037                        // get the eigen-pair <lambda_i, x_i>
038                        double lambda = eigenvalues[i-1];
039                        double[] x = eigenvectors[i-1];
040                        
041                        System.out.format("\u03BB_%d  = %.6f\n", i, lambda);
042                        System.out.format("x_%d = %s\n", i, Matrix.toString(x));
043                        
044                        System.out.format("Check: \u03BB_%d * x_%d = %s\n", i, i, Matrix.toString(Matrix.multiply(lambda, x)));
045                        System.out.format("Check:   M * x_%d = %s\n\n", i, Matrix.toString(Matrix.multiply(M, x)));
046                }
047
048        }
049}
