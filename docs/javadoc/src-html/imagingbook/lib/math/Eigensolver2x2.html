016/**
017 * Implements a closed form algorithm for calculating the real eigenvalues (&lambda;) and 
018 * eigenvectors (x) of a 2x2 matrix M of the form
019 * <pre>
020 *   | A B |
021 *   | C D | </pre>
022 * There are typically (but not in general) two pairs of real-valued 
023 * solutions 
024 * &lang;&lambda;<sub>1</sub>, x<sub>1</sub>&rang;,
025 * &lang;&lambda;<sub>2</sub>, x<sub>2</sub>&rang;
026 * such that M&middot;x<sub>k</sub> = &lambda;<sub>k</sub>&middot;x<sub>k</sub>.
027 * <p>
028 * This implementation is inspired by Blinn, Jim: "Jim Blinn's Corner: 
029 * Notation, Notation, Notation", Morgan Kaufmann (2002) -
030 * Ch. 5 ("Consider the Lowly 2x2 Matrix").
031 * Note that Blinn uses the (common computer graphics) notation 
032 * x&middot;M = &lambda;&middot;x,
033 * while this implementation adopts the notation 
034 * M&middot;x = &lambda;&middot;x, 
035 * i.e., the matrix M is transposed (elements B/C are exchanged).
036 * </p>
037 * 
038 * @author W. Burger
039 * @version 2020-02-09
040 */
041public class Eigensolver2x2 {
042        
043        private final boolean isReal;
044        private final double[] eVals = {Double.NaN, Double.NaN};
045        private final double[][] eVecs = new double[2][];
046        
047        /**
048         * Constructor, takes a 2x2 matrix.
049         * @param M a 2x2 matrix
050         */
051        public Eigensolver2x2(double[][] M) {
052                this(M[0][0], M[0][1], M[1][0], M[1][1]);
053        }
054        
055        /**
056         * Constructor, takes the individual elements of a 2x2 matrix.
057         * @param A matrix element M[0,0]
058         * @param B matrix element M[0,1]
059         * @param C matrix element M[1,0]
060         * @param D matrix element M[1,1]
061         */
062        public Eigensolver2x2(double A, double B, double C, double D) {
063                this.isReal = makeEigenvectors(A, B, C, D);
064        }
065        
066        private boolean makeEigenvectors(double A, double B, double C, double D) {
067                final double R = (A + D) / 2;
068                final double S = (A - D) / 2;
069                final double rho = S * S + B * C;
070                
071                if (rho < 0) {  // no real-valued eigenvalues
072                        return false;
073                }
074                
075                final double T = Math.sqrt(rho);
076                eVals[0] = R + T;
077                eVals[1] = R - T;
078                if (A - D > 0) {
079                        eVecs[0] = new double[] { S + T, C };
080                        eVecs[1] = new double[] { B, -S - T };
081                }
082                else if (A - D < 0) {
083                        eVecs[0] = new double[] { B, -S + T};
084                        eVecs[1] = new double[] { S - T, C};
085                }
086                else {          // (A - D) == 0
087                        
088                        final double aB = Math.abs(B);
089                        final double aC = Math.abs(C);
090                        final double sBC = Math.sqrt(B * C);
091                        if (aB < aC) {                                                  // |B| < |C|
092                                eVecs[0] = new double[] { sBC, C};
093                                eVecs[1] = new double[] {-sBC, C};
094                        }
095                        else if (aB > aC) {                                     // |B| > |C|
096                                eVecs[0] = new double[] { B, sBC};
097                                eVecs[1] = new double[] { B, -sBC};
098                        }
099                        else {                                                                  // |B| == |C|
100                                eVecs[0] = new double[] { C, C};
101                                eVecs[1] = new double[] { -C, C};
102                        }
103                }
104                return true;
105        }
106        
107        /**
108         * Returns {@code true} iff the associated matrix has real 
109         * eigenvalues.
110         * @return as described
111         */
112        public boolean isReal() {
113                return isReal;
114        }
115        
116        /**
117         * Returns a vector holding the two real eigenvalues. Both are
118         * {@code NaN} if the associated matrix has no real eigenvalues.
119         * The first eigenvalue is the greater of the two.
120         * Note that the method returns a reference to an internal
121         * array and should thus be used read-only!
122         * @return a 2-element array of eigenvalues
123         */
124        public double[] getEigenvalues() {
125                return this.eVals;
126        }
127        
128        /**
129         * Returns a 2x2 array holding the two real eigenvectors.
130         * When used in the form
131         * <pre>double[][] X = getEigenvalues();</pre>
132         * then {@code X[0]} and {@code X[1]} contain the 1st and 2nd eigenvector,
133         * respectively. If the associated matrix has no real eigenvalues,
134         * then {@code X[0]} and {@code X[1]} are both {@code null}.
135         * Otherwise {@code X[k][0]} and {@code X[k][1]} are the x/y components
136         * of the k-th eigenvector.
137         * The ordering of the returned eigenvectors is the same as for the
138         * eigenvalues returned by {@link #getEigenvalues()}.
139         * Note that the method returns a reference to an internal
140         * array and should thus be used read-only!
141         * @return a 2-element array of 2D eigenvectors
142         */
143        public double[][] getEigenvectors() {
144                return this.eVecs;
145        }

149        private static void checkEigen(double[][] M) {
150                System.out.format("Checking M  = \n%s\n", Matrix.toString(M));
151                Eigensolver2x2 es = new Eigensolver2x2(M);
152                double[] eVals = es.getEigenvalues();
153                double[][] eVecs = es.getEigenvectors();
154                
155                for (int i = 0; i < 2; i++) {
156                        checkEigenPair(i+1, M, eVals[i], eVecs[i]);
157                }
158                System.out.println();
159        }
160        
161        private static void checkEigenPair(int i, double[][] M, double lambda, double[] x) {
162                double[] Mx = Matrix.multiply(M, x);
163                double[] lx = Matrix.multiply(lambda, x);
164                double d = Matrix.normL2(Matrix.add(Mx, Matrix.multiply(-1, lx)));
165                System.out.format("    \u03BB_%d  = %.6f\n", i, lambda);
166                System.out.format("    x_%d = %s\n", i, Matrix.toString(x));
167//              System.out.format("    M.x = %s\n", Matrix.toString(Mx));
168//              System.out.format("    \u03BB.x = %s\n", Matrix.toString(lx));
169//              System.out.format("    error = %.6f\n", d);
170                if (d > 0.000001) {
171                        System.out.println("    **** WRONG ****");
172                }
173        }
