016/**
017 * Instances of class calculate the real eigenvalues (&lambda;) and 
018 * eigenvectors (x) for a 2x2 matrix M of the form
019 * <pre>
020 *   | A B |
021 *   | C D | </pre>
022 * There are typically (but not in general) two pairs of real-valued 
023 * solutions (&lambda;, x).
024 * such that M*x = &lambda;*x.
025 * <p>
026 * This implementation is inspired by Blinn, Jim: "Jim Blinn's Corner: 
027 * Notation, Notation, Notation", Morgan Kaufmann (2002) -
028 * Ch. 5 ("Consider the Lowly 2x2 Matrix").
029 * Note that Blinn uses the (common computer graphics) notation 
030 * x*M = &lambda;*x,
031 * while this implementation adopts the notation 
032 * M*x = &lambda;*x, 
033 * i.e., the matrix M is transposed (elements B/C are exchanged).
034 * </p>
035 * 
036 * @author W. Burger
037 * @version 2020-02-08
038 */
039public class Eigensolver2x2 {
040        
041        private final boolean isReal;
042        private final double[] eVals = {Double.NaN, Double.NaN};
043        private final double[][] eVecs = new double[2][];
044        
045        public Eigensolver2x2(double[][] M) {
046                this(M[0][0], M[0][1], M[1][0], M[1][1]);
047        }
048        
049        public Eigensolver2x2(double A, double B, double C, double D) {
050                this.isReal = makeEigenvectors(A, B, C, D);
051        }
052        
053        private boolean makeEigenvectors(double A, double B, double C, double D) {
054                final double R = (A + D) / 2;
055                final double S = (A - D) / 2;
056                final double T = Math.sqrt(S * S + B * C);      // = sqrt(rho) = NaN if negative argument
057                eVals[0] = R + T;
058                eVals[1] = R - T;
059                if (Double.isNaN(T)) {
060                        return false;
061                }
062                if (A - D > 0) {
063                        eVecs[0] = new double[] { S + T, C };
064                        eVecs[1] = new double[] { B, -S - T };
065                }
066                else if (A - D < 0) {
067                        eVecs[0] = new double[] { B, -S + T};
068                        eVecs[1] = new double[] { S - T, C};
069                }
070                else {          // (A - D) == 0
071                        
072                        final double aB = Math.abs(B);
073                        final double aC = Math.abs(C);
074                        final double sBC = Math.sqrt(B * C);
075                        if (aB < aC) {                                                  // |B| < |C|
076                                eVecs[0] = new double[] { sBC, C};
077                                eVecs[1] = new double[] {-sBC, C};
078                        }
079                        else if (aB > aC) {                                     // |B| > |C|
080                                eVecs[0] = new double[] { B, sBC};
081                                eVecs[1] = new double[] { B, -sBC};
082                        }
083                        else {                                                                  // |B| == |C|
084                                eVecs[0] = new double[] { B, B};
085                                eVecs[1] = new double[] { -B, B};
086                        }
087                }
088                return true;
089        }
090        
091        public boolean isReal() {
092                return isReal;
093        }
094        
095        /**
096         * Returns a vector holding the two real eigenvalues, which are both
097         * {@code NaN} if the associated matrix has no real eigenvalues.
098         * If the input matrix is symmetric, the first eigenvalue is the greater
099         * of the two.
100         * Note that the returned array is merely a reference to the internal array and 
101         * should thus be used read-only!
102         * @return as described
103         */
104        public double[] getEigenvalues() {
105                return this.eVals;
106        }
107        
108        /**
109         * Returns a 2x2 array holding the two real eigenvectors.
110         * When used in the form
111         * <pre>
112         * double[][] X = getEigenvalues(); 
113         * </pre>
114         * then {@code X[0]} and {@code X[1]} yield the 1st and 2nd eigenvector,
115         * respectively. If the associated matrix has no real eigenvalues,
116         * then {@code X[0]} and {@code X[1]} are both {@code null}.
117         * Otherwise {@code X[k][0]} and {@code X[k][1]} are the x/y components
118         * of the k-th eigenvector.
119         * The ordering of the returned eigenvectors is the same as for the
120         * eigenvalues returned by {@link #getEigenvalues()}.
121         * @return as described
122         */
123        public double[][] getEigenvectors() {
124                return this.eVecs;
125        }

129        static void checkEigen(double[][] M) {
130                System.out.format("Checking M  = \n%s\n", Matrix.toString(M));
131                Eigensolver2x2 es = new Eigensolver2x2(M);
132                double[] eVals = es.getEigenvalues();
133                double[][] eVecs = es.getEigenvectors();
134                
135                for (int i = 0; i < 2; i++) {
136//                      System.out.format("  Eigenpair %d:\n", (i+1));
137                        checkEigenPair(i+1, M, eVals[i], eVecs[i]);
138                }
139                System.out.println();
140        }
141        
142        static void checkEigenPair(int i, double[][] M, double lambda, double[] x) {
143                double[] Mx = Matrix.multiply(M, x);
144                double[] lx = Matrix.multiply(lambda, x);
145                double d = Matrix.normL2(Matrix.add(Mx, Matrix.multiply(-1, lx)));
146                System.out.format("    \u03BB_%d  = %.6f\n", i, lambda);
147                System.out.format("    x_%d = %s\n", i, Matrix.toString(x));
148//              System.out.format("    M.x = %s\n", Matrix.toString(Mx));
149//              System.out.format("    \u03BB.x = %s\n", Matrix.toString(lx));
150//              System.out.format("    error = %.6f\n", d);
151                if (d > 0.000001) {
152                        System.out.println("    **** WRONG ****");
153                }
154        }

213                }

216                                        {1, 2},
217                                        {2, 1}};
218                        checkEigen(A);

220                }
221        }
